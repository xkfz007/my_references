76 static inline int get_rl_index(const RLTable *rl, int last, int run, int level)
77 {
78  int index;
79  index = rl->index_run[last][run];
80  if (index >= rl->n)
81  return rl->n;
82  if (level > rl->max_level[last][run])
83  return rl->n;
84  return index + level - 1;
85 }
