34 static int txd_decode_frame(AVCodecContext *avctx, void *data, int *got_frame,
35  AVPacket *avpkt) {
36  GetByteContext gb;
37  TextureDSPContext dxtc;
38  AVFrame * const p = data;
39  unsigned int version, w, h, d3d_format, depth, stride, flags;
40  unsigned int y, v;
41  uint8_t *ptr;
42  uint32_t *pal;
43  int i, j;
44  int ret;
45 
46  ff_texturedsp_init(&dxtc);
47 
48  bytestream2_init(&gb, avpkt->data, avpkt->size);
49  version = bytestream2_get_le32(&gb);
50  bytestream2_skip(&gb, 72);
51  d3d_format = bytestream2_get_le32(&gb);
52  w = bytestream2_get_le16(&gb);
53  h = bytestream2_get_le16(&gb);
54  depth = bytestream2_get_byte(&gb);
55  bytestream2_skip(&gb, 2);
56  flags = bytestream2_get_byte(&gb);
57 
58  if (version < 8 || version > 9) {
59  av_log(avctx, AV_LOG_ERROR, "texture data version %i is unsupported\n",
60  version);
61  return AVERROR_PATCHWELCOME;
62  }
63 
64  if (depth == 8) {
65  avctx->pix_fmt = AV_PIX_FMT_PAL8;
66  } else if (depth == 16 || depth == 32) {
67  avctx->pix_fmt = AV_PIX_FMT_RGBA;
68  } else {
69  av_log(avctx, AV_LOG_ERROR, "depth of %i is unsupported\n", depth);
70  return AVERROR_PATCHWELCOME;
71  }
72 
73  if ((ret = ff_set_dimensions(avctx, w, h)) < 0)
74  return ret;
75 
76  avctx->coded_width = FFALIGN(w, 4);
77  avctx->coded_height = FFALIGN(h, 4);
78 
79  if ((ret = ff_get_buffer(avctx, p, 0)) < 0)
80  return ret;
81 
82  p->pict_type = AV_PICTURE_TYPE_I;
83 
84  ptr = p->data[0];
85  stride = p->linesize[0];
86 
87  if (depth == 8) {
88  pal = (uint32_t *) p->data[1];
89  for (y = 0; y < 256; y++) {
90  v = bytestream2_get_be32(&gb);
91  pal[y] = (v >> 8) + (v << 24);
92  }
93  if (bytestream2_get_bytes_left(&gb) < w * h)
94  return AVERROR_INVALIDDATA;
95  bytestream2_skip(&gb, 4);
96  for (y=0; y<h; y++) {
97  bytestream2_get_buffer(&gb, ptr, w);
98  ptr += stride;
99  }
100  } else if (depth == 16) {
101  bytestream2_skip(&gb, 4);
102  switch (d3d_format) {
103  case 0:
104  if (!(flags & 1))
105  goto unsupported;
106  case TXD_DXT1:
107  if (bytestream2_get_bytes_left(&gb) < AV_CEIL_RSHIFT(w, 2) * AV_CEIL_RSHIFT(h, 2) * 8)
108  return AVERROR_INVALIDDATA;
109  for (j = 0; j < avctx->height; j += 4) {
110  for (i = 0; i < avctx->width; i += 4) {
111  uint8_t *p = ptr + i * 4 + j * stride;
112  int step = dxtc.dxt1_block(p, stride, gb.buffer);
113  bytestream2_skip(&gb, step);
114  }
115  }
116  break;
117  case TXD_DXT3:
118  if (bytestream2_get_bytes_left(&gb) < AV_CEIL_RSHIFT(w, 2) * AV_CEIL_RSHIFT(h, 2) * 16)
119  return AVERROR_INVALIDDATA;
120  for (j = 0; j < avctx->height; j += 4) {
121  for (i = 0; i < avctx->width; i += 4) {
122  uint8_t *p = ptr + i * 4 + j * stride;
123  int step = dxtc.dxt3_block(p, stride, gb.buffer);
124  bytestream2_skip(&gb, step);
125  }
126  }
127  break;
128  default:
129  goto unsupported;
130  }
131  } else if (depth == 32) {
132  switch (d3d_format) {
133  case 0x15:
134  case 0x16:
135  if (bytestream2_get_bytes_left(&gb) < h * w * 4)
136  return AVERROR_INVALIDDATA;
137  for (y=0; y<h; y++) {
138  bytestream2_get_buffer(&gb, ptr, w * 4);
139  ptr += stride;
140  }
141  break;
142  default:
143  goto unsupported;
144  }
145  }
146 
147  *got_frame = 1;
148 
149  return avpkt->size;
150 
151 unsupported:
152  av_log(avctx, AV_LOG_ERROR, "unsupported d3d format (%08x)\n", d3d_format);
153  return AVERROR_PATCHWELCOME;
154 }
