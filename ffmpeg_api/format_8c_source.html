98 AVOutputFormat *av_guess_format(const char *short_name, const char *filename,
99  const char *mime_type)
100 {
101  AVOutputFormat *fmt = NULL, *fmt_found;
102  int score_max, score;
103 
104  /* specific test for image sequences */
105 #if CONFIG_IMAGE2_MUXER
106  if (!short_name && filename &&
107  av_filename_number_test(filename) &&
108  ff_guess_image2_codec(filename) != AV_CODEC_ID_NONE) {
109  return av_guess_format("image2", NULL, NULL);
110  }
111 #endif
112  /* Find the proper file type. */
113  fmt_found = NULL;
114  score_max = 0;
115  while ((fmt = av_oformat_next(fmt))) {
116  score = 0;
117  //+: check format name
118  if (fmt->name && short_name && av_match_name(short_name, fmt->name))
119  score += 100;
120  if (fmt->mime_type && mime_type && !strcmp(fmt->mime_type, mime_type))
121  score += 10;
122  //+: check extension
123  if (filename && fmt->extensions &&
124  av_match_ext(filename, fmt->extensions)) {
125  score += 5;
126  }
127  if (score > score_max) {
128  score_max = score;
129  fmt_found = fmt;
130  }
131  }
132  return fmt_found;
133 }

175 AVInputFormat *av_probe_input_format3(AVProbeData *pd, int is_opened,
176  int *score_ret)
177 {
178  AVProbeData lpd = *pd;
179  AVInputFormat *fmt1 = NULL, *fmt;
180  int score, score_max = 0;
181  const static uint8_t zerobuffer[AVPROBE_PADDING_SIZE];
182  enum nodat {
183  NO_ID3,
184  ID3_ALMOST_GREATER_PROBE,
185  ID3_GREATER_PROBE,
186  ID3_GREATER_MAX_PROBE,
187  } nodat = NO_ID3;
188 
189  if (!lpd.buf)
190  lpd.buf = (unsigned char *) zerobuffer;
191 
192  //+: check if there is id3v2 header
193  if (lpd.buf_size > 10 && ff_id3v2_match(lpd.buf, ID3v2_DEFAULT_MAGIC)) {
194  int id3len = ff_id3v2_tag_len(lpd.buf);
195  if (lpd.buf_size > id3len + 16) {
196  if (lpd.buf_size < 2LL*id3len + 16)
197  nodat = ID3_ALMOST_GREATER_PROBE;
198  lpd.buf += id3len;
199  lpd.buf_size -= id3len;
200  } else if (id3len >= PROBE_BUF_MAX) {
201  nodat = ID3_GREATER_MAX_PROBE;
202  } else
203  nodat = ID3_GREATER_PROBE;
204  }
205 
206  //+: traverse the format list to find the format with max score
207  fmt = NULL;
208  while ((fmt1 = av_iformat_next(fmt1))) {
209  if (!is_opened == !(fmt1->flags & AVFMT_NOFILE) && strcmp(fmt1->name, "image2"))
210  continue;
211  score = 0;
212  if (fmt1->read_probe) {
213  score = fmt1->read_probe(&lpd);
214  if (score)
215  av_log(NULL, AV_LOG_TRACE, "Probing %s score:%d size:%d\n", fmt1->name, score, lpd.buf_size);
216  //+: check file extension
217  if (fmt1->extensions && av_match_ext(lpd.filename, fmt1->extensions)) {
218  switch (nodat) {
219  case NO_ID3:
220  score = FFMAX(score, 1);
221  break;
222  case ID3_GREATER_PROBE:
223  case ID3_ALMOST_GREATER_PROBE:
224  score = FFMAX(score, AVPROBE_SCORE_EXTENSION / 2 - 1);
225  break;
226  case ID3_GREATER_MAX_PROBE:
227  score = FFMAX(score, AVPROBE_SCORE_EXTENSION);
228  break;
229  }
230  }
231  } else if (fmt1->extensions) {//+: just use extension
232  if (av_match_ext(lpd.filename, fmt1->extensions))
233  score = AVPROBE_SCORE_EXTENSION;
234  }
235  if (av_match_name(lpd.mime_type, fmt1->mime_type)) {
236  if (AVPROBE_SCORE_MIME > score) {
237  av_log(NULL, AV_LOG_DEBUG, "Probing %s score:%d increased to %d due to MIME type\n", fmt1->name, score, AVPROBE_SCORE_MIME);
238  score = AVPROBE_SCORE_MIME;
239  }
240  }
241  if (score > score_max) {
242  score_max = score;
243  fmt = fmt1;
244  } else if (score == score_max)
245  fmt = NULL;
246  }
247  if (nodat == ID3_GREATER_PROBE)
248  score_max = FFMIN(AVPROBE_SCORE_EXTENSION / 2 - 1, score_max);
249  *score_ret = score_max;
250 
251  return fmt;
252 }

270 /*+:
271 * Probe a bytestream to determine the input format. Each time a probe returns
272 * with a score that is too low, the probe buffer size is increased and another
273 * attempt is made. When the maximum probe size is reached, the input format
274 * with the highest score is returned.
275 */
276 int av_probe_input_buffer2(AVIOContext *pb, AVInputFormat **fmt,
277  const char *filename, void *logctx,
278  unsigned int offset, unsigned int max_probe_size)
279 {
280  AVProbeData pd = { filename ? filename : "" };//+: filename will be converted to the first element of AVProbeData
281  uint8_t *buf = NULL;
282  int ret = 0, probe_size, buf_offset = 0;
283  int score = 0;
284  int ret2;
285 
286  if (!max_probe_size)
287  max_probe_size = PROBE_BUF_MAX;
288  else if (max_probe_size < PROBE_BUF_MIN) {
289  av_log(logctx, AV_LOG_ERROR,
290  "Specified probe size value %u cannot be < %u\n", max_probe_size, PROBE_BUF_MIN);
291  return AVERROR(EINVAL);
292  }
293 
294  if (offset >= max_probe_size)
295  return AVERROR(EINVAL);
296 
297  if (pb->av_class) {
298  uint8_t *mime_type_opt = NULL;
299  char *semi;
300  av_opt_get(pb, "mime_type", AV_OPT_SEARCH_CHILDREN, &mime_type_opt);
301  pd.mime_type = (const char *)mime_type_opt;
302  semi = pd.mime_type ? strchr(pd.mime_type, ';') : NULL;
303  if (semi) {
304  *semi = '\0';
305  }
306  }
307 #if 0
308  if (!*fmt && pb->av_class && av_opt_get(pb, "mime_type", AV_OPT_SEARCH_CHILDREN, &mime_type) >= 0 && mime_type) {
309  if (!av_strcasecmp(mime_type, "audio/aacp")) {
310  *fmt = av_find_input_format("aac");
311  }
312  av_freep(&mime_type);
313  }
314 #endif
315 
316  for (probe_size = PROBE_BUF_MIN; probe_size <= max_probe_size && !*fmt;
317  probe_size = FFMIN(probe_size << 1,
318  FFMAX(max_probe_size, probe_size + 1))) {
319  score = probe_size < max_probe_size ? AVPROBE_SCORE_RETRY : 0;
320 
321  /* Read probe data. */
322  if ((ret = av_reallocp(&buf, probe_size + AVPROBE_PADDING_SIZE)) < 0)
323  goto fail;
324  if ((ret = avio_read(pb, buf + buf_offset,
325  probe_size - buf_offset)) < 0) {
326  /* Fail if error was not end of file, otherwise, lower score. */
327  if (ret != AVERROR_EOF)
328  goto fail;
329 
330  score = 0;
331  ret = 0; /* error was end of file, nothing read */
332  }
333  buf_offset += ret;
334  if (buf_offset < offset)
335  continue;
336  pd.buf_size = buf_offset - offset;
337  pd.buf = &buf[offset];
338 
339  memset(pd.buf + pd.buf_size, 0, AVPROBE_PADDING_SIZE);
340 
341  /* Guess file format. */
342  *fmt = av_probe_input_format2(&pd, 1, &score);
343  if (*fmt) {
344  /* This can only be true in the last iteration. */
345  if (score <= AVPROBE_SCORE_RETRY) {
346  av_log(logctx, AV_LOG_WARNING,
347  "Format %s detected only with low score of %d, "
348  "misdetection possible!\n", (*fmt)->name, score);
349  } else
350  av_log(logctx, AV_LOG_DEBUG,
351  "Format %s probed with size=%d and score=%d\n",
352  (*fmt)->name, probe_size, score);
353 #if 0
354  FILE *f = fopen("probestat.tmp", "ab");
355  fprintf(f, "probe_size:%d format:%s score:%d filename:%s\n", probe_size, (*fmt)->name, score, filename);
356  fclose(f);
357 #endif
358  }
359  }
360 
361  if (!*fmt)
362  ret = AVERROR_INVALIDDATA;
363 
364 fail:
365  /* Rewind. Reuse probe buffer to avoid seeking. */
366  ret2 = ffio_rewind_with_probe_data(pb, &buf, buf_offset);
367  if (ret >= 0)
368  ret = ret2;
369 
370  av_freep(&pd.mime_type);
371  return ret < 0 ? ret : score;
372 }
