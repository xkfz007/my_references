26 /* Returns true if data identifier matches a teletext stream according to EN
27  * 301 775 section 4.4.2 */
28 static av_always_inline int ff_data_identifier_is_teletext(int data_identifier)
29 {
30  return (data_identifier >= 0x10 && data_identifier <= 0x1F ||
31  data_identifier >= 0x99 && data_identifier <= 0x9B);
32 }
33 
34 /* Returns true if data unit id matches EBU teletext data according to
35  * EN 301 775 section 4.4.2 */
36 static av_always_inline int ff_data_unit_id_is_teletext(int data_unit_id)
37 {
38  return (data_unit_id == 0x02 || data_unit_id == 0x03);
39 }
