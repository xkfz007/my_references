139 static int request_frame(AVFilterLink *outlink)
140 {
141  TestSourceContext *test = outlink->src->priv;
142  AVFrame *frame;
143 
144  if (test->duration >= 0 &&
145  av_rescale_q(test->pts, test->time_base, AV_TIME_BASE_Q) >= test->duration)
146  return AVERROR_EOF;
147 
148  if (test->draw_once) {
149  if (test->draw_once_reset) {
150  av_frame_free(&test->picref);
151  test->draw_once_reset = 0;
152  }
153  if (!test->picref) {
154  test->picref =
155  ff_get_video_buffer(outlink, test->w, test->h);
156  if (!test->picref)
157  return AVERROR(ENOMEM);
158  test->fill_picture_fn(outlink->src, test->picref);
159  }
160  frame = av_frame_clone(test->picref);
161  } else
162  frame = ff_get_video_buffer(outlink, test->w, test->h);
163 
164  if (!frame)
165  return AVERROR(ENOMEM);
166  frame->pts = test->pts;
167  frame->key_frame = 1;
168  frame->interlaced_frame = 0;
169  frame->pict_type = AV_PICTURE_TYPE_I;
170  frame->sample_aspect_ratio = test->sar;
171  if (!test->draw_once)
172  test->fill_picture_fn(outlink->src, frame);
173 
174  test->pts++;
175  test->nb_frame++;
176 
177  return ff_filter_frame(outlink, frame);
178 }

289 static void haldclutsrc_fill_picture(AVFilterContext *ctx, AVFrame *frame)
290 {
291  int i, j, k, x = 0, y = 0, is16bit = 0, step;
292  uint32_t alpha = 0;
293  const TestSourceContext *hc = ctx->priv;
294  int level = hc->level;
295  float scale;
296  const int w = frame->width;
297  const int h = frame->height;
298  const uint8_t *data = frame->data[0];
299  const int linesize = frame->linesize[0];
300  const AVPixFmtDescriptor *desc = av_pix_fmt_desc_get(frame->format);
301  uint8_t rgba_map[4];
302 
303  av_assert0(w == h && w == level*level*level);
304 
305  ff_fill_rgba_map(rgba_map, frame->format);
306 
307  switch (frame->format) {
308  case AV_PIX_FMT_RGB48:
309  case AV_PIX_FMT_BGR48:
310  case AV_PIX_FMT_RGBA64:
311  case AV_PIX_FMT_BGRA64:
312  is16bit = 1;
313  alpha = 0xffff;
314  break;
315  case AV_PIX_FMT_RGBA:
316  case AV_PIX_FMT_BGRA:
317  case AV_PIX_FMT_ARGB:
318  case AV_PIX_FMT_ABGR:
319  alpha = 0xff;
320  break;
321  }
322 
323  step = av_get_padded_bits_per_pixel(desc) >> (3 + is16bit);
324  scale = ((float)(1 << (8*(is16bit+1))) - 1) / (level*level - 1);
325 
326 #define LOAD_CLUT(nbits) do { \
327  uint##nbits##_t *dst = ((uint##nbits##_t *)(data + y*linesize)) + x*step; \
328  dst[rgba_map[0]] = av_clip_uint##nbits(i * scale); \
329  dst[rgba_map[1]] = av_clip_uint##nbits(j * scale); \
330  dst[rgba_map[2]] = av_clip_uint##nbits(k * scale); \
331  if (step == 4) \
332  dst[rgba_map[3]] = alpha; \
333 } while (0)
334 
335  level *= level;
336  for (k = 0; k < level; k++) {
337  for (j = 0; j < level; j++) {
338  for (i = 0; i < level; i++) {
339  if (!is16bit)
340  LOAD_CLUT(8);
341  else
342  LOAD_CLUT(16);
343  if (++x == w) {
344  x = 0;
345  y++;
346  }
347  }
348  }
349  }
350 }

491  int segment_width)
492 {

494 #define MID_HBAR 2
495 #define BOT_HBAR 4

501  int x, y, w, h;

504  { 1, 6, 5, 1 }, /* MID_HBAR */

513  /* 1 */ RIGHT_TOP_VBAR|RIGHT_BOT_VBAR,

516  /* 4 */ MID_HBAR |LEFT_TOP_VBAR |RIGHT_TOP_VBAR|RIGHT_BOT_VBAR,
517  /* 5 */ TOP_HBAR|BOT_HBAR|MID_HBAR|LEFT_TOP_VBAR |RIGHT_BOT_VBAR,

519  /* 7 */ TOP_HBAR |RIGHT_TOP_VBAR|RIGHT_BOT_VBAR,

522  };
523  unsigned mask = masks[digit];

525

530  segments[i].x, segments[i].y, segments[i].w, segments[i].h);

535 static void test_fill_picture(AVFilterContext *ctx, AVFrame *frame)
536 {
537  TestSourceContext *test = ctx->priv;
538  uint8_t *p, *p0;
539  int x, y;
540  int color, color_rest;
541  int icolor;
542  int radius;
543  int quad0, quad;
544  int dquad_x, dquad_y;
545  int grad, dgrad, rgrad, drgrad;
546  int seg_size;
547  int second;
548  int i;
549  uint8_t *data = frame->data[0];
550  int width = frame->width;
551  int height = frame->height;
552 
553 #if DIGIT_ONLY_OPTION
554  if(!test->digits_only){
555 #endif
556  /* draw colored bars and circle */
557  radius = (width + height) / 4;
558  quad0 = width * width / 4 + height * height / 4 - radius * radius;
559  dquad_y = 1 - height;
560  p0 = data;
561  for (y = 0; y < height; y++) {
562  p = p0;
563  color = 0;
564  color_rest = 0;
565  quad = quad0;
566  dquad_x = 1 - width;
567  for (x = 0; x < width; x++) {
568  icolor = color;
569  if (quad < 0)
570  icolor ^= 7;
571  quad += dquad_x;
572  dquad_x += 2;
573  *(p++) = icolor & 1 ? 255 : 0;
574  *(p++) = icolor & 2 ? 255 : 0;
575  *(p++) = icolor & 4 ? 255 : 0;
576  color_rest += 8;
577  if (color_rest >= width) {
578  color_rest -= width;
579  color++;
580  }
581  }
582  quad0 += dquad_y;
583  dquad_y += 2;
584  p0 += frame->linesize[0];
585  }
586 
587  /* draw sliding color line */
588  p0 = p = data + frame->linesize[0] * (height * 3/4);
589  grad = (256 * test->nb_frame * test->time_base.num / test->time_base.den) %
590  GRADIENT_SIZE;
591  rgrad = 0;
592  dgrad = GRADIENT_SIZE / width;
593  drgrad = GRADIENT_SIZE % width;
594  for (x = 0; x < width; x++) {
595  *(p++) =
596  grad < 256 || grad >= 5 * 256 ? 255 :
597  grad >= 2 * 256 && grad < 4 * 256 ? 0 :
598  grad < 2 * 256 ? 2 * 256 - 1 - grad : grad - 4 * 256;
599  *(p++) =
600  grad >= 4 * 256 ? 0 :
601  grad >= 1 * 256 && grad < 3 * 256 ? 255 :
602  grad < 1 * 256 ? grad : 4 * 256 - 1 - grad;
603  *(p++) =
604  grad < 2 * 256 ? 0 :
605  grad >= 3 * 256 && grad < 5 * 256 ? 255 :
606  grad < 3 * 256 ? grad - 2 * 256 : 6 * 256 - 1 - grad;
607  grad += dgrad;
608  rgrad += drgrad;
