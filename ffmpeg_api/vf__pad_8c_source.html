102 static int config_input(AVFilterLink *inlink)
103 {
104  AVFilterContext *ctx = inlink->dst;
105  PadContext *s = ctx->priv;
106  int ret;
107  double var_values[VARS_NB], res;
108  char *expr;
109 
110  ff_draw_init(&s->draw, inlink->format, 0);
111  ff_draw_color(&s->draw, &s->color, s->rgba_color);
112 
113  var_values[VAR_IN_W] = var_values[VAR_IW] = inlink->w;
114  var_values[VAR_IN_H] = var_values[VAR_IH] = inlink->h;
115  var_values[VAR_OUT_W] = var_values[VAR_OW] = NAN;
116  var_values[VAR_OUT_H] = var_values[VAR_OH] = NAN;
117  var_values[VAR_A] = (double) inlink->w / inlink->h;
118  var_values[VAR_SAR] = inlink->sample_aspect_ratio.num ?
119  (double) inlink->sample_aspect_ratio.num / inlink->sample_aspect_ratio.den : 1;
120  var_values[VAR_DAR] = var_values[VAR_A] * var_values[VAR_SAR];
121  var_values[VAR_HSUB] = 1 << s->draw.hsub_max;
122  var_values[VAR_VSUB] = 1 << s->draw.vsub_max;
123 
124  /* evaluate width and height */
125  av_expr_parse_and_eval(&res, (expr = s->w_expr),
126  var_names, var_values,
127  NULL, NULL, NULL, NULL, NULL, 0, ctx);
128  s->w = var_values[VAR_OUT_W] = var_values[VAR_OW] = res;
129  if ((ret = av_expr_parse_and_eval(&res, (expr = s->h_expr),
130  var_names, var_values,
131  NULL, NULL, NULL, NULL, NULL, 0, ctx)) < 0)
132  goto eval_fail;
133  s->h = var_values[VAR_OUT_H] = var_values[VAR_OH] = res;
134  if (!s->h)
135  var_values[VAR_OUT_H] = var_values[VAR_OH] = s->h = inlink->h;
136 
137  /* evaluate the width again, as it may depend on the evaluated output height */
138  if ((ret = av_expr_parse_and_eval(&res, (expr = s->w_expr),
139  var_names, var_values,
140  NULL, NULL, NULL, NULL, NULL, 0, ctx)) < 0)
141  goto eval_fail;
142  s->w = var_values[VAR_OUT_W] = var_values[VAR_OW] = res;
143  if (!s->w)
144  var_values[VAR_OUT_W] = var_values[VAR_OW] = s->w = inlink->w;
145 
146  /* evaluate x and y */
147  av_expr_parse_and_eval(&res, (expr = s->x_expr),
148  var_names, var_values,
149  NULL, NULL, NULL, NULL, NULL, 0, ctx);
150  s->x = var_values[VAR_X] = res;
151  if ((ret = av_expr_parse_and_eval(&res, (expr = s->y_expr),
152  var_names, var_values,
153  NULL, NULL, NULL, NULL, NULL, 0, ctx)) < 0)
154  goto eval_fail;
155  s->y = var_values[VAR_Y] = res;
156  /* evaluate x again, as it may depend on the evaluated y value */
157  if ((ret = av_expr_parse_and_eval(&res, (expr = s->x_expr),
158  var_names, var_values,
159  NULL, NULL, NULL, NULL, NULL, 0, ctx)) < 0)
160  goto eval_fail;
161  s->x = var_values[VAR_X] = res;
162 
163  /* sanity check params */
164  if (s->w < 0 || s->h < 0 || s->x < 0 || s->y < 0) {
165  av_log(ctx, AV_LOG_ERROR, "Negative values are not acceptable.\n");
166  return AVERROR(EINVAL);
167  }
168 
169  s->w = ff_draw_round_to_sub(&s->draw, 0, -1, s->w);
170  s->h = ff_draw_round_to_sub(&s->draw, 1, -1, s->h);
171  s->x = ff_draw_round_to_sub(&s->draw, 0, -1, s->x);
172  s->y = ff_draw_round_to_sub(&s->draw, 1, -1, s->y);
173  s->in_w = ff_draw_round_to_sub(&s->draw, 0, -1, inlink->w);
174  s->in_h = ff_draw_round_to_sub(&s->draw, 1, -1, inlink->h);
175  s->inlink_w = inlink->w;
176  s->inlink_h = inlink->h;
177 
178  av_log(ctx, AV_LOG_VERBOSE, "w:%d h:%d -> w:%d h:%d x:%d y:%d color:0x%02X%02X%02X%02X\n",
179  inlink->w, inlink->h, s->w, s->h, s->x, s->y,
180  s->rgba_color[0], s->rgba_color[1], s->rgba_color[2], s->rgba_color[3]);
181 
182  if (s->x < 0 || s->y < 0 ||
183  s->w <= 0 || s->h <= 0 ||
184  (unsigned)s->x + (unsigned)inlink->w > s->w ||
185  (unsigned)s->y + (unsigned)inlink->h > s->h) {
186  av_log(ctx, AV_LOG_ERROR,
187  "Input area %d:%d:%d:%d not within the padded area 0:0:%d:%d or zero-sized\n",
188  s->x, s->y, s->x + inlink->w, s->y + inlink->h, s->w, s->h);
189  return AVERROR(EINVAL);
190  }
191 
192  return 0;
193 
194 eval_fail:
195  av_log(NULL, AV_LOG_ERROR,
196  "Error when evaluating the expression '%s'\n", expr);
197  return ret;
198 
199 }

210 static AVFrame *get_video_buffer(AVFilterLink *inlink, int w, int h)
211 {
212  PadContext *s = inlink->dst->priv;
213 
214  AVFrame *frame = ff_get_video_buffer(inlink->dst->outputs[0],
215  w + (s->w - s->in_w),
216  h + (s->h - s->in_h) + (s->x > 0));
217  int plane;
218 
219  if (!frame)
220  return NULL;
221 
222  frame->width = w;
223  frame->height = h;
224 
225  for (plane = 0; plane < 4 && frame->data[plane] && frame->linesize[plane]; plane++) {
226  int hsub = s->draw.hsub[plane];
227  int vsub = s->draw.vsub[plane];
228  frame->data[plane] += (s->x >> hsub) * s->draw.pixelstep[plane] +
229  (s->y >> vsub) * frame->linesize[plane];
230  }
231 
232  return frame;
233 }

235 /* check whether each plane in this buffer can be padded without copying */
236 static int buffer_needs_copy(PadContext *s, AVFrame *frame, AVBufferRef *buf)
237 {
238  int planes[4] = { -1, -1, -1, -1}, *p = planes;
239  int i, j;
240 
241  /* get all planes in this buffer */
242  for (i = 0; i < FF_ARRAY_ELEMS(planes) && frame->data[i]; i++) {
243  if (av_frame_get_plane_buffer(frame, i) == buf)
244  *p++ = i;
245  }
246 
247  /* for each plane in this buffer, check that it can be padded without
248  * going over buffer bounds or other planes */
249  for (i = 0; i < FF_ARRAY_ELEMS(planes) && planes[i] >= 0; i++) {
250  int hsub = s->draw.hsub[planes[i]];
251  int vsub = s->draw.vsub[planes[i]];
252 
253  uint8_t *start = frame->data[planes[i]];
254  uint8_t *end = start + (frame->height >> vsub) *
255  frame->linesize[planes[i]];
256 
257  /* amount of free space needed before the start and after the end
258  * of the plane */
259  ptrdiff_t req_start = (s->x >> hsub) * s->draw.pixelstep[planes[i]] +
260  (s->y >> vsub) * frame->linesize[planes[i]];
261  ptrdiff_t req_end = ((s->w - s->x - frame->width) >> hsub) *
262  s->draw.pixelstep[planes[i]] +
263  ((s->h - s->y - frame->height) >> vsub) * frame->linesize[planes[i]];
264 
265  if (frame->linesize[planes[i]] < (s->w >> hsub) * s->draw.pixelstep[planes[i]])
266  return 1;
267  if (start - buf->data < req_start ||
268  (buf->data + buf->size) - end < req_end)
269  return 1;
270 
271  for (j = 0; j < FF_ARRAY_ELEMS(planes) && planes[j] >= 0; j++) {
272  int vsub1 = s->draw.vsub[planes[j]];
273  uint8_t *start1 = frame->data[planes[j]];
274  uint8_t *end1 = start1 + (frame->height >> vsub1) *
275  frame->linesize[planes[j]];
276  if (i == j)
277  continue;
278 
279  if (FFSIGN(start - end1) != FFSIGN(start - end1 - req_start) ||
280  FFSIGN(end - start1) != FFSIGN(end - start1 + req_end))
281  return 1;
282  }
283  }
284 
285  return 0;
286 }

288 static int frame_needs_copy(PadContext *s, AVFrame *frame)
289 {
290  int i;
291 
292  if (!av_frame_is_writable(frame))
293  return 1;
294 
295  for (i = 0; i < 4 && frame->buf[i]; i++)
296  if (buffer_needs_copy(s, frame, frame->buf[i]))
297  return 1;
298  return 0;
299 }

301 static int filter_frame(AVFilterLink *inlink, AVFrame *in)
302 {
303  PadContext *s = inlink->dst->priv;
304  AVFilterLink *outlink = inlink->dst->outputs[0];
305  AVFrame *out;
306  int needs_copy;
307  if(s->eval_mode == EVAL_MODE_FRAME && (
308  in->width != s->inlink_w
309  || in->height != s->inlink_h
310  || in->format != outlink->format
311  || in->sample_aspect_ratio.den != outlink->sample_aspect_ratio.den || in->sample_aspect_ratio.num != outlink->sample_aspect_ratio.num)) {
312  int ret;
313 
314  inlink->dst->inputs[0]->format = in->format;
315  inlink->dst->inputs[0]->w = in->width;
316  inlink->dst->inputs[0]->h = in->height;
317 
318  inlink->dst->inputs[0]->sample_aspect_ratio.den = in->sample_aspect_ratio.den;
319  inlink->dst->inputs[0]->sample_aspect_ratio.num = in->sample_aspect_ratio.num;
320 
321 
322  if ((ret = config_input(inlink)) < 0) {
323  s->inlink_w = -1;
324  return ret;
325  }
326  if ((ret = config_output(outlink)) < 0) {
327  s->inlink_w = -1;
328  return ret;
329  }
330  }
331 
332  needs_copy = frame_needs_copy(s, in);
333 
334  if (needs_copy) {
335  av_log(inlink->dst, AV_LOG_DEBUG, "Direct padding impossible allocating new frame\n");
336  out = ff_get_video_buffer(inlink->dst->outputs[0],
337  FFMAX(inlink->w, s->w),
338  FFMAX(inlink->h, s->h));
339  if (!out) {
340  av_frame_free(&in);
341  return AVERROR(ENOMEM);
342  }
343 
344  av_frame_copy_props(out, in);
345  } else {
346  int i;
347 
348  out = in;
349  for (i = 0; i < 4 && out->data[i] && out->linesize[i]; i++) {
350  int hsub = s->draw.hsub[i];
351  int vsub = s->draw.vsub[i];
352  out->data[i] -= (s->x >> hsub) * s->draw.pixelstep[i] +
353  (s->y >> vsub) * out->linesize[i];
354  }
355  }
356 
357  /* top bar */
358  if (s->y) {
359  ff_fill_rectangle(&s->draw, &s->color,
360  out->data, out->linesize,
361  0, 0, s->w, s->y);
362  }
363 
364  /* bottom bar */
365  if (s->h > s->y + s->in_h) {
366  ff_fill_rectangle(&s->draw, &s->color,
367  out->data, out->linesize,
368  0, s->y + s->in_h, s->w, s->h - s->y - s->in_h);
369  }
370 
371  /* left border */
372  ff_fill_rectangle(&s->draw, &s->color, out->data, out->linesize,
373  0, s->y, s->x, in->height);
374 
375  if (needs_copy) {
376  ff_copy_rectangle2(&s->draw,
377  out->data, out->linesize, in->data, in->linesize,
378  s->x, s->y, 0, 0, in->width, in->height);
379  }
380 
381  /* right border */
382  ff_fill_rectangle(&s->draw, &s->color, out->data, out->linesize,
383  s->x + s->in_w, s->y, s->w - s->x - s->in_w,
384  in->height);
385 
386  out->width = s->w;
387  out->height = s->h;
388 
389  if (in != out)
390  av_frame_free(&in);
391  return ff_filter_frame(inlink->dst->outputs[0], out);
392 }
