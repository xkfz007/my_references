57 static inline void lag_rac_refill(lag_rac *l)
58 {
59  while (l->range <= 0x800000) {
60  l->low <<= 8;
61  l->range <<= 8;
62  l->low |= 0xff & (AV_RB16(l->bytestream) >> 1);
63  if (l->bytestream < l->bytestream_end)
64  l->bytestream++;
65  }
66 }

68 /**
69  * Decode a single byte from the compressed plane described by *l.
70  * @param l pointer to lag_rac for the current plane
71  * @return next byte of decoded data
72  */
73 static inline uint8_t lag_get_rac(lag_rac *l)
74 {
75  unsigned range_scaled, low_scaled;
76  int val;
77 
78  lag_rac_refill(l);
79 
80  range_scaled = l->range >> l->scale;
81 
82  if (l->low < range_scaled * l->prob[255]) {
83  /* val = 0 is frequent enough to deserve a shortcut */
84  if (l->low < range_scaled * l->prob[1]) {
85  val = 0;
86  } else {
87  low_scaled = l->low / (range_scaled<<(l->hash_shift));
88 
89  val = l->range_hash[low_scaled];
90  while (l->low >= range_scaled * l->prob[val + 1])
91  val++;
92  }
93 
94  l->range = range_scaled * (l->prob[val + 1] - l->prob[val]);
95  } else {
96  val = 255;
97  l->range -= range_scaled * l->prob[255];
98  }
99 
100  if (!l->range)
101  l->range = 0x80;
102 
103  l->low -= range_scaled * l->prob[val];
104 
105  return val;
106 }
