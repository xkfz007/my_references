40 int ff_lzf_uncompress(GetByteContext *gb, uint8_t **buf, int64_t *size)
41 {
42  int ret = 0;
43  uint8_t *p = *buf;
44  int64_t len = 0;
45 
46  while (bytestream2_get_bytes_left(gb) > 2) {
47  uint8_t s = bytestream2_get_byte(gb);
48 
49  if (s < LZF_LITERAL_MAX) {
50  s++;
51  if (s > *size - len) {
52  *size += *size /2;
53  ret = av_reallocp(buf, *size);
54  if (ret < 0)
55  return ret;
56  p = *buf + len;
57  }
58 
59  bytestream2_get_buffer(gb, p, s);
60  p += s;
61  len += s;
62  } else {
63  int l = 2 + (s >> 5);
64  int off = ((s & 0x1f) << 8) + 1;
65 
66  if (l == LZF_LONG_BACKREF)
67  l += bytestream2_get_byte(gb);
68 
69  off += bytestream2_get_byte(gb);
70 
71  if (off > len)
72  return AVERROR_INVALIDDATA;
73 
74  if (l > *size - len) {
75  *size += *size / 2;
76  ret = av_reallocp(buf, *size);

78  return ret;
79  p = *buf + len;
80  }
81
