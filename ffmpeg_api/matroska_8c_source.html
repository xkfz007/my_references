151 int ff_mkv_stereo3d_conv(AVStream *st, MatroskaVideoStereoModeType stereo_mode)
152 {
153  AVStereo3D *stereo;
154  int ret;
155 
156  stereo = av_stereo3d_alloc();
157  if (!stereo)
158  return AVERROR(ENOMEM);
159 
160  // note: the missing breaks are intentional
161  switch (stereo_mode) {
162  case MATROSKA_VIDEO_STEREOMODE_TYPE_MONO:
163  stereo->type = AV_STEREO3D_2D;
164  break;
165  case MATROSKA_VIDEO_STEREOMODE_TYPE_RIGHT_LEFT:
166  stereo->flags |= AV_STEREO3D_FLAG_INVERT;
167  case MATROSKA_VIDEO_STEREOMODE_TYPE_LEFT_RIGHT:
168  stereo->type = AV_STEREO3D_SIDEBYSIDE;
169  break;
170  case MATROSKA_VIDEO_STEREOMODE_TYPE_BOTTOM_TOP:
171  stereo->flags |= AV_STEREO3D_FLAG_INVERT;
172  case MATROSKA_VIDEO_STEREOMODE_TYPE_TOP_BOTTOM:
173  stereo->type = AV_STEREO3D_TOPBOTTOM;
174  break;
175  case MATROSKA_VIDEO_STEREOMODE_TYPE_CHECKERBOARD_RL:
176  stereo->flags |= AV_STEREO3D_FLAG_INVERT;
177  case MATROSKA_VIDEO_STEREOMODE_TYPE_CHECKERBOARD_LR:
178  stereo->type = AV_STEREO3D_CHECKERBOARD;
179  break;
180  case MATROSKA_VIDEO_STEREOMODE_TYPE_ROW_INTERLEAVED_RL:
181  stereo->flags |= AV_STEREO3D_FLAG_INVERT;
182  case MATROSKA_VIDEO_STEREOMODE_TYPE_ROW_INTERLEAVED_LR:
183  stereo->type = AV_STEREO3D_LINES;
184  break;
185  case MATROSKA_VIDEO_STEREOMODE_TYPE_COL_INTERLEAVED_RL:
186  stereo->flags |= AV_STEREO3D_FLAG_INVERT;
187  case MATROSKA_VIDEO_STEREOMODE_TYPE_COL_INTERLEAVED_LR:
188  stereo->type = AV_STEREO3D_COLUMNS;
189  break;
190  case MATROSKA_VIDEO_STEREOMODE_TYPE_BOTH_EYES_BLOCK_RL:
191  stereo->flags |= AV_STEREO3D_FLAG_INVERT;
192  case MATROSKA_VIDEO_STEREOMODE_TYPE_BOTH_EYES_BLOCK_LR:
193  stereo->type = AV_STEREO3D_FRAMESEQUENCE;
194  break;
195  }
196 
197  ret = av_stream_add_side_data(st, AV_PKT_DATA_STEREO3D, (uint8_t *)stereo,
198  sizeof(*stereo));
199  if (ret < 0) {
200  av_freep(&stereo);
201  return ret;
202  }
203 
204  return 0;
205 }
