63 static int failing_write_packet(AVFormatContext *avf, AVPacket *pkt)
64 {
65  FailingMuxerContext *ctx = avf->priv_data;
66  int ret = 0;
67  if (!pkt) {
68  ctx->flush_count++;
69  } else {
70  FailingMuxerPacketData *data = (FailingMuxerPacketData*) pkt->data;
71 
72  if (!data->recover_after) {
73  data->ret = 0;
74  } else {
75  data->recover_after--;
76  }
77 
78  ret = data->ret;
79 
80  if (data->sleep_time) {
81  int64_t slept = 0;
82  while (slept < data->sleep_time) {
83  if (ff_check_interrupt(&avf->interrupt_callback))
84  return AVERROR_EXIT;
85  av_usleep(SLEEPTIME_10_MS);
86  slept += SLEEPTIME_10_MS;
87  }
88  }
89 
90  if (!ret) {
91  ctx->pts_written[ctx->pts_written_nr++] = pkt->pts;
92  av_packet_unref(pkt);
93  }
94  }
95  return ret;
96 }

193

355 static int run_test(const TestCase *test)
356 {
357  AVDictionary *opts = NULL;
358  AVFormatContext *oc = NULL;
359  char buffer[BUFFER_SIZE];
360  int ret, ret1;
361 
362  ret = initialize_fifo_tst_muxer_chain(&oc);
363  if (ret < 0) {
364  fprintf(stderr, "Muxer initialization failed: %s\n", av_err2str(ret));
365  goto end;
366  }
367 
368  if (test->options) {
369  ret = av_dict_parse_string(&opts, test->options, "=", ":", 0);
370  if (ret < 0) {
371  fprintf(stderr, "Failed to parse options: %s\n", av_err2str(ret));
372  goto end;
373  }
374  }
375 
376  snprintf(buffer, BUFFER_SIZE,
377  "print_deinit_summary=%d:write_header_ret=%d:write_trailer_ret=%d",
378  (int)test->print_summary_on_deinit, test->write_header_ret,
379  test->write_trailer_ret);
380  ret = av_dict_set(&opts, "format_opts", buffer, 0);
381  ret1 = av_dict_set(&opts, "fifo_format", "fail", 0);
382  if (ret < 0 || ret1 < 0) {
383  fprintf(stderr, "Failed to set options for test muxer: %s\n",
384  av_err2str(ret));
385  goto end;
386  }
387 
388  ret = test->test_func(oc, &opts, &test->pkt_data);
389 
390 end:
391  printf("%s: %s\n", test->test_name, ret < 0 ? "fail" : "ok");
392  avformat_free_context(oc);
393  av_dict_free(&opts);
394  return ret;
395 }
