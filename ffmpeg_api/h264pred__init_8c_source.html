49  const int bit_depth,
50  const int chroma_format_idc)
51 {
52  const int high_depth = bit_depth > 8;
53 
54  if (high_depth)
55  return;
56 
57  if (chroma_format_idc <= 1) {
58  h->pred8x8[VERT_PRED8x8 ] = ff_pred8x8_vert_neon;
59  h->pred8x8[HOR_PRED8x8 ] = ff_pred8x8_hor_neon;
60  if (codec_id != AV_CODEC_ID_VP7 && codec_id != AV_CODEC_ID_VP8)
61  h->pred8x8[PLANE_PRED8x8] = ff_pred8x8_plane_neon;
62  h->pred8x8[DC_128_PRED8x8 ] = ff_pred8x8_128_dc_neon;
63  if (codec_id != AV_CODEC_ID_RV40 && codec_id != AV_CODEC_ID_VP7 &&
64  codec_id != AV_CODEC_ID_VP8) {
65  h->pred8x8[DC_PRED8x8 ] = ff_pred8x8_dc_neon;
66  h->pred8x8[LEFT_DC_PRED8x8] = ff_pred8x8_left_dc_neon;
67  h->pred8x8[TOP_DC_PRED8x8 ] = ff_pred8x8_top_dc_neon;
68  h->pred8x8[ALZHEIMER_DC_L0T_PRED8x8] = ff_pred8x8_l0t_dc_neon;
69  h->pred8x8[ALZHEIMER_DC_0LT_PRED8x8] = ff_pred8x8_0lt_dc_neon;
70  h->pred8x8[ALZHEIMER_DC_L00_PRED8x8] = ff_pred8x8_l00_dc_neon;
71  h->pred8x8[ALZHEIMER_DC_0L0_PRED8x8] = ff_pred8x8_0l0_dc_neon;
72  }
73  }
74 
75  h->pred16x16[DC_PRED8x8 ] = ff_pred16x16_dc_neon;
76  h->pred16x16[VERT_PRED8x8 ] = ff_pred16x16_vert_neon;
77  h->pred16x16[HOR_PRED8x8 ] = ff_pred16x16_hor_neon;
78  h->pred16x16[LEFT_DC_PRED8x8] = ff_pred16x16_left_dc_neon;
79  h->pred16x16[TOP_DC_PRED8x8 ] = ff_pred16x16_top_dc_neon;
80  h->pred16x16[DC_128_PRED8x8 ] = ff_pred16x16_128_dc_neon;
81  if (codec_id != AV_CODEC_ID_SVQ3 && codec_id != AV_CODEC_ID_RV40 &&
82  codec_id != AV_CODEC_ID_VP7 && codec_id != AV_CODEC_ID_VP8)
83  h->pred16x16[PLANE_PRED8x8 ] = ff_pred16x16_plane_neon;
84 }
