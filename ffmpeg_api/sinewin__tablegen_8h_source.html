65 // Generate a sine window.
66 av_cold void AAC_RENAME(ff_sine_window_init)(INTFLOAT *window, int n) {
67  int i;
68  for(i = 0; i < n; i++)
69  window[i] = SIN_FIX(sinf((i + 0.5) * (M_PI / (2.0 * n))));
70 }
