111 static inline void dv_calculate_mb_xy(DVVideoContext *s,
112  DVwork_chunk *work_chunk,
113  int m, int *mb_x, int *mb_y)
114 {
115  *mb_x = work_chunk->mb_coordinates[m] & 0xff;
116  *mb_y = work_chunk->mb_coordinates[m] >> 8;
117 
118  /* We work with 720p frames split in half.
119  * The odd half-frame (chan == 2,3) is displaced :-( */
120  if (s->sys->height == 720 && !(s->buf[1] & 0x0C))
121  /* shifting the Y coordinate down by 72/2 macro blocks */
122  *mb_y -= (*mb_y > 17) ? 18 : -72;
123 }
