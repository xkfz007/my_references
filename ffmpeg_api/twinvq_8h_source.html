192 static inline float twinvq_mulawinv(float y, float clip, float mu)
193 {
194  y = av_clipf(y / clip, -1, 1);
195  return clip * FFSIGN(y) * (exp(log(1 + mu) * fabs(y)) - 1) / mu;
196 }
