53 av_cold void ff_blend_init_x86(FilterParams *param, int is_16bit)
54 {
55  int cpu_flags = av_get_cpu_flags();
56 
57  if (EXTERNAL_SSE2(cpu_flags) && param->opacity == 1 && !is_16bit) {
58  switch (param->mode) {

60  case BLEND_ADDITION128: param->blend = ff_blend_addition128_sse2; break;
