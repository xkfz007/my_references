213  memcpy(dst1, dst, sz * PIXEL_STRIDE);
214  memcpy(subcoef1, subcoef0, sz * sz * SIZEOF_COEF);

217  if (memcmp(dst0, dst1, sz * PIXEL_STRIDE) ||
