176 static inline void linecpy(void *dest, const void *src, int lines, int stride) {
177  if (stride > 0) {
178  memcpy(dest, src, lines*stride);
179  } else {
180  memcpy((uint8_t*)dest+(lines-1)*stride, (const uint8_t*)src+(lines-1)*stride, -lines*stride);
181  }
