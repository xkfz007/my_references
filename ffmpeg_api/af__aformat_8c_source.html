60 #define PARSE_FORMATS(str, type, list, add_to_list, unref_fn, get_fmt, none, desc) \
61 do { \
62  char *next, *cur = str, sep; \
63  int ret; \
64  \
65  if (str && strchr(str, ',')) { \
66  av_log(ctx, AV_LOG_WARNING, "This syntax is deprecated, use '|' to "\
67  "separate %s.\n", desc); \
68  sep = ','; \
69  } else \
70  sep = '|'; \
71  \
72  while (cur) { \
73  type fmt; \
74  next = strchr(cur, sep); \
75  if (next) \
76  *next++ = 0; \
77  \
78  if ((fmt = get_fmt(cur)) == none) { \
79  av_log(ctx, AV_LOG_ERROR, "Error parsing " desc ": %s.\n", cur);\

82  if ((ret = add_to_list(&list, fmt)) < 0) { \
83  unref_fn(&list); \
84  return ret; \
85  } \
86  \
87  cur = next; \
88  } \

98 {
99  AFormatContext *s = ctx->priv;

101  PARSE_FORMATS(s->formats_str, enum AVSampleFormat, s->formats,
102  ff_add_format, ff_formats_unref, av_get_sample_fmt, AV_SAMPLE_FMT_NONE, "sample format");
103  PARSE_FORMATS(s->sample_rates_str, int, s->sample_rates, ff_add_format, ff_formats_unref,
104  get_sample_rate, 0, "sample rate");
105  PARSE_FORMATS(s->channel_layouts_str, uint64_t, s->channel_layouts,
106  ff_add_channel_layout, ff_channel_layouts_unref, av_get_channel_layout, 0,
107  "channel layout");
108 
109  return 0;
110 }

112 static int query_formats(AVFilterContext *ctx)
113 {
114  AFormatContext *s = ctx->priv;
115  int ret;
116 
117  ret = ff_set_common_formats(ctx, s->formats ? s->formats :
118  ff_all_formats(AVMEDIA_TYPE_AUDIO));
119  if (ret < 0)
120  return ret;
121  ret = ff_set_common_samplerates(ctx, s->sample_rates ? s->sample_rates :
122  ff_all_samplerates());
123  if (ret < 0)
124  return ret;
125  return ff_set_common_channel_layouts(ctx, s->channel_layouts ? s->channel_layouts :
126  ff_all_channel_counts());
127 }
