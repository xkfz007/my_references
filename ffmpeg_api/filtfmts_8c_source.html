34 static void print_formats(AVFilterContext *filter_ctx)
35 {
36  int i, j;

38 #define PRINT_FMTS(inout, outin, INOUT) \
39  for (i = 0; i < filter_ctx->nb_##inout##puts; i++) { \
40  if (filter_ctx->inout##puts[i]->type == AVMEDIA_TYPE_VIDEO) { \
41  AVFilterFormats *fmts = \
42  filter_ctx->inout##puts[i]->outin##_formats; \
43  for (j = 0; j < fmts->nb_formats; j++) \
44  if(av_get_pix_fmt_name(fmts->formats[j])) \
45  printf(#INOUT "PUT[%d] %s: fmt:%s\n", \
46  i, avfilter_pad_get_name(filter_ctx->inout##put_pads, i), \
47  av_get_pix_fmt_name(fmts->formats[j])); \
48  } else if (filter_ctx->inout##puts[i]->type == AVMEDIA_TYPE_AUDIO) { \
49  AVFilterFormats *fmts; \
50  AVFilterChannelLayouts *layouts; \
51  \
52  fmts = filter_ctx->inout##puts[i]->outin##_formats; \
53  for (j = 0; j < fmts->nb_formats; j++) \
54  printf(#INOUT "PUT[%d] %s: fmt:%s\n", \
55  i, avfilter_pad_get_name(filter_ctx->inout##put_pads, i), \
56  av_get_sample_fmt_name(fmts->formats[j])); \
57  \
58  layouts = filter_ctx->inout##puts[i]->outin##_channel_layouts; \
59  for (j = 0; j < layouts->nb_channel_layouts; j++) { \
60  char buf[256]; \
61  av_get_channel_layout_string(buf, sizeof(buf), -1, \
62  layouts->channel_layouts[j]); \
63  printf(#INOUT "PUT[%d] %s: chlayout:%s\n", \
64  i, avfilter_pad_get_name(filter_ctx->inout##put_pads, i), buf); \
65  } \
66  } \
67  } \
68 
69  PRINT_FMTS(in, out, IN);
70  PRINT_FMTS(out, in, OUT);
71 }

83  av_log_set_level(AV_LOG_DEBUG);
84 
85  if (argc < 2) {

87  return 1;
88  }
89 
90  filter_name = argv[1];
91  if (argc > 2)
92  filter_args = argv[2];
93 
94  /* allocate graph */
95  graph_ctx = avfilter_graph_alloc();
96  if (!graph_ctx)
97  return 1;
98 
99  avfilter_register_all();
100 
101  /* get a corresponding filter and open it */
102  if (!(filter = avfilter_get_by_name(filter_name))) {
103  fprintf(stderr, "Unrecognized filter with name '%s'\n", filter_name);
104  return 1;
105  }
106 
107  /* open filter and add it to the graph */
108  if (!(filter_ctx = avfilter_graph_alloc_filter(graph_ctx, filter, filter_name))) {
109  fprintf(stderr, "Impossible to open filter with name '%s'\n",
110  filter_name);
111  return 1;
112  }
113  if (avfilter_init_str(filter_ctx, filter_args) < 0) {
114  fprintf(stderr, "Impossible to init filter '%s' with arguments '%s'\n",
115  filter_name, filter_args);
116  return 1;
117  }
118 
119  /* create a link for each of the input pads */
120  for (i = 0; i < filter_ctx->nb_inputs; i++) {
121  AVFilterLink *link = av_mallocz(sizeof(AVFilterLink));
122  if (!link) {
123  fprintf(stderr, "Unable to allocate memory for filter input link\n");
124  ret = 1;
125  goto fail;
126  }
127  link->type = avfilter_pad_get_type(filter_ctx->input_pads, i);
128  filter_ctx->inputs[i] = link;
129  }
130  for (i = 0; i < filter_ctx->nb_outputs; i++) {
131  AVFilterLink *link = av_mallocz(sizeof(AVFilterLink));
132  if (!link) {
133  fprintf(stderr, "Unable to allocate memory for filter output link\n");
134  ret = 1;
135  goto fail;
136  }
137  link->type = avfilter_pad_get_type(filter_ctx->output_pads, i);
138  filter_ctx->outputs[i] = link;
139  }
140 
141  if (filter->query_formats)
142  filter->query_formats(filter_ctx);
143  else
144  ff_default_query_formats(filter_ctx);
145 
146  print_formats(filter_ctx);
147 
148 fail:
149  avfilter_free(filter_ctx);
150  avfilter_graph_free(&graph_ctx);
151  fflush(stdout);
152  return ret;
153 }
