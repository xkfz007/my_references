36 {
37  int s = 0, d = 0, url_len = 0;
38  char c;
39  char *dest = NULL;
40 
41  if (!url)
42  return NULL;
43 
44  url_len = strlen(url) + 1;
45  dest = av_malloc(url_len);
46 
47  if (!dest)
48  return NULL;
49 
50  while (s < url_len) {
51  c = url[s++];
52 
53  if (c == '%' && s + 2 < url_len) {
54  char c2 = url[s++];
55  char c3 = url[s++];
56  if (av_isxdigit(c2) && av_isxdigit(c3)) {
57  c2 = av_tolower(c2);
58  c3 = av_tolower(c3);
59 
60  if (c2 <= '9')
61  c2 = c2 - '0';
62  else
63  c2 = c2 - 'a' + 10;
64 
65  if (c3 <= '9')
66  c3 = c3 - '0';
67  else
68  c3 = c3 - 'a' + 10;
69 
70  dest[d++] = 16 * c2 + c3;
71 
72  } else { /* %zz or something other invalid */
73  dest[d++] = c;
74  dest[d++] = c2;
75  dest[d++] = c3;
76  }
77  } else if (c == '+') {
78  dest[d++] = ' ';
79  } else {
80  dest[d++] = c;
81  }
82 
83  }
84
