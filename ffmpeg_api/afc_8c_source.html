30 static int afc_read_header(AVFormatContext *s)
31 {
32  AFCDemuxContext *c = s->priv_data;
33  AVStream *st;
34 
35  st = avformat_new_stream(s, NULL);
36  if (!st)
37  return AVERROR(ENOMEM);
38  st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
39  st->codecpar->codec_id = AV_CODEC_ID_ADPCM_AFC;
40  st->codecpar->channels = 2;
41  st->codecpar->channel_layout = AV_CH_LAYOUT_STEREO;
42 
43  if (ff_alloc_extradata(st->codecpar, 1))
44  return AVERROR(ENOMEM);
45  st->codecpar->extradata[0] = 8 * st->codecpar->channels;
46 
47  c->data_end = avio_rb32(s->pb) + 32LL;
48  st->duration = avio_rb32(s->pb);
49  st->codecpar->sample_rate = avio_rb16(s->pb);
50  avio_skip(s->pb, 22);
51  avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
52 
53  return 0;
54 }

56 static int afc_read_packet(AVFormatContext *s, AVPacket *pkt)
57 {
58  AFCDemuxContext *c = s->priv_data;
59  int64_t size;
60  int ret;
61 
62  size = FFMIN(c->data_end - avio_tell(s->pb), 18 * 128);
63  if (size <= 0)
64  return AVERROR_EOF;
65 
66  ret = av_get_packet(s->pb, pkt, size);
67  pkt->stream_index = 0;
68  return ret;
69 }
