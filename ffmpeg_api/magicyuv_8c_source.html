77 static int huff_cmp_len(const void *a, const void *b)
78 {
79  const HuffEntry *aa = a, *bb = b;
80  return (aa->len - bb->len) * 256 + aa->sym - bb->sym;
81 }

89 static int huff_build10(VLC *vlc, uint8_t *len)
90 {
91  HuffEntry he[1024];
92  uint32_t codes[1024];
93  uint8_t bits[1024];
94  uint16_t syms[1024];
95  uint32_t code;
96  int i;
97 
98  for (i = 0; i < 1024; i++) {
99  he[i].sym = 1023 - i;
100  he[i].len = len[i];
101  }
102  AV_QSORT(he, 1024, HuffEntry, huff_cmp_len10);
103 
104  code = 1;
105  for (i = 1023; i >= 0; i--) {
106  codes[i] = code >> (32 - he[i].len);
107  bits[i] = he[i].len;
108  syms[i] = he[i].sym;
109  code += 0x80000000u >> (he[i].len - 1);
110  }
111 
112  ff_free_vlc(vlc);
113  return ff_init_vlc_sparse(vlc, FFMIN(he[1023].len, 12), 1024,
114  bits, sizeof(*bits), sizeof(*bits),
115  codes, sizeof(*codes), sizeof(*codes),
116  syms, sizeof(*syms), sizeof(*syms), 0);
117 }

120 {
121  HuffEntry he[256];

129  he[i].sym = 255 - i;

149 static void magicyuv_median_pred10(uint16_t *dst, const uint16_t *src1,
150  const uint16_t *diff, intptr_t w,
151  int *left, int *left_top)
152 {
153  int i;
154  uint16_t l, lt;
155 
156  l = *left;
157  lt = *left_top;
158 
159  for (i = 0; i < w; i++) {
160  l = mid_pred(l, src1[i], (l + src1[i] - lt)) + diff[i];
161  l &= 0x3FF;
162  lt = src1[i];
163  dst[i] = l;
164  }
165 
166  *left = l;
167  *left_top = lt;
168 }

170 static int magy_decode_slice10(AVCodecContext *avctx, void *tdata,
171  int j, int threadnr)
172 {
173  MagicYUVContext *s = avctx->priv_data;
174  int interlaced = s->interlaced;
175  AVFrame *p = s->p;
176  int i, k, x;
177  GetBitContext gb;
178  uint16_t *dst;
179 
180  for (i = 0; i < s->planes; i++) {
181  int left, lefttop, top;
182  int height = AV_CEIL_RSHIFT(FFMIN(s->slice_height, avctx->coded_height - j * s->slice_height), s->vshift[i]);
183  int width = AV_CEIL_RSHIFT(avctx->coded_width, s->hshift[i]);
184  int sheight = AV_CEIL_RSHIFT(s->slice_height, s->vshift[i]);
185  ptrdiff_t fake_stride = (p->linesize[i] / 2) * (1 + interlaced);
186  ptrdiff_t stride = p->linesize[i] / 2;
187  int flags, pred;
188  int ret = init_get_bits8(&gb, s->buf + s->slices[i][j].start,
189  s->slices[i][j].size);
190 
191  if (ret < 0)
192  return ret;
193 
194  flags = get_bits(&gb, 8);
195  pred = get_bits(&gb, 8);
196 
197  dst = (uint16_t *)p->data[i] + j * sheight * stride;
198  if (flags & 1) {
199  for (k = 0; k < height; k++) {
200  for (x = 0; x < width; x++)
201  dst[x] = get_bits(&gb, 10);
202 
203  dst += stride;
204  }
205  } else {
206  for (k = 0; k < height; k++) {
207  for (x = 0; x < width; x++) {
208  int pix;
209  if (get_bits_left(&gb) <= 0)
210  return AVERROR_INVALIDDATA;
211 
212  pix = get_vlc2(&gb, s->vlc[i].table, s->vlc[i].bits, 3);
213  if (pix < 0)
214  return AVERROR_INVALIDDATA;
215 
216  dst[x] = 1023 - pix;
217  }
218  dst += stride;
219  }
220  }
221 
222  switch (pred) {
223  case LEFT:
224  dst = (uint16_t *)p->data[i] + j * sheight * stride;
225  s->llviddsp.add_hfyu_left_pred_int16(dst, dst, 1023, width, 0);
226  dst += stride;
227  if (interlaced) {
228  s->llviddsp.add_hfyu_left_pred_int16(dst, dst, 1023, width, 0);
229  dst += stride;
230  }
231  for (k = 1 + interlaced; k < height; k++) {
232  s->llviddsp.add_hfyu_left_pred_int16(dst, dst, 1023, width, dst[-fake_stride]);
233  dst += stride;
234  }
235  break;
236  case GRADIENT:
237  dst = (uint16_t *)p->data[i] + j * sheight * stride;
238  s->llviddsp.add_hfyu_left_pred_int16(dst, dst, 1023, width, 0);
239  left = lefttop = 0;
240  dst += stride;
241  if (interlaced) {
242  s->llviddsp.add_hfyu_left_pred_int16(dst, dst, 1023, width, 0);
243  left = lefttop = 0;
244  dst += stride;
245  }
246  for (k = 1 + interlaced; k < height; k++) {
247  top = dst[-fake_stride];
248  left = top + dst[0];
249  dst[0] = left & 0x3FF;
250  for (x = 1; x < width; x++) {
251  top = dst[x - fake_stride];
252  lefttop = dst[x - (fake_stride + 1)];
253  left += top - lefttop + dst[x];
254  dst[x] = left & 0x3FF;
255  }
256  dst += stride;
257  }
258  break;
259  case MEDIAN:
260  dst = (uint16_t *)p->data[i] + j * sheight * stride;
261  lefttop = left = dst[0];
262  s->llviddsp.add_hfyu_left_pred_int16(dst, dst, 1023, width, 0);
263  dst += stride;
264  if (interlaced) {
265  lefttop = left = dst[0];
266  s->llviddsp.add_hfyu_left_pred_int16(dst, dst, 1023, width, 0);
267  dst += stride;
268  }
269  for (k = 1 + interlaced; k < height; k++) {
270  magicyuv_median_pred10(dst, dst - fake_stride, dst, width, &left, &lefttop);
271  lefttop = left = dst[0];
272  dst += stride;
273  }
274  break;
275  default:
276  avpriv_request_sample(avctx, "Unknown prediction: %d", pred);
277  }
278  }
279 
280  if (s->decorrelate) {
281  int height = FFMIN(s->slice_height, avctx->coded_height - j * s->slice_height);
282  int width = avctx->coded_width;
283  uint16_t *r = (uint16_t *)p->data[0] + j * s->slice_height * p->linesize[0] / 2;
284  uint16_t *g = (uint16_t *)p->data[1] + j * s->slice_height * p->linesize[1] / 2;
285  uint16_t *b = (uint16_t *)p->data[2] + j * s->slice_height * p->linesize[2] / 2;
286 
287  for (i = 0; i < height; i++) {
288  for (k = 0; k < width; k++) {
289  b[k] = (b[k] + g[k]) & 0x3FF;
290  r[k] = (r[k] + g[k]) & 0x3FF;
291  }
292  b += p->linesize[0] / 2;
293  g += p->linesize[1] / 2;
294  r += p->linesize[2] / 2;
295  }
296  }
297 
298  return 0;
299 }

302  int j, int threadnr)
303 {

305  int interlaced = s->interlaced;

314  int width = AV_CEIL_RSHIFT(avctx->coded_width, s->hshift[i]);

322  if (ret < 0)
323  return ret;
324 
325  flags = get_bits(&gb, 8);
326  pred = get_bits(&gb, 8);
327 
328  dst = p->data[i] + j * sheight * stride;
329  if (flags & 1) {
330  for (k = 0; k < height; k++) {
331  for (x = 0; x < width; x++)
332  dst[x] = get_bits(&gb, 8);
333 
334  dst += stride;
335  }
336  } else {
337  for (k = 0; k < height; k++) {

344  if (pix < 0)
345  return AVERROR_INVALIDDATA;
346 
347  dst[x] = 255 - pix;
348  }
349  dst += stride;
350  }
351  }
352 
353  switch (pred) {
354  case LEFT:
355  dst = p->data[i] + j * sheight * stride;
356  s->hdsp.add_hfyu_left_pred(dst, dst, width, 0);
357  dst += stride;
358  if (interlaced) {
359  s->hdsp.add_hfyu_left_pred(dst, dst, width, 0);
360  dst += stride;

364  dst += stride;
365  }
366  break;
367  case GRADIENT:
368  dst = p->data[i] + j * sheight * stride;
369  s->hdsp.add_hfyu_left_pred(dst, dst, width, 0);
370  left = lefttop = 0;
371  dst += stride;
372  if (interlaced) {
373  s->hdsp.add_hfyu_left_pred(dst, dst, width, 0);
374  left = lefttop = 0;
375  dst += stride;
376  }
377  for (k = 1 + interlaced; k < height; k++) {
378  top = dst[-fake_stride];
379  left = top + dst[0];
380  dst[0] = left;
381  for (x = 1; x < width; x++) {
382  top = dst[x - fake_stride];
383  lefttop = dst[x - (fake_stride + 1)];
384  left += top - lefttop + dst[x];
385  dst[x] = left;
386  }
387  dst += stride;
388  }
389  break;
390  case MEDIAN:
391  dst = p->data[i] + j * sheight * stride;
392  lefttop = left = dst[0];
393  s->hdsp.add_hfyu_left_pred(dst, dst, width, 0);
394  dst += stride;
395  if (interlaced) {
396  lefttop = left = dst[0];
397  s->hdsp.add_hfyu_left_pred(dst, dst, width, 0);
398  dst += stride;
399  }
400  for (k = 1 + interlaced; k < height; k++) {
401  s->hdsp.add_hfyu_median_pred(dst, dst - fake_stride,
402  dst, width, &left, &lefttop);
403  lefttop = left = dst[0];
404  dst += stride;
405  }
406  break;
407  default:
408  avpriv_request_sample(avctx, "Unknown prediction: %d", pred);
409  }
410  }
411 
412  if (s->decorrelate) {
413  int height = FFMIN(s->slice_height, avctx->coded_height - j * s->slice_height);
414  int width = avctx->coded_width;
415  uint8_t *b = p->data[0] + j * s->slice_height * p->linesize[0];
416  uint8_t *g = p->data[1] + j * s->slice_height * p->linesize[1];
417  uint8_t *r = p->data[2] + j * s->slice_height * p->linesize[2];
418 
419  for (i = 0; i < height; i++) {
420  s->hdsp.add_bytes(b, g, width);
421  s->hdsp.add_bytes(r, g, width);
422  b += p->linesize[0];
423  g += p->linesize[1];
424  r += p->linesize[2];
425  }
426  }
427 
428  return 0;
429 }

431 static int build_huffman(AVCodecContext *avctx, GetBitContext *gbit, int max)
432 {
433  MagicYUVContext *s = avctx->priv_data;
434  int i = 0, j = 0, k;
435 
436  memset(s->len, 0, sizeof(s->len));
437  while (get_bits_left(gbit) >= 8) {
438  int b = get_bits(gbit, 4);
439  int x = get_bits(gbit, 4);
440  int l = get_bitsz(gbit, b) + 1;
441 
442  for (k = 0; k < l; k++)
443  if (j + k < max)
444  s->len[i][j + k] = x;
445 
446  j += l;
447  if (j == max) {
448  j = 0;
449  if (s->huff_build(&s->vlc[i], s->len[i])) {
450  av_log(avctx, AV_LOG_ERROR, "Cannot build Huffman codes\n");
451  return AVERROR_INVALIDDATA;
452  }
453  i++;
454  if (i == s->planes) {
455  break;
456  }
457  } else if (j > max) {
458  return AVERROR_INVALIDDATA;
459  }
460  }
461 
462  if (i != s->planes) {
463  av_log(avctx, AV_LOG_ERROR, "Huffman tables too short\n");
464  return AVERROR_INVALIDDATA;
465  }
466 
467  return 0;
468 }

470 static int magy_decode_frame(AVCodecContext *avctx, void *data,
471  int *got_frame, AVPacket *avpkt)
472 {
473  MagicYUVContext *s = avctx->priv_data;
474  ThreadFrame frame = { .f = data };
475  AVFrame *p = data;
476  GetByteContext gbyte;
477  GetBitContext gbit;
478  uint32_t first_offset, offset, next_offset, header_size, slice_width;
479  int width, height, format, version, table_size;
480  int ret, i, j;
481 
482  bytestream2_init(&gbyte, avpkt->data, avpkt->size);
483  if (bytestream2_get_le32(&gbyte) != MKTAG('M', 'A', 'G', 'Y'))
484  return AVERROR_INVALIDDATA;
485 
486  header_size = bytestream2_get_le32(&gbyte);
487  if (header_size < 32 || header_size >= avpkt->size) {
488  av_log(avctx, AV_LOG_ERROR,
489  "header or packet too small %"PRIu32"\n", header_size);
490  return AVERROR_INVALIDDATA;
491  }
492 
493  version = bytestream2_get_byte(&gbyte);
494  if (version != 7) {
495  avpriv_request_sample(avctx, "Version %d", version);
496  return AVERROR_PATCHWELCOME;
497  }
498 
499  s->hshift[1] =
500  s->vshift[1] =
501  s->hshift[2] =
502  s->vshift[2] = 0;
503  s->decorrelate = 0;
504  s->max = 256;
505  s->huff_build = huff_build;
506  s->magy_decode_slice = magy_decode_slice;
507 
508  format = bytestream2_get_byte(&gbyte);
509  switch (format) {
510  case 0x65:
511  avctx->pix_fmt = AV_PIX_FMT_GBRP;
512  s->decorrelate = 1;
513  break;
514  case 0x66:
515  avctx->pix_fmt = AV_PIX_FMT_GBRAP;
516  s->decorrelate = 1;
517  break;
518  case 0x67:
519  avctx->pix_fmt = AV_PIX_FMT_YUV444P;
520  break;
521  case 0x68:
522  avctx->pix_fmt = AV_PIX_FMT_YUV422P;
523  s->hshift[1] =
524  s->hshift[2] = 1;
525  break;
526  case 0x69:
527  avctx->pix_fmt = AV_PIX_FMT_YUV420P;
528  s->hshift[1] =
529  s->vshift[1] =
530  s->hshift[2] =
531  s->vshift[2] = 1;
532  break;
533  case 0x6a:
534  avctx->pix_fmt = AV_PIX_FMT_YUVA444P;
535  break;
536  case 0x6b:
537  avctx->pix_fmt = AV_PIX_FMT_GRAY8;
538  break;
539  case 0x6c:
540  avctx->pix_fmt = AV_PIX_FMT_YUV422P10;
541  s->hshift[1] =
542  s->hshift[2] = 1;
543  s->max = 1024;
544  s->huff_build = huff_build10;
545  s->magy_decode_slice = magy_decode_slice10;
546  break;
547  case 0x6d:
548  avctx->pix_fmt = AV_PIX_FMT_GBRP10;
549  s->decorrelate = 1;
550  s->max = 1024;
551  s->huff_build = huff_build10;
552  s->magy_decode_slice = magy_decode_slice10;
553  break;
554  case 0x6e:
555  avctx->pix_fmt = AV_PIX_FMT_GBRAP10;
556  s->decorrelate = 1;
557  s->max = 1024;
558  s->huff_build = huff_build10;
559  s->magy_decode_slice = magy_decode_slice10;
560  break;
561  case 0x73:
562  avctx->pix_fmt = AV_PIX_FMT_GRAY10;
563  s->max = 1024;
564  s->huff_build = huff_build10;
565  s->magy_decode_slice = magy_decode_slice10;
566  break;
567  default:
568  avpriv_request_sample(avctx, "Format 0x%X", format);
569  return AVERROR_PATCHWELCOME;
570  }
571  s->planes = av_pix_fmt_count_planes(avctx->pix_fmt);
572 
573  bytestream2_skip(&gbyte, 1);
574  s->color_matrix = bytestream2_get_byte(&gbyte);
575  s->flags = bytestream2_get_byte(&gbyte);
576  s->interlaced = !!(s->flags & 2);
577  bytestream2_skip(&gbyte, 3);
578 
579  width = bytestream2_get_le32(&gbyte);
580  height = bytestream2_get_le32(&gbyte);
581  ret = ff_set_dimensions(avctx, width, height);
582  if (ret < 0)
583  return ret;
584 
585  slice_width = bytestream2_get_le32(&gbyte);
586  if (slice_width != avctx->coded_width) {
587  avpriv_request_sample(avctx, "Slice width %"PRIu32, slice_width);
588  return AVERROR_PATCHWELCOME;
589  }
590  s->slice_height = bytestream2_get_le32(&gbyte);
591  if (s->slice_height <= 0 || s->slice_height > INT_MAX - avctx->coded_height) {
592  av_log(avctx, AV_LOG_ERROR,
593  "invalid slice height: %d\n", s->slice_height);
594  return AVERROR_INVALIDDATA;
595  }
596 
597  bytestream2_skip(&gbyte, 4);
598 
599  s->nb_slices = (avctx->coded_height + s->slice_height - 1) / s->slice_height;
600  if (s->nb_slices > INT_MAX / sizeof(Slice)) {
601  av_log(avctx, AV_LOG_ERROR,
602  "invalid number of slices: %d\n", s->nb_slices);
603  return AVERROR_INVALIDDATA;
604  }
605 
606  for (i = 0; i < s->planes; i++) {
607  av_fast_malloc(&s->slices[i], &s->slices_size[i], s->nb_slices * sizeof(Slice));
608  if (!s->slices[i])
609  return AVERROR(ENOMEM);
610 
611  offset = bytestream2_get_le32(&gbyte);
612  if (offset >= avpkt->size - header_size)
613  return AVERROR_INVALIDDATA;
614 
615  if (i == 0)
