38 static int v410_encode_frame(AVCodecContext *avctx, AVPacket *pkt,
39  const AVFrame *pic, int *got_packet)
40 {
41  uint8_t *dst;
42  uint16_t *y, *u, *v;
43  uint32_t val;
44  int i, j, ret;
45 
46  if ((ret = ff_alloc_packet2(avctx, pkt, avctx->width * avctx->height * 4,
47  avctx->width * avctx->height * 4)) < 0)
48  return ret;
49  dst = pkt->data;
50 
51 #if FF_API_CODED_FRAME
52 FF_DISABLE_DEPRECATION_WARNINGS
53  avctx->coded_frame->key_frame = 1;
54  avctx->coded_frame->pict_type = AV_PICTURE_TYPE_I;
55 FF_ENABLE_DEPRECATION_WARNINGS
56 #endif
57 
58  y = (uint16_t *)pic->data[0];
59  u = (uint16_t *)pic->data[1];
60  v = (uint16_t *)pic->data[2];
61 
62  for (i = 0; i < avctx->height; i++) {
63  for (j = 0; j < avctx->width; j++) {
64  val = u[j] << 2;
65  val |= y[j] << 12;
66  val |= (uint32_t) v[j] << 22;
67  AV_WL32(dst, val);
68  dst += 4;
69  }
70  y += pic->linesize[0] >> 1;
71  u += pic->linesize[1] >> 1;
72  v += pic->linesize[2] >> 1;
73  }
74 
75  pkt->flags |= AV_PKT_FLAG_KEY;
76  *got_packet = 1;
77  return 0;
78 }
