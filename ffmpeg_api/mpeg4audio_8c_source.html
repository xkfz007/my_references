27 /**
28  * Parse MPEG-4 audio configuration for ALS object type.
29  * @param[in] gb bit reader context
30  * @param[in] c MPEG4AudioConfig structure to fill
31  * @return on success 0 is returned, otherwise a value < 0
32  */
33 static int parse_config_ALS(GetBitContext *gb, MPEG4AudioConfig *c)
34 {
35  if (get_bits_left(gb) < 112)
36  return AVERROR_INVALIDDATA;
37 
38  if (get_bits_long(gb, 32) != MKBETAG('A','L','S','\0'))
39  return AVERROR_INVALIDDATA;
40 
41  // override AudioSpecificConfig channel configuration and sample rate
42  // which are buggy in old ALS conformance files
43  c->sample_rate = get_bits_long(gb, 32);
44 
45  if (c->sample_rate <= 0) {
46  av_log(NULL, AV_LOG_ERROR, "Invalid sample rate %d\n", c->sample_rate);
47  return AVERROR_INVALIDDATA;
48  }
49 
50  // skip number of samples
51  skip_bits_long(gb, 32);
52 
53  // read number of channels
54  c->chan_config = 0;
55  c->channels = get_bits(gb, 16) + 1;
56 
57  return 0;
58 }

86 int avpriv_mpeg4audio_get_config(MPEG4AudioConfig *c, const uint8_t *buf,
87  int bit_size, int sync_extension)
88 {
89  GetBitContext gb;
90  int specific_config_bitindex, ret;
91 
92  if (bit_size <= 0)
93  return AVERROR_INVALIDDATA;
94 
95  ret = init_get_bits(&gb, buf, bit_size);
96  if (ret < 0)
97  return ret;
98 
99  c->object_type = get_object_type(&gb);
100  c->sample_rate = get_sample_rate(&gb, &c->sampling_index);
101  c->chan_config = get_bits(&gb, 4);
102  if (c->chan_config < FF_ARRAY_ELEMS(ff_mpeg4audio_channels))
103  c->channels = ff_mpeg4audio_channels[c->chan_config];
104  c->sbr = -1;
105  c->ps = -1;
106  if (c->object_type == AOT_SBR || (c->object_type == AOT_PS &&
107  // check for W6132 Annex YYYY draft MP3onMP4
108  !(show_bits(&gb, 3) & 0x03 && !(show_bits(&gb, 9) & 0x3F)))) {
109  if (c->object_type == AOT_PS)
110  c->ps = 1;
111  c->ext_object_type = AOT_SBR;
112  c->sbr = 1;
113  c->ext_sample_rate = get_sample_rate(&gb, &c->ext_sampling_index);
114  c->object_type = get_object_type(&gb);
115  if (c->object_type == AOT_ER_BSAC)
116  c->ext_chan_config = get_bits(&gb, 4);
117  } else {
118  c->ext_object_type = AOT_NULL;
119  c->ext_sample_rate = 0;
120  }
121  specific_config_bitindex = get_bits_count(&gb);
122 
123  if (c->object_type == AOT_ALS) {
124  skip_bits(&gb, 5);
125  if (show_bits_long(&gb, 24) != MKBETAG('\0','A','L','S'))
126  skip_bits_long(&gb, 24);
127 
128  specific_config_bitindex = get_bits_count(&gb);
129 
130  ret = parse_config_ALS(&gb, c);
131  if (ret < 0)
132  return ret;
133  }
134 
135  if (c->ext_object_type != AOT_SBR && sync_extension) {
136  while (get_bits_left(&gb) > 15) {
137  if (show_bits(&gb, 11) == 0x2b7) { // sync extension
138  get_bits(&gb, 11);
139  c->ext_object_type = get_object_type(&gb);
140  if (c->ext_object_type == AOT_SBR && (c->sbr = get_bits1(&gb)) == 1) {
141  c->ext_sample_rate = get_sample_rate(&gb, &c->ext_sampling_index);
142  if (c->ext_sample_rate == c->sample_rate)
143  c->sbr = -1;
144  }
145  if (get_bits_left(&gb) > 11 && get_bits(&gb, 11) == 0x548)
146  c->ps = get_bits1(&gb);
147  break;
148  } else
149  get_bits1(&gb); // skip 1 bit
150  }
151  }
152 
153  //PS requires SBR
154  if (!c->sbr)
155  c->ps = 0;
156  //Limit implicit PS to the HE-AACv2 Profile
157  if ((c->ps == -1 && c->object_type != AOT_AAC_LC) || c->channels & ~0x01)
158  c->ps = 0;
159 
160  return specific_config_bitindex;
161 }

172 int avpriv_copy_pce_data(PutBitContext *pb, GetBitContext *gb)
173 {
174  int five_bit_ch, four_bit_ch, comment_size, bits;
175  int offset = put_bits_count(pb);
176 
177  copy_bits(pb, gb, 10); //Tag, Object Type, Frequency
178  five_bit_ch = copy_bits(pb, gb, 4); //Front
179  five_bit_ch += copy_bits(pb, gb, 4); //Side
180  five_bit_ch += copy_bits(pb, gb, 4); //Back
181  four_bit_ch = copy_bits(pb, gb, 2); //LFE
182  four_bit_ch += copy_bits(pb, gb, 3); //Data
183  five_bit_ch += copy_bits(pb, gb, 4); //Coupling
184  if (copy_bits(pb, gb, 1)) //Mono Mixdown
185  copy_bits(pb, gb, 4);
186  if (copy_bits(pb, gb, 1)) //Stereo Mixdown
187  copy_bits(pb, gb, 4);
188  if (copy_bits(pb, gb, 1)) //Matrix Mixdown
189  copy_bits(pb, gb, 3);
190  for (bits = five_bit_ch*5+four_bit_ch*4; bits > 16; bits -= 16)
191  copy_bits(pb, gb, 16);
192  if (bits)
193  copy_bits(pb, gb, bits);
194  avpriv_align_put_bits(pb);
195  align_get_bits(gb);
196  comment_size = copy_bits(pb, gb, 8);
197  for (; comment_size > 0; comment_size--)
198  copy_bits(pb, gb, 8);
199 
200  return put_bits_count(pb) - offset;
201 }
