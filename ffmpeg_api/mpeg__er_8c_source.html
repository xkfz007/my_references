24 {
25  int i;
26 
27  memset(dst, 0, sizeof(*dst));
28  if (!src) {
29  dst->f = NULL;

31  return;
32  }
33 
34  dst->f = src->f;

39  dst->ref_index[i] = src->ref_index[i];

42  dst->mb_type = src->mb_type;
43  dst->field_picture = src->field_picture;
44 }

126  return 0;
