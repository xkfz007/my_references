80 static void print_cpu_flags(int cpu_flags, const char *type)
81 {
82  int i;
83

93 int main(int argc, char **argv)
94 {
95  int cpu_flags_raw = av_get_cpu_flags();
96  int cpu_flags_eff;
97  int cpu_count = av_cpu_count();
98  char threads[5] = "auto";
99  int i;
100 
101  for(i = 0; cpu_flag_tab[i].flag; i++) {
102  unsigned tmp = 0;
103  if (av_parse_cpu_caps(&tmp, cpu_flag_tab[i].name) < 0) {
104  fprintf(stderr, "Table missing %s\n", cpu_flag_tab[i].name);
105  return 4;
106  }
107  }
108 
109  if (cpu_flags_raw < 0)
110  return 1;
111 
112  for (;;) {
113  int c = getopt(argc, argv, "c:t:");
114  if (c == -1)
115  break;
116  switch (c) {
117  case 'c':
118  {
119  unsigned flags = av_get_cpu_flags();
120  if (av_parse_cpu_caps(&flags, optarg) < 0)
121  return 2;
122 
123  av_force_cpu_flags(flags);
124  break;
125  }
126  case 't':
127  {
128  int len = av_strlcpy(threads, optarg, sizeof(threads));
129  if (len >= sizeof(threads)) {
130  fprintf(stderr, "Invalid thread count '%s'\n", optarg);
131  return 2;
132  }
133  }
134  }
135  }
136 
137  cpu_flags_eff = av_get_cpu_flags();
138 
139  if (cpu_flags_eff < 0)
140  return 3;
141 
142  print_cpu_flags(cpu_flags_raw, "raw");
143  print_cpu_flags(cpu_flags_eff, "effective");
144  printf("threads = %s (cpu_count = %d)\n", threads, cpu_count);
145 
146  return 0;
147 }
