51 static av_cold int msrle_decode_init(AVCodecContext *avctx)
52 {
53  MsrleContext *s = avctx->priv_data;
54  int i;
55 
56  s->avctx = avctx;
57 
58  switch (avctx->bits_per_coded_sample) {
59  case 1:
60  avctx->pix_fmt = AV_PIX_FMT_MONOWHITE;
61  break;
62  case 4:
63  case 8:
64  avctx->pix_fmt = AV_PIX_FMT_PAL8;
65  break;
66  case 24:
67  avctx->pix_fmt = AV_PIX_FMT_BGR24;
68  break;
69  default:
70  av_log(avctx, AV_LOG_ERROR, "unsupported bits per sample\n");
71  return AVERROR_INVALIDDATA;
72  }
73 
74  s->frame = av_frame_alloc();
75  if (!s->frame)
76  return AVERROR(ENOMEM);
77 
78  if (avctx->extradata_size >= 4)
79  for (i = 0; i < FFMIN(avctx->extradata_size, AVPALETTE_SIZE)/4; i++)
80  s->pal[i] = 0xFFU<<24 | AV_RL32(avctx->extradata+4*i);
81 
82  return 0;
83 }

85 static int msrle_decode_frame(AVCodecContext *avctx,
86  void *data, int *got_frame,
87  AVPacket *avpkt)
88 {
89  const uint8_t *buf = avpkt->data;
90  int buf_size = avpkt->size;
91  MsrleContext *s = avctx->priv_data;
92  int istride = FFALIGN(avctx->width*avctx->bits_per_coded_sample, 32) / 8;
93  int ret;
94 
95  s->buf = buf;
96  s->size = buf_size;
97 
98  if ((ret = ff_reget_buffer(avctx, s->frame)) < 0)
99  return ret;
100 
101  if (avctx->bits_per_coded_sample > 1 && avctx->bits_per_coded_sample <= 8) {
102  int size;
103  const uint8_t *pal = av_packet_get_side_data(avpkt, AV_PKT_DATA_PALETTE, &size);
104 
105  if (pal && size == AVPALETTE_SIZE) {
106  s->frame->palette_has_changed = 1;
107  memcpy(s->pal, pal, AVPALETTE_SIZE);
108  } else if (pal) {
109  av_log(avctx, AV_LOG_ERROR, "Palette size %d is wrong\n", size);
110  }
111  /* make the palette available */
112  memcpy(s->frame->data[1], s->pal, AVPALETTE_SIZE);
113  }
114 
115  /* FIXME how to correctly detect RLE ??? */
116  if (avctx->height * istride == avpkt->size) { /* assume uncompressed */
117  int linesize = av_image_get_linesize(avctx->pix_fmt, avctx->width, 0);
118  uint8_t *ptr = s->frame->data[0];
119  uint8_t *buf = avpkt->data + (avctx->height-1)*istride;
120  int i, j;
121 
122  if (linesize < 0)
123  return linesize;
124 
125  for (i = 0; i < avctx->height; i++) {
126  if (avctx->bits_per_coded_sample == 4) {
127  for (j = 0; j < avctx->width - 1; j += 2) {
128  ptr[j+0] = buf[j>>1] >> 4;
129  ptr[j+1] = buf[j>>1] & 0xF;
130  }
131  if (avctx->width & 1)
132  ptr[j+0] = buf[j>>1] >> 4;
133  } else {
134  memcpy(ptr, buf, linesize);
135  }
136  buf -= istride;
137  ptr += s->frame->linesize[0];
138  }
139  } else {
140  bytestream2_init(&s->gb, buf, buf_size);
141  ff_msrle_decode(avctx, s->frame, avctx->bits_per_coded_sample, &s->gb);
142  }
143 
144  if ((ret = av_frame_ref(data, s->frame)) < 0)
145  return ret;
146 
147  *got_frame = 1;
148 
149  /* report that the buffer was completely consumed */
150  return buf_size;
151 }

159
