49 {
50  const uint8_t *d = p->buf;
51  if (d[0] == 'M' && d[1] == 'P' && d[2] == '+' && (d[3] == 0x17 || d[3] == 0x7))

56 static int mpc_read_header(AVFormatContext *s)
57 {
58  MPCContext *c = s->priv_data;
59  AVStream *st;
60 
61  if(avio_rl24(s->pb) != MKTAG('M', 'P', '+', 0)){
62  av_log(s, AV_LOG_ERROR, "Not a Musepack file\n");
63  return AVERROR_INVALIDDATA;
64  }
65  c->ver = avio_r8(s->pb);
66  if(c->ver != 0x07 && c->ver != 0x17){
67  av_log(s, AV_LOG_ERROR, "Can demux Musepack SV7, got version %02X\n", c->ver);
68  return AVERROR_INVALIDDATA;
69  }
70  c->fcount = avio_rl32(s->pb);
71  if((int64_t)c->fcount * sizeof(MPCFrame) >= UINT_MAX){
72  av_log(s, AV_LOG_ERROR, "Too many frames, seeking is not possible\n");
73  return AVERROR_INVALIDDATA;
74  }
75  if(c->fcount){
76  c->frames = av_malloc(c->fcount * sizeof(MPCFrame));
77  if(!c->frames){
78  av_log(s, AV_LOG_ERROR, "Cannot allocate seektable\n");
79  return AVERROR(ENOMEM);
80  }
81  }else{
82  av_log(s, AV_LOG_WARNING, "Container reports no frames\n");
83  }
84  c->curframe = 0;
85  c->lastframe = -1;
86  c->curbits = 8;
87  c->frames_noted = 0;
88 
89  st = avformat_new_stream(s, NULL);
90  if (!st)
91  return AVERROR(ENOMEM);
92  st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
93  st->codecpar->codec_id = AV_CODEC_ID_MUSEPACK7;
94  st->codecpar->channels = 2;
95  st->codecpar->channel_layout = AV_CH_LAYOUT_STEREO;
96  st->codecpar->bits_per_coded_sample = 16;
97 
98  if (ff_get_extradata(s, st->codecpar, s->pb, 16) < 0)
99  return AVERROR(ENOMEM);
100  st->codecpar->sample_rate = mpc_rate[st->codecpar->extradata[2] & 3];
101  avpriv_set_pts_info(st, 32, MPC_FRAMESIZE, st->codecpar->sample_rate);
102  /* scan for seekpoints */
103  st->start_time = 0;
104  st->duration = c->fcount;
105 
106  /* try to read APE tags */
107  if (s->pb->seekable) {
108  int64_t pos = avio_tell(s->pb);
109  ff_ape_parse_tag(s);
110  if (!av_dict_get(s->metadata, "", NULL, AV_DICT_IGNORE_SUFFIX))
111  ff_id3v1_read(s);
112  avio_seek(s->pb, pos, SEEK_SET);
113  }
114 
115  return 0;
116 }

118 static int mpc_read_packet(AVFormatContext *s, AVPacket *pkt)
119 {
120  MPCContext *c = s->priv_data;
121  int ret, size, size2, curbits, cur = c->curframe;
122  unsigned tmp;
123  int64_t pos;
124 
125  if (c->curframe >= c->fcount && c->fcount)
126  return AVERROR_EOF;
127 
128  if(c->curframe != c->lastframe + 1){
129  avio_seek(s->pb, c->frames[c->curframe].pos, SEEK_SET);
130  c->curbits = c->frames[c->curframe].skip;
131  }
132  c->lastframe = c->curframe;
133  c->curframe++;
134  curbits = c->curbits;
135  pos = avio_tell(s->pb);
136  tmp = avio_rl32(s->pb);
137  if(curbits <= 12){
138  size2 = (tmp >> (12 - curbits)) & 0xFFFFF;
139  }else{
140  size2 = (tmp << (curbits - 12) | avio_rl32(s->pb) >> (44 - curbits)) & 0xFFFFF;
141  }
142  curbits += 20;
143  avio_seek(s->pb, pos, SEEK_SET);
144 
145  size = ((size2 + curbits + 31) & ~31) >> 3;
146  if(cur == c->frames_noted && c->fcount){
147  c->frames[cur].pos = pos;
148  c->frames[cur].size = size;
149  c->frames[cur].skip = curbits - 20;
150  av_add_index_entry(s->streams[0], cur, cur, size, 0, AVINDEX_KEYFRAME);
151  c->frames_noted++;
152  }
153  c->curbits = (curbits + size2) & 0x1F;
154 
155  if ((ret = av_new_packet(pkt, size + 4)) < 0)
156  return ret;
157 
158  pkt->data[0] = curbits;
159  pkt->data[1] = (c->curframe > c->fcount) && c->fcount;
160  pkt->data[2] = 0;
161  pkt->data[3] = 0;
162 
163  pkt->stream_index = 0;
164  pkt->pts = cur;
165  ret = avio_read(s->pb, pkt->data + 4, size);
166  if(c->curbits)
167  avio_seek(s->pb, -4, SEEK_CUR);
168  if(ret < size){
169  av_packet_unref(pkt);
170  return ret < 0 ? ret : AVERROR(EIO);
171  }
172  pkt->size = ret + 4;
173 
174  return 0;
175 }

185 /**
186  * Seek to the given position
187  * If position is unknown but is within the limits of file
188  * then packets are skipped unless desired position is reached
189  *
190  * Also this function makes use of the fact that timestamp == frameno
191  */
192 static int mpc_read_seek(AVFormatContext *s, int stream_index, int64_t timestamp, int flags)
193 {
194  AVStream *st = s->streams[stream_index];
195  MPCContext *c = s->priv_data;
196  AVPacket pkt1, *pkt = &pkt1;
197  int ret;
198  int index = av_index_search_timestamp(st, FFMAX(timestamp - DELAY_FRAMES, 0), flags);
199  uint32_t lastframe;
200 
201  /* if found, seek there */
202  if (index >= 0 && st->index_entries[st->nb_index_entries-1].timestamp >= timestamp - DELAY_FRAMES){
203  c->curframe = st->index_entries[index].pos;
204  return 0;
205  }
206  /* if timestamp is out of bounds, return error */
207  if(timestamp < 0 || timestamp >= c->fcount)
208  return -1;
209  timestamp -= DELAY_FRAMES;
210  /* seek to the furthest known position and read packets until
211  we reach desired position */
212  lastframe = c->curframe;
213  if(c->frames_noted) c->curframe = c->frames_noted - 1;
214  while(c->curframe < timestamp){
215  ret = av_read_frame(s, pkt);
216  if (ret < 0){
217  c->curframe = lastframe;
218  return ret;
219  }
220  av_packet_unref(pkt);
221  }
222  return 0;
223 }
