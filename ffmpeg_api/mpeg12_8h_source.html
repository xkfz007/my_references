32 static inline int decode_dc(GetBitContext *gb, int component)
33 {
34  int code, diff;
35 
36  if (component == 0) {
37  code = get_vlc2(gb, ff_dc_lum_vlc.table, DC_VLC_BITS, 2);
38  } else {
39  code = get_vlc2(gb, ff_dc_chroma_vlc.table, DC_VLC_BITS, 2);
40  }
41  if (code < 0){
42  av_log(NULL, AV_LOG_ERROR, "invalid dc code at\n");
43  return 0xffff;
44  }
45  if (code == 0) {
46  diff = 0;
47  } else {
48  diff = get_xbits(gb, code);
49  }
50  return diff;
51 }
