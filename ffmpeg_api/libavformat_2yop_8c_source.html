39 static int yop_probe(AVProbeData *probe_packet)
40 {
41  if (AV_RB16(probe_packet->buf) == AV_RB16("YO") &&
42  probe_packet->buf[2]<10 &&
43  probe_packet->buf[3]<10 &&
44  probe_packet->buf[6] &&
45  probe_packet->buf[7] &&
46  !(probe_packet->buf[8] & 1) &&
47  !(probe_packet->buf[10] & 1) &&
48  AV_RL16(probe_packet->buf + 12 + 6) >= 920 &&
49  AV_RL16(probe_packet->buf + 12 + 6) < probe_packet->buf[12] * 3 + 4 + probe_packet->buf[7] * 2048
50  )
51  return AVPROBE_SCORE_MAX * 3 / 4;
52 
53  return 0;
54 }

56 static int yop_read_header(AVFormatContext *s)
57 {
58  YopDecContext *yop = s->priv_data;
59  AVIOContext *pb = s->pb;
60 
61  AVCodecParameters *audio_par, *video_par;
62  AVStream *audio_stream, *video_stream;
63 
64  int frame_rate, ret;
65 
66  audio_stream = avformat_new_stream(s, NULL);
67  video_stream = avformat_new_stream(s, NULL);
68  if (!audio_stream || !video_stream)
69  return AVERROR(ENOMEM);
70 
71  // Extra data that will be passed to the decoder
72  if (ff_alloc_extradata(video_stream->codecpar, 8))
73  return AVERROR(ENOMEM);
74 
75  // Audio
76  audio_par = audio_stream->codecpar;
77  audio_par->codec_type = AVMEDIA_TYPE_AUDIO;
78  audio_par->codec_id = AV_CODEC_ID_ADPCM_IMA_APC;
79  audio_par->channels = 1;
80  audio_par->channel_layout = AV_CH_LAYOUT_MONO;
81  audio_par->sample_rate = 22050;
82 
83  // Video
84  video_par = video_stream->codecpar;
85  video_par->codec_type = AVMEDIA_TYPE_VIDEO;
86  video_par->codec_id = AV_CODEC_ID_YOP;
87 
88  avio_skip(pb, 6);
89 
90  frame_rate = avio_r8(pb);
91  yop->frame_size = avio_r8(pb) * 2048;
92  video_par->width = avio_rl16(pb);
93  video_par->height = avio_rl16(pb);
94 
95  video_stream->sample_aspect_ratio = (AVRational){1, 2};
96 
97  ret = avio_read(pb, video_par->extradata, 8);
98  if (ret < 8)
99  return ret < 0 ? ret : AVERROR_EOF;
100 
101  yop->palette_size = video_par->extradata[0] * 3 + 4;
102  yop->audio_block_length = AV_RL16(video_par->extradata + 6);
103 
104  video_par->bit_rate = 8 * (yop->frame_size - yop->audio_block_length) * frame_rate;
105 
106  // 1840 samples per frame, 1 nibble per sample; hence 1840/2 = 920
107  if (yop->audio_block_length < 920 ||
108  yop->audio_block_length + yop->palette_size >= yop->frame_size) {
109  av_log(s, AV_LOG_ERROR, "YOP has invalid header\n");
110  return AVERROR_INVALIDDATA;
111  }
112 
113  avio_seek(pb, 2048, SEEK_SET);
114 
115  avpriv_set_pts_info(video_stream, 32, 1, frame_rate);
116 
117  return 0;
118 }

120 static int yop_read_packet(AVFormatContext *s, AVPacket *pkt)
121 {
122  YopDecContext *yop = s->priv_data;
123  AVIOContext *pb = s->pb;
124 
125  int ret;
126  int actual_video_data_size = yop->frame_size -
127  yop->audio_block_length - yop->palette_size;
128 
129  yop->video_packet.stream_index = 1;
130 
131  if (yop->video_packet.data) {
132  *pkt = yop->video_packet;
133  yop->video_packet.data = NULL;
134  yop->video_packet.buf = NULL;
135  yop->video_packet.size = 0;
136  pkt->data[0] = yop->odd_frame;
137  pkt->flags |= AV_PKT_FLAG_KEY;
138  yop->odd_frame ^= 1;
139  return pkt->size;
140  }
141  ret = av_new_packet(&yop->video_packet,
142  yop->frame_size - yop->audio_block_length);
143  if (ret < 0)
144  return ret;
145 
146  yop->video_packet.pos = avio_tell(pb);
147 
148  ret = avio_read(pb, yop->video_packet.data, yop->palette_size);
149  if (ret < 0) {
150  goto err_out;
151  }else if (ret < yop->palette_size) {
152  ret = AVERROR_EOF;
153  goto err_out;
154  }
155 
156  ret = av_get_packet(pb, pkt, 920);
157  if (ret < 0)
158  goto err_out;
159 
160  // Set position to the start of the frame
161  pkt->pos = yop->video_packet.pos;
162 
163  avio_skip(pb, yop->audio_block_length - ret);
164 
165  ret = avio_read(pb, yop->video_packet.data + yop->palette_size,
166  actual_video_data_size);
167  if (ret < 0)
168  goto err_out;
169  else if (ret < actual_video_data_size)
170  av_shrink_packet(&yop->video_packet, yop->palette_size + ret);
171 
172  // Arbitrarily return the audio data first
173  return yop->audio_block_length;
174 
175 err_out:
176  av_packet_unref(&yop->video_packet);
177  return ret;
178 }

187 static int yop_read_seek(AVFormatContext *s, int stream_index,
188  int64_t timestamp, int flags)
189 {
190  YopDecContext *yop = s->priv_data;
191  int64_t frame_pos, pos_min, pos_max;
192  int frame_count;
193 
194  if (!stream_index)
195  return -1;
196 
197  pos_min = s->internal->data_offset;
198  pos_max = avio_size(s->pb) - yop->frame_size;
199  frame_count = (pos_max - pos_min) / yop->frame_size;
200 
201  timestamp = FFMAX(0, FFMIN(frame_count, timestamp));
202 
203  frame_pos = timestamp * yop->frame_size + pos_min;
204 
205  if (avio_seek(s->pb, frame_pos, SEEK_SET) < 0)
206  return -1;
207 
208  av_packet_unref(&yop->video_packet);
209  yop->odd_frame = timestamp & 1;
210 
211  return 0;
212 }
