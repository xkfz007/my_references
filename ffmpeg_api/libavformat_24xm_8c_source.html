92 static int parse_vtrk(AVFormatContext *s,
93  FourxmDemuxContext *fourxm, uint8_t *buf, int size,
94  int left)
95 {
96  AVStream *st;
97  /* check that there is enough data */
98  if (size != vtrk_SIZE || left < size + 8) {
99  return AVERROR_INVALIDDATA;
100  }
101 
102  /* allocate a new AVStream */
103  st = avformat_new_stream(s, NULL);
104  if (!st)
105  return AVERROR(ENOMEM);
106 
107  avpriv_set_pts_info(st, 60, fourxm->fps.den, fourxm->fps.num);
108 
109  fourxm->video_stream_index = st->index;
110 
111  st->codecpar->codec_type = AVMEDIA_TYPE_VIDEO;
112  st->codecpar->codec_id = AV_CODEC_ID_4XM;
113 
114  st->codecpar->extradata = av_mallocz(4 + AV_INPUT_BUFFER_PADDING_SIZE);
115  if (!st->codecpar->extradata)
116  return AVERROR(ENOMEM);
117  st->codecpar->extradata_size = 4;
118  AV_WL32(st->codecpar->extradata, AV_RL32(buf + 16));
119  st->codecpar->width = AV_RL32(buf + 36);
120  st->codecpar->height = AV_RL32(buf + 40);
121 
122  return 0;
123 }

126 static int parse_strk(AVFormatContext *s,
127  FourxmDemuxContext *fourxm, uint8_t *buf, int size,
128  int left)
129 {
130  AVStream *st;
131  int track;
132  /* check that there is enough data */
133  if (size != strk_SIZE || left < size + 8)
134  return AVERROR_INVALIDDATA;
135 
136  track = AV_RL32(buf + 8);
137  if ((unsigned)track >= UINT_MAX / sizeof(AudioTrack) - 1) {
138  av_log(s, AV_LOG_ERROR, "current_track too large\n");
139  return AVERROR_INVALIDDATA;
140  }
141 
142  if (track + 1 > fourxm->track_count) {
143  if (av_reallocp_array(&fourxm->tracks, track + 1, sizeof(AudioTrack)))
144  return AVERROR(ENOMEM);
145  memset(&fourxm->tracks[fourxm->track_count], 0,
146  sizeof(AudioTrack) * (track + 1 - fourxm->track_count));
147  fourxm->track_count = track + 1;
148  }
149  fourxm->tracks[track].adpcm = AV_RL32(buf + 12);
150  fourxm->tracks[track].channels = AV_RL32(buf + 36);
151  fourxm->tracks[track].sample_rate = AV_RL32(buf + 40);
152  fourxm->tracks[track].bits = AV_RL32(buf + 44);
153  fourxm->tracks[track].audio_pts = 0;
154 
155  if (fourxm->tracks[track].channels <= 0 ||
156  fourxm->tracks[track].sample_rate <= 0 ||
157  fourxm->tracks[track].bits <= 0) {
158  av_log(s, AV_LOG_ERROR, "audio header invalid\n");
159  return AVERROR_INVALIDDATA;
160  }
161  if (!fourxm->tracks[track].adpcm && fourxm->tracks[track].bits<8) {
162  av_log(s, AV_LOG_ERROR, "bits unspecified for non ADPCM\n");
163  return AVERROR_INVALIDDATA;
164  }
165 
166  if (fourxm->tracks[track].sample_rate > INT64_MAX / fourxm->tracks[track].bits / fourxm->tracks[track].channels) {
167  av_log(s, AV_LOG_ERROR, "Overflow during bit rate calculation %d * %d * %d\n",
168  fourxm->tracks[track].sample_rate, fourxm->tracks[track].bits, fourxm->tracks[track].channels);
169  return AVERROR_INVALIDDATA;
170  }
171 
172  /* allocate a new AVStream */
173  st = avformat_new_stream(s, NULL);
174  if (!st)
175  return AVERROR(ENOMEM);
176 
177  st->id = track;
178  avpriv_set_pts_info(st, 60, 1, fourxm->tracks[track].sample_rate);
179 
180  fourxm->tracks[track].stream_index = st->index;
181 
182  st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
183  st->codecpar->codec_tag = 0;
184  st->codecpar->channels = fourxm->tracks[track].channels;
185  st->codecpar->sample_rate = fourxm->tracks[track].sample_rate;
186  st->codecpar->bits_per_coded_sample = fourxm->tracks[track].bits;
187  st->codecpar->bit_rate = (int64_t)st->codecpar->channels *
188  st->codecpar->sample_rate *
189  st->codecpar->bits_per_coded_sample;
190  st->codecpar->block_align = st->codecpar->channels *
191  st->codecpar->bits_per_coded_sample;
192 
193  if (fourxm->tracks[track].adpcm){
194  st->codecpar->codec_id = AV_CODEC_ID_ADPCM_4XM;
195  } else if (st->codecpar->bits_per_coded_sample == 8) {
196  st->codecpar->codec_id = AV_CODEC_ID_PCM_U8;
197  } else
198  st->codecpar->codec_id = AV_CODEC_ID_PCM_S16LE;
199 
200  return 0;
201 }

203 static int fourxm_read_header(AVFormatContext *s)
204 {
205  AVIOContext *pb = s->pb;
206  unsigned int fourcc_tag;
207  unsigned int size;
208  int header_size;
209  FourxmDemuxContext *fourxm = s->priv_data;
210  unsigned char *header;
211  int i, ret;
212 
213  fourxm->track_count = 0;
214  fourxm->tracks = NULL;
215  fourxm->fps = (AVRational){1,1};
216 
217  /* skip the first 3 32-bit numbers */
218  avio_skip(pb, 12);
219 
220  /* check for LIST-HEAD */
221  GET_LIST_HEADER();
222  header_size = size - 4;
223  if (fourcc_tag != HEAD_TAG || header_size < 0)
224  return AVERROR_INVALIDDATA;
225 
226  /* allocate space for the header and load the whole thing */
227  header = av_malloc(header_size);
228  if (!header)
229  return AVERROR(ENOMEM);
230  if (avio_read(pb, header, header_size) != header_size) {
231  av_free(header);
232  return AVERROR(EIO);
233  }
234 
235  /* take the lazy approach and search for any and all vtrk and strk chunks */
236  for (i = 0; i < header_size - 8; i++) {
237  fourcc_tag = AV_RL32(&header[i]);
238  size = AV_RL32(&header[i + 4]);
239  if (size > header_size - i - 8 && (fourcc_tag == vtrk_TAG || fourcc_tag == strk_TAG)) {
240  av_log(s, AV_LOG_ERROR, "chunk larger than array %d>%d\n", size, header_size - i - 8);
241  return AVERROR_INVALIDDATA;
242  }
243 
244  if (fourcc_tag == std__TAG) {
245  if (header_size - i < 16) {
246  av_log(s, AV_LOG_ERROR, "std TAG truncated\n");
247  ret = AVERROR_INVALIDDATA;
248  goto fail;
249  }
250  fourxm->fps = av_d2q(av_int2float(AV_RL32(&header[i + 12])), 10000);
251  } else if (fourcc_tag == vtrk_TAG) {
252  if ((ret = parse_vtrk(s, fourxm, header + i, size,
253  header_size - i)) < 0)
254  goto fail;
255 
256  i += 8 + size;
257  } else if (fourcc_tag == strk_TAG) {
258  if ((ret = parse_strk(s, fourxm, header + i, size,
259  header_size - i)) < 0)
260  goto fail;
261 
262  i += 8 + size;
263  }
264  }
265 
266  /* skip over the LIST-MOVI chunk (which is where the stream should be */
267  GET_LIST_HEADER();
268  if (fourcc_tag != MOVI_TAG) {
269  ret = AVERROR_INVALIDDATA;
270  goto fail;
271  }
272 
273  av_free(header);
274  /* initialize context members */
275  fourxm->video_pts = -1; /* first frame will push to 0 */
276 
277  return 0;
278 fail:
279  av_freep(&fourxm->tracks);
280  av_free(header);
281  return ret;
282 }

284 static int fourxm_read_packet(AVFormatContext *s,
285  AVPacket *pkt)
286 {
287  FourxmDemuxContext *fourxm = s->priv_data;
288  AVIOContext *pb = s->pb;
289  unsigned int fourcc_tag;
290  unsigned int size;
291  int ret = 0;
292  unsigned int track_number;
293  int packet_read = 0;
294  unsigned char header[8];
295  int audio_frame_count;
296 
297  while (!packet_read) {
298  if ((ret = avio_read(s->pb, header, 8)) < 0)
299  return ret;
300  fourcc_tag = AV_RL32(&header[0]);
301  size = AV_RL32(&header[4]);
302  if (avio_feof(pb))
303  return AVERROR(EIO);
304  switch (fourcc_tag) {
305  case LIST_TAG:
306  /* this is a good time to bump the video pts */
307  fourxm->video_pts++;
308 
309  /* skip the LIST-* tag and move on to the next fourcc */
310  avio_rl32(pb);
311  break;
312 
313  case ifrm_TAG:
314  case pfrm_TAG:
315  case cfrm_TAG:
316  case ifr2_TAG:
317  case pfr2_TAG:
318  case cfr2_TAG:
319  /* allocate 8 more bytes than 'size' to account for fourcc
320  * and size */
321  if (size + 8 < size || av_new_packet(pkt, size + 8))
322  return AVERROR(EIO);
323  pkt->stream_index = fourxm->video_stream_index;
324  pkt->pts = fourxm->video_pts;
325  pkt->pos = avio_tell(s->pb);
326  memcpy(pkt->data, header, 8);
327  ret = avio_read(s->pb, &pkt->data[8], size);
328 
329  if (ret < 0) {
330  av_packet_unref(pkt);
331  } else {
332  packet_read = 1;
333  av_shrink_packet(pkt, ret + 8);
334  }
335  break;
336 
337  case snd__TAG:
338  track_number = avio_rl32(pb);
339  avio_skip(pb, 4);
340  size -= 8;
341 
342  if (track_number < fourxm->track_count &&
343  fourxm->tracks[track_number].channels > 0) {
344  ret = av_get_packet(s->pb, pkt, size);
345  if (ret < 0)
346  return AVERROR(EIO);
347  pkt->stream_index =
348  fourxm->tracks[track_number].stream_index;
349  pkt->pts = fourxm->tracks[track_number].audio_pts;
350  packet_read = 1;
351 
352  /* pts accounting */
353  audio_frame_count = size;
354  if (fourxm->tracks[track_number].adpcm)
355  audio_frame_count -= 2 * (fourxm->tracks[track_number].channels);
356  audio_frame_count /= fourxm->tracks[track_number].channels;
357  if (fourxm->tracks[track_number].adpcm) {
358  audio_frame_count *= 2;
359  } else
360  audio_frame_count /=
361  (fourxm->tracks[track_number].bits / 8);
362  fourxm->tracks[track_number].audio_pts += audio_frame_count;
363  } else {
364  avio_skip(pb, size);
365  }
366  break;
367 
368  default:
369  avio_skip(pb, size);
370  break;
371  }
372  }
373  return ret;
374 }
