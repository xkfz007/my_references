458 static struct rendition *new_rendition(HLSContext *c, struct rendition_info *info,
459  const char *url_base)
460 {
461  struct rendition *rend;
462  enum AVMediaType type = AVMEDIA_TYPE_UNKNOWN;
463  char *characteristic;
464  char *chr_ptr;
465  char *saveptr;
466 
467  if (!strcmp(info->type, "AUDIO"))
468  type = AVMEDIA_TYPE_AUDIO;
469  else if (!strcmp(info->type, "VIDEO"))
470  type = AVMEDIA_TYPE_VIDEO;
471  else if (!strcmp(info->type, "SUBTITLES"))
472  type = AVMEDIA_TYPE_SUBTITLE;
473  else if (!strcmp(info->type, "CLOSED-CAPTIONS"))
474  /* CLOSED-CAPTIONS is ignored since we do not support CEA-608 CC in
475  * AVC SEI RBSP anyway */
476  return NULL;
477 
478  if (type == AVMEDIA_TYPE_UNKNOWN)
479  return NULL;
480 
481  /* URI is mandatory for subtitles as per spec */
482  if (type == AVMEDIA_TYPE_SUBTITLE && !info->uri[0])
483  return NULL;
484 
485  /* TODO: handle subtitles (each segment has to parsed separately) */
486  if (c->strict_std_compliance > FF_COMPLIANCE_EXPERIMENTAL)
487  if (type == AVMEDIA_TYPE_SUBTITLE)
488  return NULL;
489 
490  rend = av_mallocz(sizeof(struct rendition));
491  if (!rend)
492  return NULL;
493 
494  dynarray_add(&c->renditions, &c->n_renditions, rend);
495 
496  rend->type = type;
497  strcpy(rend->group_id, info->group_id);
498  strcpy(rend->language, info->language);
499  strcpy(rend->name, info->name);
500 
501  /* add the playlist if this is an external rendition */
502  if (info->uri[0]) {
503  rend->playlist = new_playlist(c, info->uri, url_base);
504  if (rend->playlist)
505  dynarray_add(&rend->playlist->renditions,
506  &rend->playlist->n_renditions, rend);
507  }
508 
509  if (info->assoc_language[0]) {
510  int langlen = strlen(rend->language);
511  if (langlen < sizeof(rend->language) - 3) {
512  rend->language[langlen] = ',';
513  strncpy(rend->language + langlen + 1, info->assoc_language,
514  sizeof(rend->language) - langlen - 2);
515  }
516  }
517 
518  if (!strcmp(info->defaultr, "YES"))
519  rend->disposition |= AV_DISPOSITION_DEFAULT;
520  if (!strcmp(info->forced, "YES"))
521  rend->disposition |= AV_DISPOSITION_FORCED;
522 
523  chr_ptr = info->characteristics;
524  while ((characteristic = av_strtok(chr_ptr, ",", &saveptr))) {
525  if (!strcmp(characteristic, "public.accessibility.describes-music-and-sound"))
526  rend->disposition |= AV_DISPOSITION_HEARING_IMPAIRED;
527  else if (!strcmp(characteristic, "public.accessibility.describes-video"))
528  rend->disposition |= AV_DISPOSITION_VISUAL_IMPAIRED;
529 
530  chr_ptr = NULL;
531  }
532 
533  return rend;
534 }

598 static int open_url(AVFormatContext *s, AVIOContext **pb, const char *url,
599  AVDictionary *opts, AVDictionary *opts2, int *is_http)
600 {
601  HLSContext *c = s->priv_data;
602  AVDictionary *tmp = NULL;
603  const char *proto_name = NULL;
604  int ret;
605 
606  av_dict_copy(&tmp, opts, 0);
607  av_dict_copy(&tmp, opts2, 0);
608 
609  if (av_strstart(url, "crypto", NULL)) {
610  if (url[6] == '+' || url[6] == ':')
611  proto_name = avio_find_protocol_name(url + 7);
612  }
613 
614  if (!proto_name)
615  proto_name = avio_find_protocol_name(url);
616 
617  if (!proto_name)
618  return AVERROR_INVALIDDATA;
619 
620  // only http(s) & file are allowed
621  if (!av_strstart(proto_name, "http", NULL) && !av_strstart(proto_name, "file", NULL))
622  return AVERROR_INVALIDDATA;
623  if (!strncmp(proto_name, url, strlen(proto_name)) && url[strlen(proto_name)] == ':')
624  ;
625  else if (av_strstart(url, "crypto", NULL) && !strncmp(proto_name, url + 7, strlen(proto_name)) && url[7 + strlen(proto_name)] == ':')
626  ;
627  else if (strcmp(proto_name, "file") || !strncmp(url, "file,", 5))
628  return AVERROR_INVALIDDATA;
629 
630  ret = s->io_open(s, pb, url, AVIO_FLAG_READ, &tmp);
631  if (ret >= 0) {
632  // update cookies on http response with setcookies.
633  void *u = (s->flags & AVFMT_FLAG_CUSTOM_IO) ? NULL : s->pb;
634  update_options(&c->cookies, "cookies", u);
635  av_dict_set(&opts, "cookies", c->cookies, 0);
636  }
637 
638  av_dict_free(&tmp);
639 
640  if (is_http)
641  *is_http = av_strstart(proto_name, "http", NULL);
642 
643  return ret;
644 }

771 #endif

815  av_free(seg);
816  ret = AVERROR(ENOMEM);
817  goto fail;
818  }
819  } else {
820  seg->key = NULL;
821  }
822 
823  ff_make_absolute_url(tmp_str, sizeof(tmp_str), url, line);
824 #if DEBUG_LOGS
825  av_log(c,AV_LOG_DEBUG,"hls demux:ts url=%s\n",tmp_str);
826 #endif
827  seg->url = av_strdup(tmp_str);
828  if (!seg->url) {
829  av_free(seg->key);
830  av_free(seg);
831  ret = AVERROR(ENOMEM);
832  goto fail;
833  }
834

836  is_segment = 0;
837 
838  seg->size = seg_size;
839  if (seg_size >= 0) {
840  seg->url_offset = seg_offset;
841  seg_offset += seg_size;
842  seg_size = -1;
843  } else {
844  seg->url_offset = 0;
845  seg_offset = 0;
846  }
847 
848  seg->init_section = cur_init_section;
849  }
850  }
851  }
852  if (pls)
853  pls->last_load_time = av_gettime_relative();
854 
855 fail:
856  av_free(new_url);
857  if (close_in)
858  ff_format_io_close(c->ctx, &in);
859  return ret;
860 }

895 /* Parse the raw ID3 data and pass contents to caller */
896 static void parse_id3(AVFormatContext *s, AVIOContext *pb,
897  AVDictionary **metadata, int64_t *dts,
898  ID3v2ExtraMetaAPIC **apic, ID3v2ExtraMeta **extra_meta)
899 {
900  static const char id3_priv_owner_ts[] = "com.apple.streaming.transportStreamTimestamp";
901  ID3v2ExtraMeta *meta;
902 
903  ff_id3v2_read_dict(pb, metadata, ID3v2_DEFAULT_MAGIC, extra_meta);
904  for (meta = *extra_meta; meta; meta = meta->next) {
905  if (!strcmp(meta->tag, "PRIV")) {
906  ID3v2ExtraMetaPRIV *priv = meta->data;
907  if (priv->datasize == 8 && !strcmp(priv->owner, id3_priv_owner_ts)) {
908  /* 33-bit MPEG timestamp */
909  int64_t ts = AV_RB64(priv->data);
910  av_log(s, AV_LOG_DEBUG, "HLS ID3 audio timestamp %"PRId64"\n", ts);
911  if ((ts & ~((1ULL << 33) - 1)) == 0)
912  *dts = ts;
913  else
914  av_log(s, AV_LOG_ERROR, "Invalid HLS ID3 audio timestamp %"PRId64"\n", ts);
915  }
916  } else if (!strcmp(meta->tag, "APIC") && apic)
917  *apic = meta->data;
918  }
919 }

921 /* Check if the ID3 metadata contents have changed */
922 static int id3_has_changed_values(struct playlist *pls, AVDictionary *metadata,
923  ID3v2ExtraMetaAPIC *apic)
924 {
925  AVDictionaryEntry *entry = NULL;
926  AVDictionaryEntry *oldentry;
927  /* check that no keys have changed values */
928  while ((entry = av_dict_get(metadata, "", entry, AV_DICT_IGNORE_SUFFIX))) {
929  oldentry = av_dict_get(pls->id3_initial, entry->key, NULL, AV_DICT_MATCH_CASE);
930  if (!oldentry || strcmp(oldentry->value, entry->value) != 0)
931  return 1;
932  }
933 
934  /* check if apic appeared */
935  if (apic && (pls->ctx->nb_streams != 2 || !pls->ctx->streams[1]->attached_pic.data))
936  return 1;
937 
938  if (apic) {
939  int size = pls->ctx->streams[1]->attached_pic.size;
940  if (size != apic->buf->size - AV_INPUT_BUFFER_PADDING_SIZE)
941  return 1;
942 
943  if (memcmp(apic->buf->data, pls->ctx->streams[1]->attached_pic.data, size) != 0)
944  return 1;
945  }
946 
947  return 0;
948 }

951 static void handle_id3(AVIOContext *pb, struct playlist *pls)
952 {
953  AVDictionary *metadata = NULL;
954  ID3v2ExtraMetaAPIC *apic = NULL;
955  ID3v2ExtraMeta *extra_meta = NULL;

962  pls->id3_offset = 0;

966  /* initial ID3 tags */
967  av_assert0(!pls->id3_deferred_extra);
968  pls->id3_found = 1;

971  if (pls->ctx->nb_streams)
972  ff_id3v2_parse_apic(pls->ctx, &extra_meta);
973  else
974  /* demuxer not yet opened, defer picture attachment */
975  pls->id3_deferred_extra = extra_meta;

978  pls->id3_initial = metadata;

983  pls->id3_changed = 1;

985  av_dict_free(&metadata);
986  }
987 
988  if (!pls->id3_deferred_extra)
989  ff_id3v2_free_extra_meta(&extra_meta);

992 static void intercept_id3(struct playlist *pls, uint8_t *buf,
993  int buf_size, int *len)
994 {
995  /* intercept id3 tags, we do not want to pass them to the raw
996  * demuxer on all segment switches */
997  int bytes;
998  int id3_buf_pos = 0;
999  int fill_buf = 0;
1000  struct segment *seg = current_segment(pls);
1001 
1002  /* gather all the id3 tags */
1003  while (1) {
1004  /* see if we can retrieve enough data for ID3 header */
1005  if (*len < ID3v2_HEADER_SIZE && buf_size >= ID3v2_HEADER_SIZE) {
1006  bytes = read_from_url(pls, seg, buf + *len, ID3v2_HEADER_SIZE - *len, READ_COMPLETE);
1007  if (bytes > 0) {
1008 
1009  if (bytes == ID3v2_HEADER_SIZE - *len)
1010  /* no EOF yet, so fill the caller buffer again after
1011  * we have stripped the ID3 tags */
1012  fill_buf = 1;
1013 
1014  *len += bytes;
1015 
1016  } else if (*len <= 0) {
1017  /* error/EOF */
1018  *len = bytes;
1019  fill_buf = 0;
1020  }
1021  }
1022 
1023  if (*len < ID3v2_HEADER_SIZE)
1024  break;
1025 
1026  if (ff_id3v2_match(buf, ID3v2_DEFAULT_MAGIC)) {
1027  int64_t maxsize = seg->size >= 0 ? seg->size : 1024*1024;
1028  int taglen = ff_id3v2_tag_len(buf);
1029  int tag_got_bytes = FFMIN(taglen, *len);
1030  int remaining = taglen - tag_got_bytes;
1031 
1032  if (taglen > maxsize) {
1033  av_log(pls->ctx, AV_LOG_ERROR, "Too large HLS ID3 tag (%d > %"PRId64" bytes)\n",
1034  taglen, maxsize);
1035  break;
1036  }
1037 
1038  /*
1039  * Copy the id3 tag to our temporary id3 buffer.
1040  * We could read a small id3 tag directly without memcpy, but
1041  * we would still need to copy the large tags, and handling
1042  * both of those cases together with the possibility for multiple
1043  * tags would make the handling a bit complex.
1044  */
1045  pls->id3_buf = av_fast_realloc(pls->id3_buf, &pls->id3_buf_size, id3_buf_pos + taglen);
1046  if (!pls->id3_buf)
1047  break;
1048  memcpy(pls->id3_buf + id3_buf_pos, buf, tag_got_bytes);
1049  id3_buf_pos += tag_got_bytes;
1050 
1051  /* strip the intercepted bytes */
1052  *len -= tag_got_bytes;
1053  memmove(buf, buf + tag_got_bytes, *len);
1054  av_log(pls->ctx, AV_LOG_DEBUG, "Stripped %d HLS ID3 bytes\n", tag_got_bytes);
1055 
1056  if (remaining > 0) {
1057  /* read the rest of the tag in */
1058  if (read_from_url(pls, seg, pls->id3_buf + id3_buf_pos, remaining, READ_COMPLETE) != remaining)
1059  break;
1060  id3_buf_pos += remaining;
1061  av_log(pls->ctx, AV_LOG_DEBUG, "Stripped additional %d HLS ID3 bytes\n", remaining);
1062  }
1063 
1064  } else {
1065  /* no more ID3 tags */
1066  break;
1067  }
1068  }
1069 
1070  /* re-fill buffer for the caller unless EOF */
1071  if (*len >= 0 && (fill_buf || *len == 0)) {
1072  bytes = read_from_url(pls, seg, buf + *len, buf_size - *len, READ_NORMAL);
1073 
1074  /* ignore error if we already had some data */
1075  if (bytes >= 0)
1076  *len += bytes;
1077  else if (*len == 0)
1078  *len = bytes;
1079  }
1080 
1081  if (pls->id3_buf) {
1082  /* Now parse all the ID3 tags */
1083  AVIOContext id3ioctx;
1084  ffio_init_context(&id3ioctx, pls->id3_buf, id3_buf_pos, 0, NULL, NULL, NULL, NULL);
1085  handle_id3(&id3ioctx, pls);
1086  }
1087 
1088  if (pls->is_id3_timestamped == -1)
1089  pls->is_id3_timestamped = (pls->id3_mpegts_timestamp != AV_NOPTS_VALUE);
1090 }

1092 static int open_input(HLSContext *c, struct playlist *pls, struct segment *seg)
1093 {
1094  AVDictionary *opts = NULL;
1095  int ret;
1096  int is_http = 0;
1097 
1098  // broker prior HTTP options that should be consistent across requests
1099  av_dict_set(&opts, "user-agent", c->user_agent, 0);
1100  av_dict_set(&opts, "cookies", c->cookies, 0);
1101  av_dict_set(&opts, "headers", c->headers, 0);
1102  av_dict_set(&opts, "http_proxy", c->http_proxy, 0);
1103  av_dict_set(&opts, "seekable", "0", 0);
1104 
1105  if (seg->size >= 0) {
1106  /* try to restrict the HTTP request to the part we want
1107  * (if this is in fact a HTTP request) */
1108  av_dict_set_int(&opts, "offset", seg->url_offset, 0);
1109  av_dict_set_int(&opts, "end_offset", seg->url_offset + seg->size, 0);
1110  }
1111 
1112  av_log(pls->parent, AV_LOG_VERBOSE, "HLS request for url '%s', offset %"PRId64", playlist %d\n",
1113  seg->url, seg->url_offset, pls->index);
1114 
1115  if (seg->key_type == KEY_NONE) {
1116  ret = open_url(pls->parent, &pls->input, seg->url, c->avio_opts, opts, &is_http);
1117  } else if (seg->key_type == KEY_AES_128) {
1118  AVDictionary *opts2 = NULL;
1119  char iv[33], key[33], url[MAX_URL_SIZE];
1120  if (strcmp(seg->key, pls->key_url)) {
1121  AVIOContext *pb;
1122  if (open_url(pls->parent, &pb, seg->key, c->avio_opts, opts, NULL) == 0) {
1123  ret = avio_read(pb, pls->key, sizeof(pls->key));
1124  if (ret != sizeof(pls->key)) {
1125  av_log(NULL, AV_LOG_ERROR, "Unable to read key file %s\n",
1126  seg->key);
1127  }
1128  ff_format_io_close(pls->parent, &pb);
1129  } else {
1130  av_log(NULL, AV_LOG_ERROR, "Unable to open key file %s\n",
1131  seg->key);
1132  }
1133  av_strlcpy(pls->key_url, seg->key, sizeof(pls->key_url));
1134  }
1135  ff_data_to_hex(iv, seg->iv, sizeof(seg->iv), 0);
1136  ff_data_to_hex(key, pls->key, sizeof(pls->key), 0);
1137  iv[32] = key[32] = '\0';
1138  if (strstr(seg->url, "://"))
1139  snprintf(url, sizeof(url), "crypto+%s", seg->url);
1140  else
1141  snprintf(url, sizeof(url), "crypto:%s", seg->url);
1142 
1143  av_dict_copy(&opts2, c->avio_opts, 0);
1144  av_dict_set(&opts2, "key", key, 0);
1145  av_dict_set(&opts2, "iv", iv, 0);
1146 
1147  ret = open_url(pls->parent, &pls->input, url, opts2, opts, &is_http);
1148 
1149  av_dict_free(&opts2);
1150 
1151  if (ret < 0) {
1152  goto cleanup;
1153  }
1154  ret = 0;
1155  } else if (seg->key_type == KEY_SAMPLE_AES) {
1156  av_log(pls->parent, AV_LOG_ERROR,
1157  "SAMPLE-AES encryption is not supported yet\n");
1158  ret = AVERROR_PATCHWELCOME;
1159  }
1160  else
1161  ret = AVERROR(ENOSYS);
1162 
1163  /* Seek to the requested position. If this was a HTTP request, the offset
1164  * should already be where want it to, but this allows e.g. local testing
1165  * without a HTTP server.
1166  *
1167  * This is not done for HTTP at all as avio_seek() does internal bookkeeping
1168  * of file offset which is out-of-sync with the actual offset when "offset"
1169  * AVOption is used with http protocol, causing the seek to not be a no-op
1170  * as would be expected. Wrong offset received from the server will not be
1171  * noticed without the call, though.
1172  */
1173  if (ret == 0 && !is_http && seg->key_type == KEY_NONE && seg->url_offset) {
1174  int64_t seekret = avio_seek(pls->input, seg->url_offset, SEEK_SET);
1175  if (seekret < 0) {
1176  av_log(pls->parent, AV_LOG_ERROR, "Unable to seek to offset %"PRId64" of HLS segment '%s'\n", seg->url_offset, seg->url);
1177  ret = seekret;
1178  ff_format_io_close(pls->parent, &pls->input);
1179  }
1180  }
1181 
1182 cleanup:
1183  av_dict_free(&opts);
1184  pls->cur_seg_offset = 0;
1185  return ret;
1186 }

1192  int64_t sec_size;

1194  int ret;
1195 
1196  if (seg->init_section == pls->cur_init_section)
1197  return 0;
1198 
1199  pls->cur_init_section = NULL;
1200

1252 static int read_data(void *opaque, uint8_t *buf, int buf_size)
1253 {

1256  int ret, i;
1257  int just_opened = 0;
1258 
1259 restart:
1260  if (!v->needed)
1261  return AVERROR_EOF;
1262 
1263  if (!v->input) {
1264  int64_t reload_interval;
1265  struct segment *seg;
1266 
1267  /* Check that the playlist is still needed before opening a new
1268  * segment. */
1269  if (v->ctx && v->ctx->nb_streams) {
1270  v->needed = 0;
1271  for (i = 0; i < v->n_main_streams; i++) {
1272  if (v->main_streams[i]->discard < AVDISCARD_ALL) {
1273  v->needed = 1;
1274  break;
1275  }
1276  }
1277  }
1278  if (!v->needed) {
1279  av_log(v->parent, AV_LOG_INFO, "No longer receiving playlist %d\n",
1280  v->index);
1281  return AVERROR_EOF;
1282  }
1283 
1284  /* If this is a live stream and the reload interval has elapsed since
1285  * the last playlist reload, reload the playlists now. */
1286  reload_interval = default_reload_interval(v);
1287 
1288 reload:
1289  if (!v->finished &&
1290  av_gettime_relative() - v->last_load_time >= reload_interval) {
1291  if ((ret = parse_playlist(c, v->url, v, NULL)) < 0) {
1292  av_log(v->parent, AV_LOG_WARNING, "Failed to reload playlist %d\n",
1293  v->index);

1295  }
1296  /* If we need to reload the playlist again below (if
1297  * there's still no more segments), switch to a reload
1298  * interval of half the target duration. */
1299  reload_interval = v->target_duration / 2;
1300  }
1301  if (v->cur_seq_no < v->start_seq_no) {
1302  av_log(NULL, AV_LOG_WARNING,
1303  "skipping %d segments ahead, expired from playlists\n",
1304  v->start_seq_no - v->cur_seq_no);
1305 #if DEBUG_LOGS
1306  av_log(NULL, AV_LOG_DEBUG,
1307  "v->start_seq_no=%d v->cur_seq_no=%d\n",
1308  v->start_seq_no , v->cur_seq_no);
1309 #endif
1310  v->cur_seq_no = v->start_seq_no;
1311  }
1312  if (v->cur_seq_no >= v->start_seq_no + v->n_segments) {
1313  if (v->finished)
1314  return AVERROR_EOF;
1315  while (av_gettime_relative() - v->last_load_time < reload_interval) {
1316  if (ff_check_interrupt(c->interrupt_callback))
1317  return AVERROR_EXIT;
1318  av_usleep(100*1000);
1319  }
1320  /* Enough time has elapsed since the last reload */
1321  goto reload;
1322  }
1323 
1324  seg = current_segment(v);
1325 
1326  /* load/update Media Initialization Section, if any */
1327  ret = update_init_section(v, seg);
1328  if (ret)
1329  return ret;
1330 
1331  ret = open_input(c, v, seg);
1332  if (ret < 0) {
1333  if (ff_check_interrupt(c->interrupt_callback))
1334  return AVERROR_EXIT;
1335  av_log(v->parent, AV_LOG_WARNING, "Failed to open segment of playlist %d\n",
1336  v->index);
1337  v->cur_seq_no += 1;
1338  goto reload;
1339  }
1340  just_opened = 1;
1341  }
1342 
1343  if (v->init_sec_buf_read_offset < v->init_sec_data_len) {
1344  /* Push init section out first before first actual segment */
1345  int copy_size = FFMIN(v->init_sec_data_len - v->init_sec_buf_read_offset, buf_size);
1346  memcpy(buf, v->init_sec_buf, copy_size);
1347  v->init_sec_buf_read_offset += copy_size;
1348  return copy_size;
1349  }
1350 
1351  ret = read_from_url(v, current_segment(v), buf, buf_size, READ_NORMAL);
1352  if (ret > 0) {
1353  if (just_opened && v->is_id3_timestamped != 0) {
1354  /* Intercept ID3 tags here, elementary audio streams are required
1355  * to convey timestamps using them in the beginning of each segment. */
1356  intercept_id3(v, buf, buf_size, &ret);
1357  }
1358 
1359  return ret;
1360  }
1361  ff_format_io_close(v->parent, &v->input);
1362  v->cur_seq_no++;
1363 
1364  c->cur_seq_no = v->cur_seq_no;
1365 
1366  goto restart;
1367 }

1423 /* if timestamp was in valid range: returns 1 and sets seq_no
1424  * if not: returns 0 and sets seq_no to closest segment */
1425 static int find_timestamp_in_playlist(HLSContext *c, struct playlist *pls,

1427 {

1430  0 : c->first_timestamp;
1431 
1432  if (timestamp < pos) {
1433  *seq_no = pls->start_seq_no;
1434  return 0;
1435  }
1436 
1437  for (i = 0; i < pls->n_segments; i++) {
1438  int64_t diff = pos + pls->segments[i]->duration - timestamp;
1439  if (diff > 0) {
1440  *seq_no = pls->start_seq_no + i;
1441  return 1;
1442  }
1443  pos += pls->segments[i]->duration;

1451 static int select_cur_seq_no(HLSContext *c, struct playlist *pls)
1452 {
1453  int seq_no;
1454 
1455  if (!pls->finished && !c->first_packet &&
1456  av_gettime_relative() - pls->last_load_time >= default_reload_interval(pls))
1457  /* reload the playlist since it was suspended */
1458  parse_playlist(c, pls->url, pls, NULL);
1459 
1460  /* If playback is already in progress (we are just selecting a new
1461  * playlist) and this is a complete file, find the matching segment
1462  * by counting durations. */
1463  if (pls->finished && c->cur_timestamp != AV_NOPTS_VALUE) {
1464  find_timestamp_in_playlist(c, pls, c->cur_timestamp, &seq_no);
1465  return seq_no;
1466  }
1467 
1468  if (!pls->finished) {
1469  if (!c->first_packet && /* we are doing a segment selection during playback */
1470  c->cur_seq_no >= pls->start_seq_no &&
1471  c->cur_seq_no < pls->start_seq_no + pls->n_segments)
1472  /* While spec 3.4.3 says that we cannot assume anything about the
1473  * content at the same sequence number on different playlists,
1474  * in practice this seems to work and doing it otherwise would
1475  * require us to download a segment to inspect its timestamps. */
1476  return c->cur_seq_no;
1477 
1478  /* If this is a live stream, start live_start_index segments from the
1479  * start or end */
1480  if (c->live_start_index < 0)
1481  return pls->start_seq_no + FFMAX(pls->n_segments + c->live_start_index, 0);
1482  else
1483  return pls->start_seq_no + FFMIN(c->live_start_index, pls->n_segments - 1);
1484  }
1485 
1486  /* Otherwise just start on the first segment. */
1487  return pls->start_seq_no;
1488 }
