31 static float sbr_sum_square_c(float (*x)[2], int n)
32 {
33  float sum0 = 0.0f, sum1 = 0.0f;
34  int i;
35 
36  for (i = 0; i < n; i += 2)
37  {
38  sum0 += x[i + 0][0] * x[i + 0][0];
39  sum1 += x[i + 0][1] * x[i + 0][1];
40  sum0 += x[i + 1][0] * x[i + 1][0];
41  sum1 += x[i + 1][1] * x[i + 1][1];
42  }
43 
44  return sum0 + sum1;
45 }
