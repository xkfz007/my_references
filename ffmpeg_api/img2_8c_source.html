83 static enum AVCodecID str2id(const IdStrMap *tags, const char *str)
84 {
85  str = strrchr(str, '.');
86  if (!str)
87  return AV_CODEC_ID_NONE;
88  str++;
89 
90  while (tags->id) {
91  if (!av_strcasecmp(str, tags->str))
92  return tags->id;
93 
94  tags++;
95  }
96  return AV_CODEC_ID_NONE;
97 }
