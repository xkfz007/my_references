303  status += nc_get_vara_float(ncid, sp_id, (size_t[2]){ 0, 0 } ,

305  status += nc_get_vara_float(ncid, sp_id, (size_t[2]){ 0, 1 } ,
306  (size_t[2]){ m_dim, 1}, sp_e); /* read & store elevation angles */
307  status += nc_get_vara_float(ncid, sp_id, (size_t[2]){ 0, 2 } ,

309  if (status != NC_NOERR) { /* if any source position variable coudn't be read */
310  av_log(ctx, AV_LOG_ERROR, "Couldn't read SourcePosition.\n");
311  ret = AVERROR(EINVAL);
312  goto error;
313  }
314 
315  /* read Data.Delay, check for errors and fit it to data_delay */
316  status = nc_inq_varid(ncid, "Data.Delay", &data_delay_id);
317  status += nc_inq_vardimid(ncid, data_delay_id, &data_delay_dim_id[0]);
318  status += nc_inq_dimname(ncid, data_delay_dim_id[0], data_delay_dim_name);

323  }

326  /* dimension of Data.Delay is [I R]: */
327  if (!strncmp(data_delay_dim_name, "I", 2)) {
328  /* check 2 characters to assure string is 0-terminated after "I" */
329  int delay[2]; /* delays get from SOFA file: */
330  int *data_delay_r;
331 
332  av_log(ctx, AV_LOG_DEBUG, "Data.Delay has dimension [I R]\n");
333  status = nc_get_var_int(ncid, data_delay_id, &delay[0]);
334  if (status != NC_NOERR) {
335  av_log(ctx, AV_LOG_ERROR, "Couldn't read Data.Delay\n");
336  ret = AVERROR(EINVAL);
337  goto error;
338  }
339  data_delay_r = data_delay + m_dim;
340  for (i = 0; i < m_dim; i++) { /* extend given dimension [I R] to [M R] */
341  /* assign constant delay value for all measurements to data_delay fields */
342  data_delay[i] = delay[0];
343  data_delay_r[i] = delay[1];
344  }
345  /* dimension of Data.Delay is [M R] */
346  } else if (!strncmp(data_delay_dim_name, "M", 2)) {
347  av_log(ctx, AV_LOG_ERROR, "Data.Delay in dimension [M R]\n");
348  /* get delays from SOFA file: */
349  status = nc_get_var_int(ncid, data_delay_id, data_delay);
350  if (status != NC_NOERR) {
351  av_log(ctx, AV_LOG_ERROR, "Couldn't read Data.Delay\n");
352  ret = AVERROR(EINVAL);
353  goto error;
354  }
355  } else { /* dimension of Data.Delay is neither [I R] nor [M R] */
356  av_log(ctx, AV_LOG_ERROR, "Data.Delay does not have the required dimensions [I R] or [M R].\n");
357  ret = AVERROR(EINVAL);
358  goto error;
359  }
360 
361  /* save information in SOFA struct: */
362  s->sofa.m_dim = m_dim; /* no. measurement positions */
363  s->sofa.n_samples = n_samples; /* length on one IR */
364  s->sofa.ncid = ncid; /* netCDF ID of SOFA file */
365  nc_close(ncid); /* close SOFA file */
366 
367  av_log(ctx, AV_LOG_DEBUG, "m_dim: %d n_samples %d\n", m_dim, n_samples);
368 
369  return 0;
370 
371 error:
372  close_sofa(&s->sofa);
373  return ret;
374 }

376 static int parse_channel_name(char **arg, int *rchannel, char *buf)
377 {
378  int len, i, channel_id = 0;
379  int64_t layout, layout0;
380 
381  /* try to parse a channel name, e.g. "FL" */
382  if (sscanf(*arg, "%7[A-Z]%n", buf, &len)) {
383  layout0 = layout = av_get_channel_layout(buf);
384  /* channel_id <- first set bit in layout */
385  for (i = 32; i > 0; i >>= 1) {
386  if (layout >= (int64_t)1 << i) {
387  channel_id += i;
388  layout >>= i;
389  }
390  }
391  /* reject layouts that are not a single channel */
392  if (channel_id >= 64 || layout0 != (int64_t)1 << channel_id)
393  return AVERROR(EINVAL);
394  *rchannel = channel_id;
395  *arg += len;
396  return 0;
397  }
398  return AVERROR(EINVAL);
399 }

521 static int find_m(SOFAlizerContext *s, int azim, int elev, float radius)
522 {
523  /* get source positions and M of currently selected SOFA file */
524  float *sp_a = s->sofa.sp_a; /* azimuth angle */
525  float *sp_e = s->sofa.sp_e; /* elevation angle */
526  float *sp_r = s->sofa.sp_r; /* radius */
527  int m_dim = s->sofa.m_dim; /* no. measurements */
528  int best_id = 0; /* index m currently closest to desired source pos. */
529  float delta = 1000; /* offset between desired and currently best pos. */
530  float current;
531  int i;
532 
533  for (i = 0; i < m_dim; i++) {
534  /* search through all measurements in currently selected SOFA file */
535  /* distance of current to desired source position: */
536  current = fabs(sp_a[i] - azim) +
537  fabs(sp_e[i] - elev) +
538  fabs(sp_r[i] - radius);
539  if (current <= delta) {
540  /* if current distance is smaller than smallest distance so far */
541  delta = current;
542  best_id = i; /* remember index */
543  }
544  }
545 
546  return best_id;
547 }

555  int m;
556 
557  if (s->sofa.ncid) {
558  /* find IR at front center position in the SOFA file (IR closest to 0°,0°,1m) */
559  struct NCSofa *sofa = &s->sofa;
560  m = find_m(s, 0, 0, 1);
561  /* get energy of that IR and compensate volume */
562  ir = sofa->data_ir + 2 * m * sofa->n_samples;
563  if (sofa->n_samples & 31) {
564  energy = avpriv_scalarproduct_float_c(ir, ir, sofa->n_samples);
565  } else {
566  energy = s->fdsp->scalarproduct_float(ir, ir, sofa->n_samples);
567  }
568  compensate = 256 / (sofa->n_samples * sqrt(energy));
569  av_log(ctx, AV_LOG_DEBUG, "Compensate-factor: %f\n", compensate);
570  ir = sofa->data_ir;
571  /* apply volume compensation to IRs */
572  if (sofa->n_samples & 31) {
573  int i;
574  for (i = 0; i < sofa->n_samples * sofa->m_dim * 2; i++) {
575  ir[i] = ir[i] * compensate;
576  }
577  } else {
578  s->fdsp->vector_fmul_scalar(ir, ir, compensate, sofa->n_samples * sofa->m_dim * 2);
579  emms_c();
580  }
581  }
582 
583  return 0;
584 }

597 static int sofalizer_convolute(AVFilterContext *ctx, void *arg, int jobnr, int nb_jobs)
598 {
599  SOFAlizerContext *s = ctx->priv;
600  ThreadData *td = arg;
601  AVFrame *in = td->in, *out = td->out;
602  int offset = jobnr;
603  int *write = &td->write[jobnr];
604  const int *const delay = td->delay[jobnr];
605  const float *const ir = td->ir[jobnr];
606  int *n_clippings = &td->n_clippings[jobnr];
607  float *ringbuffer = td->ringbuffer[jobnr];
608  float *temp_src = td->temp_src[jobnr];
609  const int n_samples = s->sofa.n_samples; /* length of one IR */
610  const float *src = (const float *)in->data[0]; /* get pointer to audio input buffer */
611  float *dst = (float *)out->data[0]; /* get pointer to audio output buffer */
612  const int in_channels = s->n_conv; /* number of input channels */
613  /* ring buffer length is: longest IR plus max. delay -> next power of 2 */
614  const int buffer_length = s->buffer_length;
615  /* -1 for AND instead of MODULO (applied to powers of 2): */
616  const uint32_t modulo = (uint32_t)buffer_length - 1;
617  float *buffer[16]; /* holds ringbuffer for each input channel */
618  int wr = *write;
619  int read;
620  int i, l;
621 
622  dst += offset;
623  for (l = 0; l < in_channels; l++) {
624  /* get starting address of ringbuffer for each input channel */
625  buffer[l] = ringbuffer + l * buffer_length;
626  }
627 
628  for (i = 0; i < in->nb_samples; i++) {
629  const float *temp_ir = ir; /* using same set of IRs for each sample */
630 
631  *dst = 0;
632  for (l = 0; l < in_channels; l++) {
633  /* write current input sample to ringbuffer (for each channel) */
634  *(buffer[l] + wr) = src[l];
635  }
636 
637  /* loop goes through all channels to be convolved */
638  for (l = 0; l < in_channels; l++) {
639  const float *const bptr = buffer[l];
640 
641  if (l == s->lfe_channel) {
642  /* LFE is an input channel but requires no convolution */
643  /* apply gain to LFE signal and add to output buffer */
644  *dst += *(buffer[s->lfe_channel] + wr) * s->gain_lfe;
645  temp_ir += FFALIGN(n_samples, 16);
646  continue;
647  }
648 
649  /* current read position in ringbuffer: input sample write position
650  * - delay for l-th ch. + diff. betw. IR length and buffer length
651  * (mod buffer length) */
652  read = (wr - *(delay + l) - (n_samples - 1) + buffer_length) & modulo;
653 
654  if (read + n_samples < buffer_length) {
655  memcpy(temp_src, bptr + read, n_samples * sizeof(*temp_src));
656  } else {
657  int len = FFMIN(n_samples - (read % n_samples), buffer_length - read);
658 
659  memcpy(temp_src, bptr + read, len * sizeof(*temp_src));
660  memcpy(temp_src + len, bptr, (n_samples - len) * sizeof(*temp_src));
661  }
662 
663  /* multiply signal and IR, and add up the results */
664  dst[0] += s->fdsp->scalarproduct_float(temp_ir, temp_src, n_samples);
665  temp_ir += FFALIGN(n_samples, 16);
666  }
667 
668  /* clippings counter */
669  if (fabs(*dst) > 1)
670  *n_clippings += 1;
671 
672  /* move output buffer pointer by +2 to get to next sample of processed channel: */
673  dst += 2;
674  src += in_channels;
675  wr = (wr + 1) & modulo; /* update ringbuffer write position */
676  }
677 
678  *write = wr; /* remember write position in ringbuffer for next call */
679 
680  return 0;
681 }

683 static int sofalizer_fast_convolute(AVFilterContext *ctx, void *arg, int jobnr, int nb_jobs)
684 {
685  SOFAlizerContext *s = ctx->priv;
686  ThreadData *td = arg;
687  AVFrame *in = td->in, *out = td->out;
688  int offset = jobnr;
689  int *write = &td->write[jobnr];
690  FFTComplex *hrtf = s->data_hrtf[jobnr]; /* get pointers to current HRTF data */
691  int *n_clippings = &td->n_clippings[jobnr];
692  float *ringbuffer = td->ringbuffer[jobnr];
693  const int n_samples = s->sofa.n_samples; /* length of one IR */
694  const float *src = (const float *)in->data[0]; /* get pointer to audio input buffer */
695  float *dst = (float *)out->data[0]; /* get pointer to audio output buffer */
696  const int in_channels = s->n_conv; /* number of input channels */
697  /* ring buffer length is: longest IR plus max. delay -> next power of 2 */
698  const int buffer_length = s->buffer_length;
699  /* -1 for AND instead of MODULO (applied to powers of 2): */
700  const uint32_t modulo = (uint32_t)buffer_length - 1;
701  FFTComplex *fft_in = s->temp_fft[jobnr]; /* temporary array for FFT input/output data */
702  FFTContext *ifft = s->ifft[jobnr];
703  FFTContext *fft = s->fft[jobnr];
704  const int n_conv = s->n_conv;
705  const int n_fft = s->n_fft;
706  int wr = *write;
707  int n_read;
708  int i, j;
709 
710  dst += offset;
711 
712  /* find minimum between number of samples and output buffer length:
713  * (important, if one IR is longer than the output buffer) */
714  n_read = FFMIN(s->sofa.n_samples, in->nb_samples);
715  for (j = 0; j < n_read; j++) {
716  /* initialize output buf with saved signal from overflow buf */
717  dst[2 * j] = ringbuffer[wr];
718  ringbuffer[wr] = 0.0; /* re-set read samples to zero */
719  /* update ringbuffer read/write position */
720  wr = (wr + 1) & modulo;
721  }
722 
723  /* initialize rest of output buffer with 0 */
724  for (j = n_read; j < in->nb_samples; j++) {
725  dst[2 * j] = 0;
726  }
727 
728  for (i = 0; i < n_conv; i++) {
729  if (i == s->lfe_channel) { /* LFE */
730  for (j = 0; j < in->nb_samples; j++) {
731  /* apply gain to LFE signal and add to output buffer */
732  dst[2 * j] += src[i + j * in_channels] * s->gain_lfe;
733  }
734  continue;
735  }
736 
737  /* outer loop: go through all input channels to be convolved */
738  offset = i * n_fft; /* no. samples already processed */
739 
740  /* fill FFT input with 0 (we want to zero-pad) */
741  memset(fft_in, 0, sizeof(FFTComplex) * n_fft);
742 
743  for (j = 0; j < in->nb_samples; j++) {
744  /* prepare input for FFT */
745  /* write all samples of current input channel to FFT input array */
746  fft_in[j].re = src[j * in_channels + i];
747  }
748 
749  /* transform input signal of current channel to frequency domain */
750  av_fft_permute(fft, fft_in);
751  av_fft_calc(fft, fft_in);
752  for (j = 0; j < n_fft; j++) {
753  const float re = fft_in[j].re;
754  const float im = fft_in[j].im;
755 
756  /* complex multiplication of input signal and HRTFs */
757  /* output channel (real): */
758  fft_in[j].re = re * (hrtf + offset + j)->re - im * (hrtf + offset + j)->im;
759  /* output channel (imag): */
760  fft_in[j].im = re * (hrtf + offset + j)->im + im * (hrtf + offset + j)->re;
761  }
762 
763  /* transform output signal of current channel back to time domain */
764  av_fft_permute(ifft, fft_in);
765  av_fft_calc(ifft, fft_in);
766 
767  for (j = 0; j < in->nb_samples; j++) {
768  /* write output signal of current channel to output buffer */
769  dst[2 * j] += fft_in[j].re / (float)n_fft;
770  }
771 
772  for (j = 0; j < n_samples - 1; j++) { /* overflow length is IR length - 1 */
773  /* write the rest of output signal to overflow buffer */
774  int write_pos = (wr + j) & modulo;
775 
776  *(ringbuffer + write_pos) += fft_in[in->nb_samples + j].re / (float)n_fft;
777  }
778  }
779 
780  /* go through all samples of current output buffer: count clippings */
781  for (i = 0; i < out->nb_samples; i++) {
782  /* clippings counter */
783  if (fabs(*dst) > 1) { /* if current output sample > 1 */
784  *n_clippings = *n_clippings + 1;
785  }
786 
787  /* move output buffer pointer by +2 to get to next sample of processed channel: */
788  dst += 2;
789  }
790 
791  /* remember read/write position in ringbuffer for next call */
792  *write = wr;
793 
794  return 0;
795 }

873 static int load_data(AVFilterContext *ctx, int azim, int elev, float radius)
874 {
875  struct SOFAlizerContext *s = ctx->priv;
876  const int n_samples = s->sofa.n_samples;
877  int n_conv = s->n_conv; /* no. channels to convolve */
878  int n_fft = s->n_fft;
879  int delay_l[16]; /* broadband delay for each IR */
880  int delay_r[16];
881  int nb_input_channels = ctx->inputs[0]->channels; /* no. input channels */
882  float gain_lin = expf((s->gain - 3 * nb_input_channels) / 20 * M_LN10); /* gain - 3dB/channel */
883  FFTComplex *data_hrtf_l = NULL;
884  FFTComplex *data_hrtf_r = NULL;
885  FFTComplex *fft_in_l = NULL;
886  FFTComplex *fft_in_r = NULL;
887  float *data_ir_l = NULL;
888  float *data_ir_r = NULL;
889  int offset = 0; /* used for faster pointer arithmetics in for-loop */
890  int m[16]; /* measurement index m of IR closest to required source positions */
891  int i, j, azim_orig = azim, elev_orig = elev;
892 
893  if (!s->sofa.ncid) { /* if an invalid SOFA file has been selected */
894  av_log(ctx, AV_LOG_ERROR, "Selected SOFA file is invalid. Please select valid SOFA file.\n");
895  return AVERROR_INVALIDDATA;
896  }
897 
898  if (s->type == TIME_DOMAIN) {
899  s->temp_src[0] = av_calloc(FFALIGN(n_samples, 16), sizeof(float));
900  s->temp_src[1] = av_calloc(FFALIGN(n_samples, 16), sizeof(float));
901 
902  /* get temporary IR for L and R channel */
903  data_ir_l = av_calloc(n_conv * FFALIGN(n_samples, 16), sizeof(*data_ir_l));
904  data_ir_r = av_calloc(n_conv * FFALIGN(n_samples, 16), sizeof(*data_ir_r));
905  if (!data_ir_r || !data_ir_l || !s->temp_src[0] || !s->temp_src[1]) {
906  av_free(data_ir_l);
907  av_free(data_ir_r);
908  return AVERROR(ENOMEM);
909  }
910  } else {
911  /* get temporary HRTF memory for L and R channel */
912  data_hrtf_l = av_malloc_array(n_fft, sizeof(*data_hrtf_l) * n_conv);
913  data_hrtf_r = av_malloc_array(n_fft, sizeof(*data_hrtf_r) * n_conv);
914  if (!data_hrtf_r || !data_hrtf_l) {
915  av_free(data_hrtf_l);
916  av_free(data_hrtf_r);
917  return AVERROR(ENOMEM);
918  }
919  }
920 
921  for (i = 0; i < s->n_conv; i++) {
922  /* load and store IRs and corresponding delays */
923  azim = (int)(s->speaker_azim[i] + azim_orig) % 360;
924  elev = (int)(s->speaker_elev[i] + elev_orig) % 90;
925  /* get id of IR closest to desired position */
926  m[i] = find_m(s, azim, elev, radius);
927 
928  /* load the delays associated with the current IRs */
929  delay_l[i] = *(s->sofa.data_delay + 2 * m[i]);
930  delay_r[i] = *(s->sofa.data_delay + 2 * m[i] + 1);
931 
932  if (s->type == TIME_DOMAIN) {
933  offset = i * FFALIGN(n_samples, 16); /* no. samples already written */
934  for (j = 0; j < n_samples; j++) {
935  /* load reversed IRs of the specified source position
936  * sample-by-sample for left and right ear; and apply gain */
937  *(data_ir_l + offset + j) = /* left channel */
938  *(s->sofa.data_ir + 2 * m[i] * n_samples + n_samples - 1 - j) * gain_lin;
939  *(data_ir_r + offset + j) = /* right channel */
940  *(s->sofa.data_ir + 2 * m[i] * n_samples + n_samples - 1 - j + n_samples) * gain_lin;
941  }
942  } else {
943  fft_in_l = av_calloc(n_fft, sizeof(*fft_in_l));
944  fft_in_r = av_calloc(n_fft, sizeof(*fft_in_r));
945  if (!fft_in_l || !fft_in_r) {
946  av_free(data_hrtf_l);
947  av_free(data_hrtf_r);
948  av_free(fft_in_l);
949  av_free(fft_in_r);
950  return AVERROR(ENOMEM);
951  }
952 
953  offset = i * n_fft; /* no. samples already written */
954  for (j = 0; j < n_samples; j++) {
955  /* load non-reversed IRs of the specified source position
956  * sample-by-sample and apply gain,
957  * L channel is loaded to real part, R channel to imag part,
958  * IRs ared shifted by L and R delay */
959  fft_in_l[delay_l[i] + j].re = /* left channel */
960  *(s->sofa.data_ir + 2 * m[i] * n_samples + j) * gain_lin;
961  fft_in_r[delay_r[i] + j].re = /* right channel */
962  *(s->sofa.data_ir + (2 * m[i] + 1) * n_samples + j) * gain_lin;
963  }
964 
965  /* actually transform to frequency domain (IRs -> HRTFs) */
966  av_fft_permute(s->fft[0], fft_in_l);
967  av_fft_calc(s->fft[0], fft_in_l);
968  memcpy(data_hrtf_l + offset, fft_in_l, n_fft * sizeof(*fft_in_l));
969  av_fft_permute(s->fft[0], fft_in_r);
970  av_fft_calc(s->fft[0], fft_in_r);
971  memcpy(data_hrtf_r + offset, fft_in_r, n_fft * sizeof(*fft_in_r));
972  }
973 
974  av_log(ctx, AV_LOG_DEBUG, "Index: %d, Azimuth: %f, Elevation: %f, Radius: %f of SOFA file.\n",
975  m[i], *(s->sofa.sp_a + m[i]), *(s->sofa.sp_e + m[i]), *(s->sofa.sp_r + m[i]));
976  }
977 
978  if (s->type == TIME_DOMAIN) {
979  /* copy IRs and delays to allocated memory in the SOFAlizerContext struct: */
980  memcpy(s->data_ir[0], data_ir_l, sizeof(float) * n_conv * FFALIGN(n_samples, 16));
981  memcpy(s->data_ir[1], data_ir_r, sizeof(float) * n_conv * FFALIGN(n_samples, 16));
982 
983  av_freep(&data_ir_l); /* free temporary IR memory */
984  av_freep(&data_ir_r);
985  } else {
986  s->data_hrtf[0] = av_malloc_array(n_fft * s->n_conv, sizeof(FFTComplex));
987  s->data_hrtf[1] = av_malloc_array(n_fft * s->n_conv, sizeof(FFTComplex));
988  if (!s->data_hrtf[0] || !s->data_hrtf[1]) {
989  av_freep(&data_hrtf_l);
990  av_freep(&data_hrtf_r);
991  av_freep(&fft_in_l);
992  av_freep(&fft_in_r);
993  return AVERROR(ENOMEM); /* memory allocation failed */
994  }
995 
996  memcpy(s->data_hrtf[0], data_hrtf_l, /* copy HRTF data to */
997  sizeof(FFTComplex) * n_conv * n_fft); /* filter struct */
998  memcpy(s->data_hrtf[1], data_hrtf_r,
999  sizeof(FFTComplex) * n_conv * n_fft);
1000 
1001  av_freep(&data_hrtf_l); /* free temporary HRTF memory */
1002  av_freep(&data_hrtf_r);
1003 
1004  av_freep(&fft_in_l); /* free temporary FFT memory */
1005  av_freep(&fft_in_r);
1006  }
1007 
1008  memcpy(s->delay[0], &delay_l[0], sizeof(int) * s->n_conv);
1009  memcpy(s->delay[1], &delay_r[0], sizeof(int) * s->n_conv);
1010 
1011  return 0;
1012 }

1049 static int config_input(AVFilterLink *inlink)

1062  inlink->max_samples = inlink->sample_rate;
1063  }

1066  s->gain_lfe = expf((s->gain - 3 * inlink->channels - 6) / 20 * M_LN10);
1067

1073  if (n_current > n_max) {

1076  /* length of longest IR (without delay, in all SOFA files) */
1077  n_max_ir = s->sofa.n_samples;
