347  mbstate_t state; /* Conversion State */
348 
349  /* In case SRC or DESP is NULL, conversion doesn't take place. */
350  if (src == NULL || destp == NULL) {

352  *destp = NULL;

354  }

357  wsbuf_size = WSBUF_INC;

360  if (wsbuf == NULL) {
361  *destp = NULL;
362  return (size_t) -1;
363  }
364

366  if (indices == NULL) {
367  free(wsbuf);
368  *destp = NULL;
369  return (size_t) -1;
370  }

372  p = src;
373  wcnum = 0;

379  wc = L'\0';
380  mblength = 1;
381  } else if (*p == '\\') {
382  wc = L'\\';
383  mblength = 1;
384  } else
385  mblength = mbrtowc(&wc, p, MB_LEN_MAX, &state);
