44 static int rsd_probe(AVProbeData *p)
45 {
46  if (memcmp(p->buf, "RSD", 3) || p->buf[3] - '0' < 2 || p->buf[3] - '0' > 6)
47  return 0;
48  if (AV_RL32(p->buf + 8) > 256 || !AV_RL32(p->buf + 8))
49  return AVPROBE_SCORE_MAX / 8;
50  if (AV_RL32(p->buf + 16) > 8*48000 || !AV_RL32(p->buf + 16))
51  return AVPROBE_SCORE_MAX / 8;
52  return AVPROBE_SCORE_MAX;
53 }

55 static int rsd_read_header(AVFormatContext *s)
56 {
57  AVIOContext *pb = s->pb;
58  int i, ret, version, start = 0x800;
59  AVCodecParameters *par;
60  AVStream *st = avformat_new_stream(s, NULL);
61 
62  if (!st)
63  return AVERROR(ENOMEM);
64 
65  avio_skip(pb, 3); // "RSD"
66  version = avio_r8(pb) - '0';
67 
68  par = st->codecpar;
69  par->codec_type = AVMEDIA_TYPE_AUDIO;
70  par->codec_tag = avio_rl32(pb);
71  par->codec_id = ff_codec_get_id(rsd_tags, par->codec_tag);
72  if (!par->codec_id) {
73  char tag_buf[32];
74 
75  av_get_codec_tag_string(tag_buf, sizeof(tag_buf), par->codec_tag);
76  for (i=0; i < FF_ARRAY_ELEMS(rsd_unsupported_tags); i++) {
77  if (par->codec_tag == rsd_unsupported_tags[i]) {
78  avpriv_request_sample(s, "Codec tag: %s", tag_buf);
79  return AVERROR_PATCHWELCOME;
80  }
81  }
82  av_log(s, AV_LOG_ERROR, "Unknown codec tag: %s\n", tag_buf);
83  return AVERROR_INVALIDDATA;
84  }
85 
86  par->channels = avio_rl32(pb);
87  if (par->channels <= 0 || par->channels > INT_MAX / 36) {
88  av_log(s, AV_LOG_ERROR, "Invalid number of channels: %d\n", par->channels);
89  return AVERROR_INVALIDDATA;
90  }
91 
92  avio_skip(pb, 4); // Bit depth
93  par->sample_rate = avio_rl32(pb);
94  if (!par->sample_rate)
95  return AVERROR_INVALIDDATA;
96 
97  avio_skip(pb, 4); // Unknown
98 
99  switch (par->codec_id) {
100  case AV_CODEC_ID_XMA2:
101  par->block_align = 2048;
102  ff_alloc_extradata(par, 34);
103  if (!par->extradata)
104  return AVERROR(ENOMEM);
105  memset(par->extradata, 0, 34);
106  break;
107  case AV_CODEC_ID_ADPCM_PSX:
108  par->block_align = 16 * par->channels;
109  if (pb->seekable)
110  st->duration = av_get_audio_frame_duration2(par, avio_size(pb) - start);
111  break;
112  case AV_CODEC_ID_ADPCM_IMA_RAD:
113  par->block_align = 20 * par->channels;
114  if (pb->seekable)
115  st->duration = av_get_audio_frame_duration2(par, avio_size(pb) - start);
116  break;
117  case AV_CODEC_ID_ADPCM_IMA_WAV:
118  if (version == 2)
119  start = avio_rl32(pb);
120 
121  par->bits_per_coded_sample = 4;
122  par->block_align = 36 * par->channels;
123  if (pb->seekable)
124  st->duration = av_get_audio_frame_duration2(par, avio_size(pb) - start);
125  break;
126  case AV_CODEC_ID_ADPCM_THP_LE:
127  /* RSD3GADP is mono, so only alloc enough memory
128  to store the coeff table for a single channel. */
129 
130  start = avio_rl32(pb);
131 
132  if ((ret = ff_get_extradata(s, par, s->pb, 32)) < 0)
133  return ret;
134  if (pb->seekable)
135  st->duration = av_get_audio_frame_duration2(par, avio_size(pb) - start);
136  break;
137  case AV_CODEC_ID_ADPCM_THP:
138  par->block_align = 8 * par->channels;
139  avio_skip(s->pb, 0x1A4 - avio_tell(s->pb));
140 
141  if ((ret = ff_alloc_extradata(st->codecpar, 32 * par->channels)) < 0)
142  return ret;
143 
144  for (i = 0; i < par->channels; i++) {
145  avio_read(s->pb, st->codecpar->extradata + 32 * i, 32);
146  avio_skip(s->pb, 8);
147  }
148  if (pb->seekable)
149  st->duration = (avio_size(pb) - start) / (8 * par->channels) * 14;
150  break;
151  case AV_CODEC_ID_PCM_S16LE:
152  case AV_CODEC_ID_PCM_S16BE:
153  if (version != 4)
154  start = avio_rl32(pb);
155 
156  if (pb->seekable)
157  st->duration = (avio_size(pb) - start) / 2 / par->channels;
158  break;
159  }
160 
161  avio_skip(pb, start - avio_tell(pb));
162  if (par->codec_id == AV_CODEC_ID_XMA2) {
163  avio_skip(pb, avio_rb32(pb) + avio_rb32(pb));
164  st->duration = avio_rb32(pb);
165  }
166 
167  avpriv_set_pts_info(st, 64, 1, par->sample_rate);
168 
169  return 0;
170 }

172 static int rsd_read_packet(AVFormatContext *s, AVPacket *pkt)
173 {
174  AVCodecParameters *par = s->streams[0]->codecpar;
175  int ret, size = 1024;
176  int64_t pos;
177 
178  if (avio_feof(s->pb))
179  return AVERROR_EOF;
180 
181  pos = avio_tell(s->pb);
182  if (par->codec_id == AV_CODEC_ID_ADPCM_IMA_RAD ||
183  par->codec_id == AV_CODEC_ID_ADPCM_PSX ||
184  par->codec_id == AV_CODEC_ID_ADPCM_IMA_WAV ||
185  par->codec_id == AV_CODEC_ID_XMA2) {
186  ret = av_get_packet(s->pb, pkt, par->block_align);
187  } else if (par->codec_tag == MKTAG('W','A','D','P') &&
188  par->channels > 1) {
189  int i, ch;
190 
191  ret = av_new_packet(pkt, par->block_align);
192  if (ret < 0)
193  return ret;
194  for (i = 0; i < 4; i++) {
195  for (ch = 0; ch < par->channels; ch++) {
196  pkt->data[ch * 8 + i * 2 + 0] = avio_r8(s->pb);
197  pkt->data[ch * 8 + i * 2 + 1] = avio_r8(s->pb);
198  }
199  }
200  ret = 0;
201  } else {
202  ret = av_get_packet(s->pb, pkt, size);
203  }
204 
205  if (par->codec_id == AV_CODEC_ID_XMA2 && pkt->size >= 1)
206  pkt->duration = (pkt->data[0] >> 2) * 512;
207 
208  pkt->pos = pos;
209  pkt->stream_index = 0;
210 
211  return ret;
212 }
