33 static void parse_mb_skip(Wmv2Context *w)
34 {
35  int mb_x, mb_y;
36  MpegEncContext *const s = &w->s;
37  uint32_t *const mb_type = s->current_picture_ptr->mb_type;
38 
39  w->skip_type = get_bits(&s->gb, 2);
40  switch (w->skip_type) {
41  case SKIP_TYPE_NONE:
42  for (mb_y = 0; mb_y < s->mb_height; mb_y++)
43  for (mb_x = 0; mb_x < s->mb_width; mb_x++)
44  mb_type[mb_y * s->mb_stride + mb_x] =
45  MB_TYPE_16x16 | MB_TYPE_L0;
46  break;
47  case SKIP_TYPE_MPEG:
48  for (mb_y = 0; mb_y < s->mb_height; mb_y++)
49  for (mb_x = 0; mb_x < s->mb_width; mb_x++)
50  mb_type[mb_y * s->mb_stride + mb_x] =
51  (get_bits1(&s->gb) ? MB_TYPE_SKIP : 0) | MB_TYPE_16x16 | MB_TYPE_L0;
52  break;
53  case SKIP_TYPE_ROW:
54  for (mb_y = 0; mb_y < s->mb_height; mb_y++) {
55  if (get_bits1(&s->gb)) {
56  for (mb_x = 0; mb_x < s->mb_width; mb_x++)
57  mb_type[mb_y * s->mb_stride + mb_x] =
58  MB_TYPE_SKIP | MB_TYPE_16x16 | MB_TYPE_L0;
59  } else {
60  for (mb_x = 0; mb_x < s->mb_width; mb_x++)
61  mb_type[mb_y * s->mb_stride + mb_x] =
62  (get_bits1(&s->gb) ? MB_TYPE_SKIP : 0) | MB_TYPE_16x16 | MB_TYPE_L0;
63  }
64  }
65  break;
66  case SKIP_TYPE_COL:
67  for (mb_x = 0; mb_x < s->mb_width; mb_x++) {
68  if (get_bits1(&s->gb)) {
69  for (mb_y = 0; mb_y < s->mb_height; mb_y++)
70  mb_type[mb_y * s->mb_stride + mb_x] =
71  MB_TYPE_SKIP | MB_TYPE_16x16 | MB_TYPE_L0;
72  } else {
73  for (mb_y = 0; mb_y < s->mb_height; mb_y++)
74  mb_type[mb_y * s->mb_stride + mb_x] =
75  (get_bits1(&s->gb) ? MB_TYPE_SKIP : 0) | MB_TYPE_16x16 | MB_TYPE_L0;
76  }
77  }
78  break;
79  }
80 }

82 static int decode_ext_header(Wmv2Context *w)
83 {
84  MpegEncContext *const s = &w->s;
85  GetBitContext gb;
86  int fps;
87  int code;
88 
89  if (s->avctx->extradata_size < 4)
90  return AVERROR_INVALIDDATA;
91 
92  init_get_bits(&gb, s->avctx->extradata, 32);
93 
94  fps = get_bits(&gb, 5);
95  s->bit_rate = get_bits(&gb, 11) * 1024;
96  w->mspel_bit = get_bits1(&gb);
97  s->loop_filter = get_bits1(&gb);
98  w->abt_flag = get_bits1(&gb);
99  w->j_type_bit = get_bits1(&gb);
100  w->top_left_mv_flag = get_bits1(&gb);
101  w->per_mb_rl_bit = get_bits1(&gb);
102  code = get_bits(&gb, 3);
103 
104  if (code == 0)
105  return AVERROR_INVALIDDATA;
106 
107  s->slice_height = s->mb_height / code;
108 
109  if (s->avctx->debug & FF_DEBUG_PICT_INFO)
110  av_log(s->avctx, AV_LOG_DEBUG,
111  "fps:%d, br:%"PRId64", qpbit:%d, abt_flag:%d, j_type_bit:%d, "
112  "tl_mv_flag:%d, mbrl_bit:%d, code:%d, loop_filter:%d, "
113  "slices:%d\n",
114  fps, s->bit_rate, w->mspel_bit, w->abt_flag, w->j_type_bit,
115  w->top_left_mv_flag, w->per_mb_rl_bit, code, s->loop_filter,
116  code);
117  return 0;
118 }

121 {
122  Wmv2Context *const w = (Wmv2Context *) s;
123  int code;
124 
125  if (s->picture_number == 0)
126  decode_ext_header(w);
127 
128  s->pict_type = get_bits1(&s->gb) + 1;
129  if (s->pict_type == AV_PICTURE_TYPE_I) {

202

204  s->no_rounding ^= 1;

214  s->inter_intra_pred);
215  }
216  }
217  s->esc3_level_length = 0;
218  s->esc3_run_length = 0;
219  s->picture_number++; // FIXME ?
220 
221  if (w->j_type) {
222  ff_intrax8_decode_picture(&w->x8, &s->current_picture,
223  &s->gb, &s->mb_x, &s->mb_y,
224  2 * s->qscale, (s->qscale - 1) | 1,
225  s->loop_filter, s->low_delay);
226 
227  ff_er_add_slice(&w->s.er, 0, 0,
228  (w->s.mb_x >> 1) - 1, (w->s.mb_y >> 1) - 1,
229  ER_MB_END);
230  return 1;
231  }
232 
233  return 0;

254 static int16_t *wmv2_pred_motion(Wmv2Context *w, int *px, int *py)
255 {
256  MpegEncContext *const s = &w->s;
257  int xy, wrap, diff, type;
258  int16_t *A, *B, *C, *mot_val;
259 
260  wrap = s->b8_stride;
261  xy = s->block_index[0];
262 
263  mot_val = s->current_picture.motion_val[0][xy];
264 
265  A = s->current_picture.motion_val[0][xy - 1];
266  B = s->current_picture.motion_val[0][xy - wrap];
267  C = s->current_picture.motion_val[0][xy + 2 - wrap];
268 
269  if (s->mb_x && !s->first_slice_line && !s->mspel && w->top_left_mv_flag)
270  diff = FFMAX(FFABS(A[0] - B[0]), FFABS(A[1] - B[1]));
271  else
272  diff = 0;
273 
274  if (diff >= 8)
275  type = get_bits1(&s->gb);
276  else
277  type = 2;
278 
279  if (type == 0) {
280  *px = A[0];
281  *py = A[1];
282  } else if (type == 1) {
283  *px = B[0];
284  *py = B[1];
285  } else {
286  /* special case for first (slice) line */
287  if (s->first_slice_line) {
288  *px = A[0];
289  *py = A[1];
290  } else {
291  *px = mid_pred(A[0], B[0], C[0]);
292  *py = mid_pred(A[1], B[1], C[1]);
293  }
294  }
295 
296  return mot_val;
297 }

299 static inline int wmv2_decode_inter_block(Wmv2Context *w, int16_t *block,
300  int n, int cbp)
301 {
302  MpegEncContext *const s = &w->s;
303  static const int sub_cbp_table[3] = { 2, 3, 1 };
304  int sub_cbp, ret;
305 
306  if (!cbp) {
307  s->block_last_index[n] = -1;
308  return 0;
309  }
310 
311  if (w->per_block_abt)
312  w->abt_type = decode012(&s->gb);
313  w->abt_type_table[n] = w->abt_type;
314 
315  if (w->abt_type) {
316 // const uint8_t *scantable = w->abt_scantable[w->abt_type - 1].permutated;
317  const uint8_t *scantable = w->abt_scantable[w->abt_type - 1].scantable;
318 // const uint8_t *scantable = w->abt_type - 1 ? w->abt_scantable[1].permutated : w->abt_scantable[0].scantable;
319 
320  sub_cbp = sub_cbp_table[decode012(&s->gb)];
321 
322  if (sub_cbp & 1)
323  if ((ret = ff_msmpeg4_decode_block(s, block, n, 1, scantable)) < 0)
324  return ret;
325 
326  if (sub_cbp & 2)
327  if ((ret = ff_msmpeg4_decode_block(s, w->abt_block2[n], n, 1, scantable)) < 0)
328  return ret;
329 
330  s->block_last_index[n] = 63;
331 
332  return 0;
333  } else {
334  return ff_msmpeg4_decode_block(s, block, n, 1,
335  s->inter_scantable.permutated);
336  }
337 }

339 int ff_wmv2_decode_mb(MpegEncContext *s, int16_t block[6][64])
340 {
341  Wmv2Context *const w = (Wmv2Context *) s;
342  int cbp, code, i, ret;
343  uint8_t *coded_val;
344 
345  if (w->j_type)
346  return 0;
347 
348  if (s->pict_type == AV_PICTURE_TYPE_P) {
349  if (IS_SKIP(s->current_picture.mb_type[s->mb_y * s->mb_stride + s->mb_x])) {
350  /* skip mb */
351  s->mb_intra = 0;
352  for (i = 0; i < 6; i++)
353  s->block_last_index[i] = -1;
354  s->mv_dir = MV_DIR_FORWARD;
355  s->mv_type = MV_TYPE_16X16;
356  s->mv[0][0][0] = 0;
357  s->mv[0][0][1] = 0;
358  s->mb_skipped = 1;
359  w->hshift = 0;
360  return 0;
361  }
362 
363  code = get_vlc2(&s->gb, ff_mb_non_intra_vlc[w->cbp_table_index].table,
364  MB_NON_INTRA_VLC_BITS, 3);
365  if (code < 0)
366  return AVERROR_INVALIDDATA;
367  s->mb_intra = (~code & 0x40) >> 6;
368 
369  cbp = code & 0x3f;
370  } else {
371  s->mb_intra = 1;
372  code = get_vlc2(&s->gb, ff_msmp4_mb_i_vlc.table, MB_INTRA_VLC_BITS, 2);
373  if (code < 0) {
374  av_log(s->avctx, AV_LOG_ERROR,
375  "II-cbp illegal at %d %d\n", s->mb_x, s->mb_y);
376  return AVERROR_INVALIDDATA;
377  }
378  /* predict coded block pattern */
379  cbp = 0;
380  for (i = 0; i < 6; i++) {
381  int val = ((code >> (5 - i)) & 1);
382  if (i < 4) {
383  int pred = ff_msmpeg4_coded_block_pred(s, i, &coded_val);
384  val = val ^ pred;
385  *coded_val = val;
386  }
387  cbp |= val << (5 - i);
388  }
389  }
390 
391  if (!s->mb_intra) {
392  int mx, my;
393  wmv2_pred_motion(w, &mx, &my);
394 
395  if (cbp) {
396  s->bdsp.clear_blocks(s->block[0]);
397  if (s->per_mb_rl_table) {
398  s->rl_table_index = decode012(&s->gb);
399  s->rl_chroma_table_index = s->rl_table_index;
400  }
401 
402  if (w->abt_flag && w->per_mb_abt) {
403  w->per_block_abt = get_bits1(&s->gb);
404  if (!w->per_block_abt)
405  w->abt_type = decode012(&s->gb);
406  } else
407  w->per_block_abt = 0;
408  }
409 
410  if ((ret = wmv2_decode_motion(w, &mx, &my)) < 0)
411  return ret;
412 
413  s->mv_dir = MV_DIR_FORWARD;
414  s->mv_type = MV_TYPE_16X16;
415  s->mv[0][0][0] = mx;
416  s->mv[0][0][1] = my;
417 
418  for (i = 0; i < 6; i++) {
419  if ((ret = wmv2_decode_inter_block(w, block[i], i, (cbp >> (5 - i)) & 1)) < 0) {
420  av_log(s->avctx, AV_LOG_ERROR,
421  "\nerror while decoding inter block: %d x %d (%d)\n",
422  s->mb_x, s->mb_y, i);
423  return ret;
424  }
425  }
426  } else {
427  if (s->pict_type == AV_PICTURE_TYPE_P)
428  ff_dlog(s->avctx, "%d%d ", s->inter_intra_pred, cbp);
429  ff_dlog(s->avctx, "I at %d %d %d %06X\n", s->mb_x, s->mb_y,
430  ((cbp & 3) ? 1 : 0) + ((cbp & 0x3C) ? 2 : 0),
431  show_bits(&s->gb, 24));
432  s->ac_pred = get_bits1(&s->gb);
433  if (s->inter_intra_pred) {
434  s->h263_aic_dir = get_vlc2(&s->gb, ff_inter_intra_vlc.table,
435  INTER_INTRA_VLC_BITS, 1);
436  ff_dlog(s->avctx, "%d%d %d %d/",
437  s->ac_pred, s->h263_aic_dir, s->mb_x, s->mb_y);
438  }
439  if (s->per_mb_rl_table && cbp) {
440  s->rl_table_index = decode012(&s->gb);
441  s->rl_chroma_table_index = s->rl_table_index;
442  }
443 
444  s->bdsp.clear_blocks(s->block[0]);
445  for (i = 0; i < 6; i++) {
446  if ((ret = ff_msmpeg4_decode_block(s, block[i], i, (cbp >> (5 - i)) & 1, NULL)) < 0) {
447  av_log(s->avctx, AV_LOG_ERROR,
448  "\nerror while decoding intra block: %d x %d (%d)\n",
449  s->mb_x, s->mb_y, i);
450  return ret;
451  }
452  }
453  }
454 
455  return 0;
456 }
