101 static inline int h263_get_motion_length(int val, int f_code){
102  int l, bit_size, code;
103 
104  if (val == 0) {
105  return ff_mvtab[0][1];
106  } else {
107  bit_size = f_code - 1;
108  /* modulo encoding */
109  l= INT_BIT - 6 - bit_size;
110  val = (val<<l)>>l;
111  val--;
112  code = (val >> bit_size) + 1;
113 
114  return ff_mvtab[code][1] + 1 + bit_size;
115  }
116 }

129 static inline int get_p_cbp(MpegEncContext * s,
130  int16_t block[6][64],
131  int motion_x, int motion_y){
132  int cbp, i;
133 
134  if (s->mpv_flags & FF_MPV_FLAG_CBP_RD) {
135  int best_cbpy_score= INT_MAX;
136  int best_cbpc_score= INT_MAX;
137  int cbpc = (-1), cbpy= (-1);
138  const int offset= (s->mv_type==MV_TYPE_16X16 ? 0 : 16) + (s->dquant ? 8 : 0);
139  const int lambda= s->lambda2 >> (FF_LAMBDA_SHIFT - 6);
140 
141  for(i=0; i<4; i++){
142  int score= ff_h263_inter_MCBPC_bits[i + offset] * lambda;
143  if(i&1) score += s->coded_score[5];
144  if(i&2) score += s->coded_score[4];
145 
146  if(score < best_cbpc_score){
147  best_cbpc_score= score;
148  cbpc= i;
149  }
150  }
151 
152  for(i=0; i<16; i++){
153  int score= ff_h263_cbpy_tab[i ^ 0xF][1] * lambda;
154  if(i&1) score += s->coded_score[3];
155  if(i&2) score += s->coded_score[2];
156  if(i&4) score += s->coded_score[1];
157  if(i&8) score += s->coded_score[0];
158 
159  if(score < best_cbpy_score){
160  best_cbpy_score= score;
161  cbpy= i;
162  }
163  }
164  cbp= cbpc + 4*cbpy;
165  if ((motion_x | motion_y | s->dquant) == 0 && s->mv_type==MV_TYPE_16X16){
166  if(best_cbpy_score + best_cbpc_score + 2*lambda >= 0)
167  cbp= 0;
168  }
169 
170  for (i = 0; i < 6; i++) {
171  if (s->block_last_index[i] >= 0 && ((cbp >> (5 - i))&1)==0 ){
172  s->block_last_index[i]= -1;
173  s->bdsp.clear_block(s->block[i]);
174  }
175  }
176  }else{
177  cbp= 0;
178  for (i = 0; i < 6; i++) {
179  if (s->block_last_index[i] >= 0)
180  cbp |= 1 << (5 - i);
181  }
182  }
183  return cbp;
184 }
