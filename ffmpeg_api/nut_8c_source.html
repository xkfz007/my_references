240 void ff_nut_reset_ts(NUTContext *nut, AVRational time_base, int64_t val)
241 {
242  int i;
243  for (i = 0; i < nut->avf->nb_streams; i++)
244  nut->stream[i].last_pts =
245  av_rescale_rnd(val,
246  time_base.num * (int64_t)nut->stream[i].time_base->den,
247  time_base.den * (int64_t)nut->stream[i].time_base->num,
248  AV_ROUND_DOWN);
249 }

251 int64_t ff_lsb2full(StreamContext *stream, int64_t lsb)
252 {
253  int64_t mask = (1ULL << stream->msb_pts_shift) - 1;
254  int64_t delta = stream->last_pts - mask / 2;
255  return ((lsb - delta) & mask) + delta;
256 }

270 int ff_nut_add_sp(NUTContext *nut, int64_t pos, int64_t back_ptr, int64_t ts)
271 {
272  Syncpoint *sp = av_mallocz(sizeof(Syncpoint));
273  struct AVTreeNode *node = av_tree_node_alloc();
274 
275  if (!sp || !node) {
276  av_freep(&sp);
277  av_freep(&node);
278  return AVERROR(ENOMEM);
279  }
280 
281  nut->sp_count++;
282 
283  sp->pos = pos;
284  sp->back_ptr = back_ptr;
285  sp->ts = ts;
286  av_tree_insert(&nut->syncpoints, sp, ff_nut_sp_pos_cmp, &node);
287  if (node) {
288  av_free(sp);
289  av_free(node);
290  }
291 
292  return 0;
293 }
