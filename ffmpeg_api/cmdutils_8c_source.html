143 double parse_number_or_die(const char *context, const char *numstr, int type,
144  double min, double max)
145 {
146  char *tail;
147  const char *error;
148  double d = av_strtod(numstr, &tail);
149  if (*tail)
150  error = "Expected number for %s but found: %s\n";
151  else if (d < min || d > max)
152  error = "The value for %s was %s which is not within %f - %f\n";
153  else if (type == OPT_INT64 && (int64_t)d != d)
154  error = "Expected int64 for %s but found %s\n";
155  else if (type == OPT_INT && (int)d != d)
156  error = "Expected int for %s but found %s\n";
157  else
158  return d;
159  av_log(NULL, AV_LOG_FATAL, error, context, numstr, min, max);
160  exit_program(1);
161  return 0;
162 }

248 static void prepare_app_arguments(int *argc_ptr, char ***argv_ptr)
249 {

252  int i, buffsize = 0, offset = 0;
253

262  if (win32_argc <= 0 || !argv_w)

266  for (i = 0; i < win32_argc; i++)

268  NULL, 0, NULL, NULL);

271  argstr_flat = (char *)win32_argv_utf8 + sizeof(char *) * (win32_argc + 1);
272  if (!win32_argv_utf8) {

274  return;
275  }
276

280  &argstr_flat[offset],
281  buffsize - offset, NULL, NULL);
282  }
283  win32_argv_utf8[i] = NULL;

285

287  *argv_ptr = win32_argv_utf8;

296 static int write_option(void *optctx, const OptionDef *po, const char *opt,
297  const char *arg)
298 {
299  /* new-style options contain an offset into optctx, old-style address of
300  * a global var*/
301  //+:OPT_OFFSET and OPT_SPEC imply this is an new-style option
302  //+:and u.off is used. Or u.dst_ptr is used to store option argument
303  void *dst = po->flags & (OPT_OFFSET | OPT_SPEC) ?
304  (uint8_t *)optctx + po->u.off : po->u.dst_ptr;
305  int *dstcount;
306 
307  if (po->flags & OPT_SPEC) {
308  SpecifierOpt **so = dst;
309  char *p = strchr(opt, ':');
310  char *str;
311 
312  dstcount = (int *)(so + 1);
313  *so = grow_array(*so, sizeof(**so), dstcount, *dstcount + 1);
314  str = av_strdup(p ? p + 1 : "");
315  if (!str)
316  return AVERROR(ENOMEM);
317  (*so)[*dstcount - 1].specifier = str;
318  dst = &(*so)[*dstcount - 1].u;
319  }
320 
321  if (po->flags & OPT_STRING) {
322  char *str;
323  str = av_strdup(arg);
324  av_freep(dst);
325  if (!str)
326  return AVERROR(ENOMEM);
327  *(char **)dst = str;
328  } else if (po->flags & OPT_BOOL || po->flags & OPT_INT) {
329  *(int *)dst = parse_number_or_die(opt, arg, OPT_INT64, INT_MIN, INT_MAX);
330  } else if (po->flags & OPT_INT64) {
331  *(int64_t *)dst = parse_number_or_die(opt, arg, OPT_INT64, INT64_MIN, INT64_MAX);
332  } else if (po->flags & OPT_TIME) {
333  *(int64_t *)dst = parse_time_or_die(opt, arg, 1);
334  } else if (po->flags & OPT_FLOAT) {
335  *(float *)dst = parse_number_or_die(opt, arg, OPT_FLOAT, -INFINITY, INFINITY);
336  } else if (po->flags & OPT_DOUBLE) {
337  *(double *)dst = parse_number_or_die(opt, arg, OPT_DOUBLE, -INFINITY, INFINITY);
338  } else if (po->u.func_arg) {
339  int ret = po->u.func_arg(optctx, opt, arg);
340  if (ret < 0) {
341  av_log(NULL, AV_LOG_ERROR,
342  "Failed to set value '%s' for option '%s': %s\n",
343  arg, opt, av_err2str(ret));
344  return ret;
345  }
346  }
347  if (po->flags & OPT_EXIT)
348  exit_program(0);
349 
350  return 0;
351 }

353 int parse_option(void *optctx, const char *opt, const char *arg,
354  const OptionDef *options)
355 {
356  const OptionDef *po;
357  int ret;
358 
359  po = find_option(options, opt);
360  if (!po->name && opt[0] == 'n' && opt[1] == 'o') {
361  /* handle 'no' bool option */
362  po = find_option(options, opt + 2);
363  if ((po->name && (po->flags & OPT_BOOL)))
364  arg = "0";
365  } else if (po->flags & OPT_BOOL)
366  arg = "1";
367 
368  if (!po->name)
369  po = find_option(options, "default");
370  if (!po->name) {
371  av_log(NULL, AV_LOG_ERROR, "Unrecognized option '%s'\n", opt);
372  return AVERROR(EINVAL);
373  }
374  if (po->flags & HAS_ARG && !arg) {
375  av_log(NULL, AV_LOG_ERROR, "Missing argument for option '%s'\n", opt);
376  return AVERROR(EINVAL);
377  }
378 
379  ret = write_option(optctx, po, opt, arg);
380  if (ret < 0)
381  return ret;
382 
383  return !!(po->flags & HAS_ARG);
384 }

386 void parse_options(void *optctx, int argc, char **argv, const OptionDef *options,
387  void (*parse_arg_function)(void *, const char*))
388 {
389  const char *opt;
390  int optindex, handleoptions = 1, ret;
391 
392  /* perform system-dependent conversions for arguments list */
393  prepare_app_arguments(&argc, &argv);
394 
395  /* parse options */
396  optindex = 1;
397  while (optindex < argc) {
398  opt = argv[optindex++];
399 
400  if (handleoptions && opt[0] == '-' && opt[1] != '\0') {
401  if (opt[1] == '-' && opt[2] == '\0') {
402  handleoptions = 0;
403  continue;
404  }
405  opt++;
406 
407  if ((ret = parse_option(optctx, opt, argv[optindex], options)) < 0)
408  exit_program(1);
409  optindex += ret;
410  } else {
411  if (parse_arg_function)
412  parse_arg_function(optctx, opt);
413  }
414  }
415 }

417 //+:parse a option group: global options, inputfile options, outputfile options
418 //+:called by ffmpeg_parse_options and openfiles
419 int parse_optgroup(void *optctx, OptionGroup *g)
420 {
421  int i, ret;
422 
423  av_log(NULL, AV_LOG_DEBUG, "Parsing a group of options: %s %s.\n",
424  g->group_def->name, g->arg);
425 
426  for (i = 0; i < g->nb_opts; i++) {
427  Option *o = &g->opts[i];
428 
429  if (g->group_def->flags &&
430  !(g->group_def->flags & o->opt->flags)) {
431  av_log(NULL, AV_LOG_ERROR, "Option %s (%s) cannot be applied to "
432  "%s %s -- you are trying to apply an input option to an "
433  "output file or vice versa. Move this option before the "
434  "file it belongs to.\n", o->key, o->opt->help,
435  g->group_def->name, g->arg);
436  return AVERROR(EINVAL);
437  }
438 
439  av_log(NULL, AV_LOG_DEBUG, "Applying option %s (%s) with argument %s.\n",
440  o->key, o->opt->help, o->val);
441 
442  ret = write_option(optctx, o->opt, o->key, o->val);
443  if (ret < 0)
444  return ret;
445  }
446 
447  av_log(NULL, AV_LOG_DEBUG, "Successfully parsed a group of options.\n");
448 
449  return 0;
450 }
451 //+:mainly used to locate options like: -loglevel(-v)?-report ?-hide_banner, -version
452 int locate_option(int argc, char **argv, const OptionDef *options,
453  const char *optname)
454 {
455  const OptionDef *po;
456  int i;
457 
458  for (i = 1; i < argc; i++) {//+:traverse all the args
459  const char *cur_opt = argv[i];
460 
461  if (*cur_opt++ != '-')
462  continue;
463 
464  po = find_option(options, cur_opt);
465  if (!po->name && cur_opt[0] == 'n' && cur_opt[1] == 'o')//+:no-opt is supported for bool options
466  po = find_option(options, cur_opt + 2);
467 
468  if ((!po->name && !strcmp(cur_opt, optname)) ||
469  (po->name && !strcmp(optname, po->name)))
470  return i;
471 
472  if (!po->name || po->flags & HAS_ARG)
473  i++;
474  }
475  return 0;
476 }

478 static void dump_argument(const char *a)
479 {
480  const unsigned char *p;
481 
482  for (p = a; *p; p++)
483  if (!((*p >= '+' && *p <= ':') || (*p >= '@' && *p <= 'Z') ||
484  *p == '_' || (*p >= 'a' && *p <= 'z')))
485  break;
486  if (!*p) {
487  fputs(a, report_file);
488  return;
489  }
490  fputc('"', report_file);
491  for (p = a; *p; p++) {
492  if (*p == '\\' || *p == '"' || *p == '$' || *p == '`')
493  fprintf(report_file, "\\%c", *p);
494  else if (*p < ' ' || *p > '~')
495  fprintf(report_file, "\\x%02x", *p);
496  else
497  fputc(*p, report_file);
498  }
499  fputc('"', report_file);
500 }

550 int opt_default(void *optctx, const char *opt, const char *arg)
551 {
552  const AVOption *o;
553  int consumed = 0;
554  char opt_stripped[128];
555  const char *p;
556  const AVClass *cc = avcodec_get_class(), *fc = avformat_get_class();
557 #if CONFIG_AVRESAMPLE
558  const AVClass *rc = avresample_get_class();
559 #endif
560 #if CONFIG_SWSCALE
561  const AVClass *sc = sws_get_class();
562 #endif
563 #if CONFIG_SWRESAMPLE
564  const AVClass *swr_class = swr_get_class();
565 #endif
566 
567  if (!strcmp(opt, "debug") || !strcmp(opt, "fdebug"))
568  av_log_set_level(AV_LOG_DEBUG);
569 
570  if (!(p = strchr(opt, ':')))
571  p = opt + strlen(opt);
572  av_strlcpy(opt_stripped, opt, FFMIN(sizeof(opt_stripped), p - opt + 1));
573 
574  if ((o = opt_find(&cc, opt_stripped, NULL, 0,
575  AV_OPT_SEARCH_CHILDREN | AV_OPT_SEARCH_FAKE_OBJ)) ||
576  ((opt[0] == 'v' || opt[0] == 'a' || opt[0] == 's') &&
577  (o = opt_find(&cc, opt + 1, NULL, 0, AV_OPT_SEARCH_FAKE_OBJ)))) {
578  av_dict_set(&codec_opts, opt, arg, FLAGS);
579  consumed = 1;
580  }
581  if ((o = opt_find(&fc, opt, NULL, 0,
582  AV_OPT_SEARCH_CHILDREN | AV_OPT_SEARCH_FAKE_OBJ))) {
583  av_dict_set(&format_opts, opt, arg, FLAGS);
584  if (consumed)
585  av_log(NULL, AV_LOG_VERBOSE, "Routing option %s to both codec and muxer layer\n", opt);
586  consumed = 1;
587  }
588 #if CONFIG_SWSCALE

590  AV_OPT_SEARCH_CHILDREN | AV_OPT_SEARCH_FAKE_OBJ))) {
591  struct SwsContext *sws = sws_alloc_context();

598  return AVERROR(EINVAL);
599  }

601  av_log(NULL, AV_LOG_ERROR, "Error setting option %s.\n", opt);
602  return ret;
603  }
604 
605  av_dict_set(&sws_dict, opt, arg, FLAGS);
606 
607  consumed = 1;
608  }
609 #else
610  if (!consumed && !strcmp(opt, "sws_flags")) {
611  av_log(NULL, AV_LOG_WARNING, "Ignoring %s %s, due to disabled swscale\n", opt, arg);
612  consumed = 1;
613  }
614 #endif
615 #if CONFIG_SWRESAMPLE
616  if (!consumed && (o=opt_find(&swr_class, opt, NULL, 0,
617  AV_OPT_SEARCH_CHILDREN | AV_OPT_SEARCH_FAKE_OBJ))) {
618  struct SwrContext *swr = swr_alloc();
619  int ret = av_opt_set(swr, opt, arg, 0);
620  swr_free(&swr);
621  if (ret < 0) {
622  av_log(NULL, AV_LOG_ERROR, "Error setting option %s.\n", opt);
623  return ret;
624  }
625  av_dict_set(&swr_opts, opt, arg, FLAGS);
626  consumed = 1;
627  }
628 #endif

850 int split_commandline(OptionParseContext *octx, int argc, char *argv[],
851  const OptionDef *options,
852  const OptionGroupDef *groups, int nb_groups)
853 {
854  int optindex = 1;
855  int dashdash = -2;
856 
857  /* perform system-dependent conversions for arguments list */
858  prepare_app_arguments(&argc, &argv);
859 
860  init_parse_context(octx, groups, nb_groups);
861  av_log(NULL, AV_LOG_DEBUG, "Splitting the commandline.\n");
862 
863  while (optindex < argc) {
864  const char *opt = argv[optindex++], *arg;
865  const OptionDef *po;
866  int ret;
867 
868  av_log(NULL, AV_LOG_DEBUG, "Reading option '%s' ...", opt);
869 
870  if (opt[0] == '-' && opt[1] == '-' && !opt[2]) {
871  dashdash = optindex;
872  continue;
873  }
874  /* unnamed group separators, e.g. output filename */
875  if (opt[0] != '-' || !opt[1] || dashdash+1 == optindex) {
876  finish_group(octx, 0, opt);
877  av_log(NULL, AV_LOG_DEBUG, " matched as %s.\n", groups[0].name);
878  continue;
879  }
880  opt++;
881 
882 #define GET_ARG(arg) \
883 do { \
884  arg = argv[optindex++]; \
885  if (!arg) { \
886  av_log(NULL, AV_LOG_ERROR, "Missing argument for option '%s'.\n", opt);\
887  return AVERROR(EINVAL); \
888  } \
889 } while (0)
890 
891  /* named group separators, e.g. -i */
892  if ((ret = match_group_separator(groups, nb_groups, opt)) >= 0) {
893  GET_ARG(arg);
894  finish_group(octx, ret, arg);
895  av_log(NULL, AV_LOG_DEBUG, " matched as %s with argument '%s'.\n",
896  groups[ret].name, arg);
897  continue;
898  }
899 
900  /* normal options */
901  po = find_option(options, opt);
902  if (po->name) {
903  if (po->flags & OPT_EXIT) {
904  /* optional argument, e.g. -h */
905  arg = argv[optindex++];
906  } else if (po->flags & HAS_ARG) {
907  GET_ARG(arg);
908  } else {
909  arg = "1";
910  }
911 
912  add_opt(octx, po, opt, arg);
913  av_log(NULL, AV_LOG_DEBUG, " matched as option '%s' (%s) with "
914  "argument '%s'.\n", po->name, po->help, arg);
915  continue;
916  }
917 
918  /* AVOptions */
919  if (argv[optindex]) {
920  ret = opt_default(NULL, opt, argv[optindex]);
921  if (ret >= 0) {
922  av_log(NULL, AV_LOG_DEBUG, " matched as AVOption '%s' with "
923  "argument '%s'.\n", opt, argv[optindex]);
924  optindex++;
925  continue;
926  } else if (ret != AVERROR_OPTION_NOT_FOUND) {
927  av_log(NULL, AV_LOG_ERROR, "Error parsing option '%s' "
928  "with argument '%s'.\n", opt, argv[optindex]);
929  return ret;
930  }
931  }
932 
933  /* boolean -nofoo options */
934  if (opt[0] == 'n' && opt[1] == 'o' &&
935  (po = find_option(options, opt + 2)) &&
936  po->name && po->flags & OPT_BOOL) {
937  add_opt(octx, po, opt, "0");
938  av_log(NULL, AV_LOG_DEBUG, " matched as option '%s' (%s) with "
939  "argument 0.\n", po->name, po->help);
940  continue;
941  }
942 
943  av_log(NULL, AV_LOG_ERROR, "Unrecognized option '%s'.\n", opt);
944  return AVERROR_OPTION_NOT_FOUND;
945  }
946 
947  if (octx->cur_group.nb_opts || codec_opts || format_opts || resample_opts)
948  av_log(NULL, AV_LOG_WARNING, "Trailing options were found on the "
949  "commandline.\n");
950 
951  av_log(NULL, AV_LOG_DEBUG, "Finished splitting the commandline.\n");
952 
953  return 0;
954 }
