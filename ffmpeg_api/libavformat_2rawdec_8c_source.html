35 int ff_raw_read_partial_packet(AVFormatContext *s, AVPacket *pkt)
36 {
37  int ret, size;
38 
39  size = RAW_PACKET_SIZE;
40 
41  if (av_new_packet(pkt, size) < 0)
42  return AVERROR(ENOMEM);
43 
44  pkt->pos= avio_tell(s->pb);
45  pkt->stream_index = 0;
46  ret = ffio_read_partial(s->pb, pkt->data, size);
47  if (ret < 0) {
48  av_packet_unref(pkt);
49  return ret;
50  }
51  av_shrink_packet(pkt, ret);
52  return ret;
53 }

126 static int mjpeg_probe(AVProbeData *p)
127 {
128  int i;
129  int state = -1;
130  int nb_invalid = 0;
131  int nb_frames = 0;
132 
133  for (i=0; i<p->buf_size-2; i++) {
134  int c;
135  if (p->buf[i] != 0xFF)
136  continue;
137  c = p->buf[i+1];
138  switch (c) {
139  case 0xD8:
140  state = 0xD8;
141  break;
142  case 0xC0:
143  case 0xC1:
144  case 0xC2:
145  case 0xC3:
146  case 0xC5:
147  case 0xC6:
148  case 0xC7:
149  case 0xF7:
150  if (state == 0xD8) {
151  state = 0xC0;
152  } else
153  nb_invalid++;
154  break;
155  case 0xDA:
156  if (state == 0xC0) {
157  state = 0xDA;
158  } else
159  nb_invalid++;
160  break;
161  case 0xD9:
162  if (state == 0xDA) {
163  state = 0xD9;
164  nb_frames++;
165  } else
166  nb_invalid++;
167  break;
168  default:
169  if ( (c >= 0x02 && c <= 0xBF)
170  || c == 0xC8) {
171  nb_invalid++;
172  }
173  }
174  }
175 
176  if (nb_invalid*4 + 1 < nb_frames) {
177  static const char ct_jpeg[] = "\r\nContent-Type: image/jpeg\r\n";
178  int i;
179 
180  for (i=0; i<FFMIN(p->buf_size - (int)sizeof(ct_jpeg), 100); i++)
181  if (!memcmp(p->buf + i, ct_jpeg, sizeof(ct_jpeg) - 1))
182  return AVPROBE_SCORE_EXTENSION;
183 
184  if (nb_invalid == 0 && nb_frames > 2)
185  return AVPROBE_SCORE_EXTENSION / 2;
186  return AVPROBE_SCORE_EXTENSION / 4;
187  }
188 
189  return 0;
190 }
