109  *
110  * @param v input value. If v is 0, the result is undefined.
111  * @return the number of trailing 0-bits

113 /* We use the De-Bruijn method outlined in:
114  * http://supertech.csail.mit.edu/papers/debruijn.pdf. */
115 static av_always_inline av_const int ff_ctz_c(int v)
116 {
117  static const uint8_t debruijn_ctz32[32] = {

120  };
121  return debruijn_ctz32[(uint32_t)((v & -v) * 0x077CB531U) >> 27];
