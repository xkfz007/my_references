25 /**
26  * @file
27  * This is a demuxer for Sony OpenMG Music files
28  *
29  * Known file extensions: ".oma", "aa3"
30  * The format of such files consists of three parts:
31  * - "ea3" header carrying overall info and metadata. Except for starting with
32  * "ea" instead of "ID", it's an ID3v2 header.
33  * - "EA3" header is a Sony-specific header containing information about
34  * the OpenMG file: codec type (usually ATRAC, can also be MP3 or WMA),
35  * codec specific info (packet size, sample rate, channels and so on)
36  * and DRM related info (file encryption, content id).
37  * - Sound data organized in packets follow the EA3 header
38  * (can be encrypted using the Sony DRM!).
39  *
40  * Supported decoders: ATRAC3, ATRAC3+, MP3, LPCM
41  */

92 static int kset(AVFormatContext *s, const uint8_t *r_val, const uint8_t *n_val,
93  int len)
94 {
95  OMAContext *oc = s->priv_data;
96 
97  if (!r_val && !n_val)
98  return -1;
99 
100  len = FFMIN(len, 16);
101 
102  /* use first 64 bits in the third round again */
103  if (r_val) {
104  if (r_val != oc->r_val) {
105  memset(oc->r_val, 0, 24);
106  memcpy(oc->r_val, r_val, len);
107  }
108  memcpy(&oc->r_val[16], r_val, 8);
109  }
110  if (n_val) {
111  if (n_val != oc->n_val) {
112  memset(oc->n_val, 0, 24);
113  memcpy(oc->n_val, n_val, len);
114  }
115  memcpy(&oc->n_val[16], n_val, 8);
116  }
117 
118  return 0;
119 }

123 static int rprobe(AVFormatContext *s, uint8_t *enc_header, unsigned size,
124  const uint8_t *r_val)
125 {
126  OMAContext *oc = s->priv_data;
127  unsigned int pos;
128  struct AVDES *av_des;
129 
130  if (!enc_header || !r_val ||
131  size < OMA_ENC_HEADER_SIZE + oc->k_size + oc->e_size + oc->i_size ||
132  size < OMA_RPROBE_M_VAL)
133  return -1;
134 
135  av_des = av_des_alloc();
136  if (!av_des)
137  return AVERROR(ENOMEM);
138 
139  /* m_val */
140  av_des_init(av_des, r_val, 192, 1);
141  av_des_crypt(av_des, oc->m_val, &enc_header[48], 1, NULL, 1);
142 
143  /* s_val */
144  av_des_init(av_des, oc->m_val, 64, 0);
145  av_des_crypt(av_des, oc->s_val, NULL, 1, NULL, 0);
146 
147  /* sm_val */
148  pos = OMA_ENC_HEADER_SIZE + oc->k_size + oc->e_size;
149  av_des_init(av_des, oc->s_val, 64, 0);
150  av_des_mac(av_des, oc->sm_val, &enc_header[pos], (oc->i_size >> 3));
151 
152  pos += oc->i_size;
153 
154  av_free(av_des);
155 
156  return memcmp(&enc_header[pos], oc->sm_val, 8) ? -1 : 0;
157 }

159 static int nprobe(AVFormatContext *s, uint8_t *enc_header, unsigned size,
160  const uint8_t *n_val)
161 {
162  OMAContext *oc = s->priv_data;
163  uint64_t pos;
164  uint32_t taglen, datalen;
165  struct AVDES *av_des;
166 
167  if (!enc_header || !n_val ||
168  size < OMA_ENC_HEADER_SIZE + oc->k_size + 4)
169  return -1;
170 
171  pos = OMA_ENC_HEADER_SIZE + oc->k_size;
172  if (!memcmp(&enc_header[pos], "EKB ", 4))
173  pos += 32;
174 
175  if (size < pos + 44)
176  return -1;
177 
178  if (AV_RB32(&enc_header[pos]) != oc->rid)
179  av_log(s, AV_LOG_DEBUG, "Mismatching RID\n");
180 
181  taglen = AV_RB32(&enc_header[pos + 32]);
182  datalen = AV_RB32(&enc_header[pos + 36]) >> 4;
183 
184  pos += 44LL + taglen;
185 
186  if (pos + (((uint64_t)datalen) << 4) > size)
187  return -1;
188 
189  av_des = av_des_alloc();
190  if (!av_des)
191  return AVERROR(ENOMEM);
192 
193  av_des_init(av_des, n_val, 192, 1);
194  while (datalen-- > 0) {
195  av_des_crypt(av_des, oc->r_val, &enc_header[pos], 2, NULL, 1);
196  kset(s, oc->r_val, NULL, 16);
197  if (!rprobe(s, enc_header, size, oc->r_val)) {
198  av_free(av_des);
199  return 0;
200  }
201  pos += 16;
202  }
203 
204  av_free(av_des);
205  return -1;
206 }

208 static int decrypt_init(AVFormatContext *s, ID3v2ExtraMeta *em, uint8_t *header)
209 {
210  OMAContext *oc = s->priv_data;
211  ID3v2ExtraMetaGEOB *geob = NULL;
212  uint8_t *gdata;
213 
214  oc->encrypted = 1;
215  av_log(s, AV_LOG_INFO, "File is encrypted\n");
216 
217  /* find GEOB metadata */
218  while (em) {
219  if (!strcmp(em->tag, "GEOB") &&
220  (geob = em->data) &&
221  (!strcmp(geob->description, "OMG_LSI") ||
222  !strcmp(geob->description, "OMG_BKLSI"))) {
223  break;
224  }
225  em = em->next;
226  }
227  if (!em) {
228  av_log(s, AV_LOG_ERROR, "No encryption header found\n");
229  return AVERROR_INVALIDDATA;
230  }
231 
232  if (geob->datasize < 64) {
233  av_log(s, AV_LOG_ERROR,
234  "Invalid GEOB data size: %"PRIu32"\n", geob->datasize);
235  return AVERROR_INVALIDDATA;
236  }
237 
238  gdata = geob->data;
239 
240  if (AV_RB16(gdata) != 1)
241  av_log(s, AV_LOG_WARNING, "Unknown version in encryption header\n");
242 
243  oc->k_size = AV_RB16(&gdata[2]);
244  oc->e_size = AV_RB16(&gdata[4]);
245  oc->i_size = AV_RB16(&gdata[6]);
246  oc->s_size = AV_RB16(&gdata[8]);
247 
248  if (memcmp(&gdata[OMA_ENC_HEADER_SIZE], "KEYRING ", 12)) {
249  av_log(s, AV_LOG_ERROR, "Invalid encryption header\n");
250  return AVERROR_INVALIDDATA;
251  }
252  if (OMA_ENC_HEADER_SIZE + oc->k_size + oc->e_size + oc->i_size + 8 > geob->datasize ||
253  OMA_ENC_HEADER_SIZE + 48 > geob->datasize) {
254  av_log(s, AV_LOG_ERROR, "Too little GEOB data\n");
255  return AVERROR_INVALIDDATA;
256  }
257  oc->rid = AV_RB32(&gdata[OMA_ENC_HEADER_SIZE + 28]);
258  av_log(s, AV_LOG_DEBUG, "RID: %.8"PRIx32"\n", oc->rid);
259 
260  memcpy(oc->iv, &header[0x58], 8);
261  hex_log(s, AV_LOG_DEBUG, "IV", oc->iv, 8);
262 
263  hex_log(s, AV_LOG_DEBUG, "CBC-MAC",
264  &gdata[OMA_ENC_HEADER_SIZE + oc->k_size + oc->e_size + oc->i_size],
265  8);
266 
267  if (s->keylen > 0) {
268  kset(s, s->key, s->key, s->keylen);
269  }
270  if (!memcmp(oc->r_val, (const uint8_t[8]){0}, 8) ||
271  rprobe(s, gdata, geob->datasize, oc->r_val) < 0 &&
272  nprobe(s, gdata, geob->datasize, oc->n_val) < 0) {
273  int i;
274  for (i = 0; i < FF_ARRAY_ELEMS(leaf_table); i += 2) {
275  uint8_t buf[16];
276  AV_WL64(buf, leaf_table[i]);
277  AV_WL64(&buf[8], leaf_table[i + 1]);
278  kset(s, buf, buf, 16);
279  if (!rprobe(s, gdata, geob->datasize, oc->r_val) ||
280  !nprobe(s, gdata, geob->datasize, oc->n_val))
281  break;
282  }
283  if (i >= FF_ARRAY_ELEMS(leaf_table)) {
284  av_log(s, AV_LOG_ERROR, "Invalid key\n");
285  return AVERROR_INVALIDDATA;
286  }
287  }
288 
289  oc->av_des = av_des_alloc();
290  if (!oc->av_des)
291  return AVERROR(ENOMEM);
292 
293  /* e_val */
294  av_des_init(oc->av_des, oc->m_val, 64, 0);
295  av_des_crypt(oc->av_des, oc->e_val,
296  &gdata[OMA_ENC_HEADER_SIZE + 40], 1, NULL, 0);
297  hex_log(s, AV_LOG_DEBUG, "EK", oc->e_val, 8);
298 
299  /* init e_val */
300  av_des_init(oc->av_des, oc->e_val, 64, 1);
301 
302  return 0;
303 }

305 static int oma_read_header(AVFormatContext *s)
306 {
307  int ret, framesize, jsflag, samplerate;
308  uint32_t codec_params, channel_id;
309  int16_t eid;
310  uint8_t buf[EA3_HEADER_SIZE];
311  uint8_t *edata;
312  AVStream *st;
313  ID3v2ExtraMeta *extra_meta = NULL;
314  OMAContext *oc = s->priv_data;
315 
316  ff_id3v2_read(s, ID3v2_EA3_MAGIC, &extra_meta, 0);
317  ret = avio_read(s->pb, buf, EA3_HEADER_SIZE);
318  if (ret < EA3_HEADER_SIZE)
319  return -1;
320 
321  if (memcmp(buf, ((const uint8_t[]){'E', 'A', '3'}), 3) ||
322  buf[4] != 0 || buf[5] != EA3_HEADER_SIZE) {
323  av_log(s, AV_LOG_ERROR, "Couldn't find the EA3 header !\n");
324  return AVERROR_INVALIDDATA;
325  }
326 
327  oc->content_start = avio_tell(s->pb);
328 
329  /* encrypted file */
330  eid = AV_RB16(&buf[6]);
331  if (eid != -1 && eid != -128 && decrypt_init(s, extra_meta, buf) < 0) {
332  ff_id3v2_free_extra_meta(&extra_meta);
333  return -1;
334  }
335 
336  ff_id3v2_free_extra_meta(&extra_meta);
337 
338  codec_params = AV_RB24(&buf[33]);
339 
340  st = avformat_new_stream(s, NULL);
341  if (!st)
342  return AVERROR(ENOMEM);
343 
344  st->start_time = 0;
345  st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
346  st->codecpar->codec_tag = buf[32];
347  st->codecpar->codec_id = ff_codec_get_id(ff_oma_codec_tags,
348  st->codecpar->codec_tag);
349 
350  switch (buf[32]) {
351  case OMA_CODECID_ATRAC3:
352  samplerate = ff_oma_srate_tab[(codec_params >> 13) & 7] * 100;
353  if (!samplerate) {
354  av_log(s, AV_LOG_ERROR, "Unsupported sample rate\n");
355  return AVERROR_INVALIDDATA;
356  }
357  if (samplerate != 44100)
358  avpriv_request_sample(s, "Sample rate %d", samplerate);
359 
360  framesize = (codec_params & 0x3FF) * 8;
361 
362  /* get stereo coding mode, 1 for joint-stereo */
363  jsflag = (codec_params >> 17) & 1;
364 
365  st->codecpar->channels = 2;
366  st->codecpar->channel_layout = AV_CH_LAYOUT_STEREO;
367  st->codecpar->sample_rate = samplerate;
368  st->codecpar->bit_rate = st->codecpar->sample_rate * framesize * 8 / 1024;
369 
370  /* fake the ATRAC3 extradata
371  * (wav format, makes stream copy to wav work) */
372  if (ff_alloc_extradata(st->codecpar, 14))
373  return AVERROR(ENOMEM);
374 
375  edata = st->codecpar->extradata;
376  AV_WL16(&edata[0], 1); // always 1
377  AV_WL32(&edata[2], samplerate); // samples rate
378  AV_WL16(&edata[6], jsflag); // coding mode
379  AV_WL16(&edata[8], jsflag); // coding mode
380  AV_WL16(&edata[10], 1); // always 1
381  // AV_WL16(&edata[12], 0); // always 0
382 
383  avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
384  break;
385  case OMA_CODECID_ATRAC3P:
386  channel_id = (codec_params >> 10) & 7;
387  if (!channel_id) {
388  av_log(s, AV_LOG_ERROR,
389  "Invalid ATRAC-X channel id: %"PRIu32"\n", channel_id);
390  return AVERROR_INVALIDDATA;
391  }
392  st->codecpar->channel_layout = ff_oma_chid_to_native_layout[channel_id - 1];
393  st->codecpar->channels = ff_oma_chid_to_num_channels[channel_id - 1];
394  framesize = ((codec_params & 0x3FF) * 8) + 8;
395  samplerate = ff_oma_srate_tab[(codec_params >> 13) & 7] * 100;
396  if (!samplerate) {
397  av_log(s, AV_LOG_ERROR, "Unsupported sample rate\n");
398  return AVERROR_INVALIDDATA;
399  }
400  st->codecpar->sample_rate = samplerate;
401  st->codecpar->bit_rate = samplerate * framesize * 8 / 2048;
402  avpriv_set_pts_info(st, 64, 1, samplerate);
403  break;
404  case OMA_CODECID_MP3:
405  st->need_parsing = AVSTREAM_PARSE_FULL_RAW;
406  framesize = 1024;
407  break;
408  case OMA_CODECID_LPCM:
409  /* PCM 44.1 kHz 16 bit stereo big-endian */
410  st->codecpar->channels = 2;
411  st->codecpar->channel_layout = AV_CH_LAYOUT_STEREO;
412  st->codecpar->sample_rate = 44100;
413  framesize = 1024;
414  /* bit rate = sample rate x PCM block align (= 4) x 8 */
415  st->codecpar->bit_rate = st->codecpar->sample_rate * 32;
416  st->codecpar->bits_per_coded_sample =
417  av_get_bits_per_sample(st->codecpar->codec_id);
418  avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
419  break;
420  default:
421  av_log(s, AV_LOG_ERROR, "Unsupported codec %d!\n", buf[32]);
422  return AVERROR(ENOSYS);
423  }
424 
425  st->codecpar->block_align = framesize;
426 
427  return 0;
428 }

431 static int oma_read_packet(AVFormatContext *s, AVPacket *pkt)
432 {
433  OMAContext *oc = s->priv_data;
434  AVStream *st = s->streams[0];
435  int packet_size = st->codecpar->block_align;
436  int byte_rate = st->codecpar->bit_rate >> 3;
437  int64_t pos = avio_tell(s->pb);
438  int ret = av_get_packet(s->pb, pkt, packet_size);
439 
440  if (ret < packet_size)
441  pkt->flags |= AV_PKT_FLAG_CORRUPT;
442 
443  if (ret < 0)
444  return ret;
445  if (!ret)
446  return AVERROR_EOF;
447 
448  pkt->stream_index = 0;
449 
450  if (pos >= oc->content_start && byte_rate > 0) {
451  pkt->pts =
452  pkt->dts = av_rescale(pos - oc->content_start, st->time_base.den,
453  byte_rate * (int64_t)st->time_base.num);
454  }
455 
456  if (oc->encrypted) {
457  /* previous unencrypted block saved in IV for
458  * the next packet (CBC mode) */
459  if (ret == packet_size)
460  av_des_crypt(oc->av_des, pkt->data, pkt->data,
461  (packet_size >> 3), oc->iv, 1);
462  else
463  memset(oc->iv, 0, 8);
464  }
465 
466  return ret;
467 }

469 static int oma_read_probe(AVProbeData *p)
470 {
471  const uint8_t *buf = p->buf;
472  unsigned tag_len = 0;
473 
474  if (p->buf_size >= ID3v2_HEADER_SIZE && ff_id3v2_match(buf, ID3v2_EA3_MAGIC))
475  tag_len = ff_id3v2_tag_len(buf);
476 
477  /* This check cannot overflow as tag_len has at most 28 bits */
478  if (p->buf_size < tag_len + 5)
479  /* EA3 header comes late, might be outside of the probe buffer */
480  return tag_len ? AVPROBE_SCORE_EXTENSION/2 : 0;
481 
482  buf += tag_len;
483 
484  if (!memcmp(buf, "EA3", 3) && !buf[4] && buf[5] == EA3_HEADER_SIZE)
485  return AVPROBE_SCORE_MAX;
486  else
487  return 0;
488 }

490 static int oma_read_seek(struct AVFormatContext *s,
491  int stream_index, int64_t timestamp, int flags)
492 {
493  OMAContext *oc = s->priv_data;
494  int64_t err = ff_pcm_read_seek(s, stream_index, timestamp, flags);
495 
496  if (!oc->encrypted)
497  return err;
498 
499  /* readjust IV for CBC */
500  if (err || avio_tell(s->pb) < oc->content_start)
501  goto wipe;
502  if ((err = avio_seek(s->pb, -8, SEEK_CUR)) < 0)
503  goto wipe;
504  if ((err = avio_read(s->pb, oc->iv, 8)) < 8) {
505  if (err >= 0)
506  err = AVERROR_EOF;
507  goto wipe;
508  }
509 
510  return 0;
511 wipe:
512  memset(oc->iv, 0, 8);
513  return err;
514 }
