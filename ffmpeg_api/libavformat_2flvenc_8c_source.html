127 static int get_audio_flags(AVFormatContext *s, AVCodecParameters *par)
128 {
129  int flags = (par->bits_per_coded_sample == 16) ? FLV_SAMPLESSIZE_16BIT
130  : FLV_SAMPLESSIZE_8BIT;
131 
132  if (par->codec_id == AV_CODEC_ID_AAC) // specs force these parameters
133  return FLV_CODECID_AAC | FLV_SAMPLERATE_44100HZ |
134  FLV_SAMPLESSIZE_16BIT | FLV_STEREO;
135  else if (par->codec_id == AV_CODEC_ID_SPEEX) {
136  if (par->sample_rate != 16000) {
137  av_log(s, AV_LOG_ERROR,
138  "FLV only supports wideband (16kHz) Speex audio\n");
139  return AVERROR(EINVAL);
140  }
141  if (par->channels != 1) {
142  av_log(s, AV_LOG_ERROR, "FLV only supports mono Speex audio\n");
143  return AVERROR(EINVAL);
144  }
145  return FLV_CODECID_SPEEX | FLV_SAMPLERATE_11025HZ | FLV_SAMPLESSIZE_16BIT;
146  } else {
147  switch (par->sample_rate) {
148  case 48000:
149  // 48khz mp3 is stored with 44k1 samplerate identifer
150  if (par->codec_id == AV_CODEC_ID_MP3) {
151  flags |= FLV_SAMPLERATE_44100HZ;
152  break;
153  } else {
154  goto error;
155  }
156  case 44100:
157  flags |= FLV_SAMPLERATE_44100HZ;
158  break;
159  case 22050:
160  flags |= FLV_SAMPLERATE_22050HZ;
161  break;
162  case 11025:
163  flags |= FLV_SAMPLERATE_11025HZ;
164  break;
165  case 16000: // nellymoser only
166  case 8000: // nellymoser only
167  case 5512: // not MP3
168  if (par->codec_id != AV_CODEC_ID_MP3) {
169  flags |= FLV_SAMPLERATE_SPECIAL;
170  break;
171  }
172  default:
173 error:
174  av_log(s, AV_LOG_ERROR,
175  "FLV does not support sample rate %d, "
176  "choose from (44100, 22050, 11025)\n", par->sample_rate);
177  return AVERROR(EINVAL);
178  }
179  }
180 
181  if (par->channels > 1)
182  flags |= FLV_STEREO;
183 
184  switch (par->codec_id) {
185  case AV_CODEC_ID_MP3:
186  flags |= FLV_CODECID_MP3 | FLV_SAMPLESSIZE_16BIT;
187  break;
188  case AV_CODEC_ID_PCM_U8:
189  flags |= FLV_CODECID_PCM | FLV_SAMPLESSIZE_8BIT;
190  break;
191  case AV_CODEC_ID_PCM_S16BE:
192  flags |= FLV_CODECID_PCM | FLV_SAMPLESSIZE_16BIT;
193  break;
194  case AV_CODEC_ID_PCM_S16LE:
195  flags |= FLV_CODECID_PCM_LE | FLV_SAMPLESSIZE_16BIT;
196  break;
197  case AV_CODEC_ID_ADPCM_SWF:
198  flags |= FLV_CODECID_ADPCM | FLV_SAMPLESSIZE_16BIT;
199  break;
200  case AV_CODEC_ID_NELLYMOSER:
201  if (par->sample_rate == 8000)
202  flags |= FLV_CODECID_NELLYMOSER_8KHZ_MONO | FLV_SAMPLESSIZE_16BIT;
203  else if (par->sample_rate == 16000)
204  flags |= FLV_CODECID_NELLYMOSER_16KHZ_MONO | FLV_SAMPLESSIZE_16BIT;
205  else
206  flags |= FLV_CODECID_NELLYMOSER | FLV_SAMPLESSIZE_16BIT;
207  break;
208  case AV_CODEC_ID_PCM_MULAW:
209  flags = FLV_CODECID_PCM_MULAW | FLV_SAMPLERATE_SPECIAL | FLV_SAMPLESSIZE_16BIT;
210  break;
211  case AV_CODEC_ID_PCM_ALAW:
212  flags = FLV_CODECID_PCM_ALAW | FLV_SAMPLERATE_SPECIAL | FLV_SAMPLESSIZE_16BIT;
213  break;
214  case 0:
215  flags |= par->codec_tag << 4;
216  break;
217  default:
218  av_log(s, AV_LOG_ERROR, "Audio codec '%s' not compatible with FLV\n",
219  avcodec_get_name(par->codec_id));
220  return AVERROR(EINVAL);
221  }
222 
223  return flags;
224 }

233 static void put_avc_eos_tag(AVIOContext *pb, unsigned ts)
234 {
235  avio_w8(pb, FLV_TAG_TYPE_VIDEO);
236  avio_wb24(pb, 5); /* Tag Data Size */
237  avio_wb24(pb, ts); /* lower 24 bits of timestamp in ms */
238  avio_w8(pb, (ts >> 24) & 0x7F); /* MSB of ts in ms */
239  avio_wb24(pb, 0); /* StreamId = 0 */
240  avio_w8(pb, 23); /* ub[4] FrameType = 1, ub[4] CodecId = 7 */
241  avio_w8(pb, 2); /* AVC end of sequence */
242  avio_wb24(pb, 0); /* Always 0 for AVC EOS. */
243  avio_wb32(pb, 16); /* Size of FLV tag */
244 }

269 static void write_metadata(AVFormatContext *s, unsigned int ts)
270 {
271  AVIOContext *pb = s->pb;
272  FLVContext *flv = s->priv_data;
273  int write_duration_filesize = !(flv->flags & FLV_NO_DURATION_FILESIZE);
274  int metadata_count = 0;
275  int64_t metadata_count_pos;
276  AVDictionaryEntry *tag = NULL;
277 
278  /* write meta_tag */
279  avio_w8(pb, FLV_TAG_TYPE_META); // tag type META
280  flv->metadata_size_pos = avio_tell(pb);//+felix:used to calculate size of metadata to write into stream
281  avio_wb24(pb, 0); // size of data part (sum of all parts below)//+felix:0 is for current, will be changed later
282  avio_wb24(pb, ts); // timestamp
283  avio_wb32(pb, 0); // reserved
284 
285  /* now data of data_size size */
286 
287  /* first event name as a string */
288  avio_w8(pb, AMF_DATA_TYPE_STRING);
289  put_amf_string(pb, "onMetaData"); // 12 bytes
290 
291  /* mixed array (hash) with size and string/type/data tuples */
292  avio_w8(pb, AMF_DATA_TYPE_MIXEDARRAY);
293  metadata_count_pos = avio_tell(pb);
294  metadata_count = 4 * !!flv->video_par +
295  5 * !!flv->audio_par +
296  1 * !!flv->data_par;
297  if (write_duration_filesize) {
298  metadata_count += 2; // +2 for duration and file size
299  }
300  avio_wb32(pb, metadata_count);
301 
302  if (write_duration_filesize) {
303  put_amf_string(pb, "duration");
304  flv->duration_offset = avio_tell(pb);
305  // fill in the guessed duration, it'll be corrected later if incorrect
306  put_amf_double(pb, s->duration / AV_TIME_BASE);
307  }
308 
309  if (flv->video_par) {
310  put_amf_string(pb, "width");
311  put_amf_double(pb, flv->video_par->width);
312 
313  put_amf_string(pb, "height");
314  put_amf_double(pb, flv->video_par->height);
315 
316  put_amf_string(pb, "videodatarate");
317  put_amf_double(pb, flv->video_par->bit_rate / 1024.0);
318 
319  if (flv->framerate != 0.0) {
320  put_amf_string(pb, "framerate");
321  put_amf_double(pb, flv->framerate);
322  metadata_count++;
323  }
324 
325  put_amf_string(pb, "videocodecid");
326  put_amf_double(pb, flv->video_par->codec_tag);
327  }
328 
329  if (flv->audio_par) {
330  put_amf_string(pb, "audiodatarate");
331  put_amf_double(pb, flv->audio_par->bit_rate / 1024.0);
332 
333  put_amf_string(pb, "audiosamplerate");
334  put_amf_double(pb, flv->audio_par->sample_rate);
335 
336  put_amf_string(pb, "audiosamplesize");
337  put_amf_double(pb, flv->audio_par->codec_id == AV_CODEC_ID_PCM_U8 ? 8 : 16);
338 
339  put_amf_string(pb, "stereo");
340  put_amf_bool(pb, flv->audio_par->channels == 2);
341 
342  put_amf_string(pb, "audiocodecid");
343  put_amf_double(pb, flv->audio_par->codec_tag);
344  }
345 
346  if (flv->data_par) {
347  put_amf_string(pb, "datastream");
348  put_amf_double(pb, 0.0);
349  }
350 
351  ff_standardize_creation_time(s);
352  //+felix: add the tags from the input format
353  while ((tag = av_dict_get(s->metadata, "", tag, AV_DICT_IGNORE_SUFFIX))) {
354  if( !strcmp(tag->key, "width")
355  ||!strcmp(tag->key, "height")
356  ||!strcmp(tag->key, "videodatarate")
357  ||!strcmp(tag->key, "framerate")
358  ||!strcmp(tag->key, "videocodecid")
359  ||!strcmp(tag->key, "audiodatarate")
360  ||!strcmp(tag->key, "audiosamplerate")
361  ||!strcmp(tag->key, "audiosamplesize")
362  ||!strcmp(tag->key, "stereo")
363  ||!strcmp(tag->key, "audiocodecid")
364  ||!strcmp(tag->key, "duration")
365  ||!strcmp(tag->key, "onMetaData")
366  ||!strcmp(tag->key, "datasize")
367  ||!strcmp(tag->key, "lasttimestamp")
368  ||!strcmp(tag->key, "totalframes")
369  ||!strcmp(tag->key, "hasAudio")
370  ||!strcmp(tag->key, "hasVideo")
371  ||!strcmp(tag->key, "hasCuePoints")
372  ||!strcmp(tag->key, "hasMetadata")
373  ||!strcmp(tag->key, "hasKeyframes")
374  ){
375  av_log(s, AV_LOG_DEBUG, "Ignoring metadata for %s\n", tag->key);
376  continue;
377  }
378  put_amf_string(pb, tag->key);
379  avio_w8(pb, AMF_DATA_TYPE_STRING);
380  put_amf_string(pb, tag->value);
381  metadata_count++;
382  }
383 
384  if (write_duration_filesize) {
385  put_amf_string(pb, "filesize");
386  flv->filesize_offset = avio_tell(pb);
387  put_amf_double(pb, 0); // delayed write
388  }
389 
390  if (flv->flags & FLV_ADD_KEYFRAME_INDEX) {
391  flv->acurframeindex = 0;
392  flv->keyframe_index_size = 0;
393 
394  put_amf_string(pb, "hasVideo");
395  put_amf_bool(pb, !!flv->video_par);
396  metadata_count++;
397 
398  put_amf_string(pb, "hasKeyframes");
399  put_amf_bool(pb, 1);
400  metadata_count++;
401 
402  put_amf_string(pb, "hasAudio");
403  put_amf_bool(pb, !!flv->audio_par);
404  metadata_count++;
405 
406  put_amf_string(pb, "hasMetadata");
407  put_amf_bool(pb, 1);
408  metadata_count++;
409 
410  put_amf_string(pb, "canSeekToEnd");
411  put_amf_bool(pb, 1);
412  metadata_count++;
413 
414  put_amf_string(pb, "datasize");
415  flv->datasize_offset = avio_tell(pb);
416  flv->datasize = 0;
417  put_amf_double(pb, flv->datasize);
418  metadata_count++;
419 
420  put_amf_string(pb, "videosize");
421  flv->videosize_offset = avio_tell(pb);
422  flv->videosize = 0;
423  put_amf_double(pb, flv->videosize);
424  metadata_count++;
425 
426  put_amf_string(pb, "audiosize");
427  flv->audiosize_offset = avio_tell(pb);
428  flv->audiosize = 0;
429  put_amf_double(pb, flv->audiosize);
430  metadata_count++;
431 
432  put_amf_string(pb, "lasttimestamp");
433  flv->lasttimestamp_offset = avio_tell(pb);
434  flv->lasttimestamp = 0;
435  put_amf_double(pb, 0);
436  metadata_count++;
437 
438  put_amf_string(pb, "lastkeyframetimestamp");
439  flv->lastkeyframetimestamp_offset = avio_tell(pb);
440  flv->lastkeyframetimestamp = 0;
441  put_amf_double(pb, 0);
442  metadata_count++;
443 
444  put_amf_string(pb, "lastkeyframelocation");
445  flv->lastkeyframelocation_offset = avio_tell(pb);
446  flv->lastkeyframelocation = 0;
447  put_amf_double(pb, 0);
448  metadata_count++;
449 
450  put_amf_string(pb, "keyframes");
451  put_amf_byte(pb, AMF_DATA_TYPE_OBJECT);
452  metadata_count++;
453 
454  flv->keyframes_info_offset = avio_tell(pb);
455  }
456 
457  put_amf_string(pb, "");
458  avio_w8(pb, AMF_END_OF_OBJECT);
459 
460  /* write total size of tag */
461  flv->metadata_totalsize = avio_tell(pb) - flv->metadata_size_pos - 10;
462 
463  avio_seek(pb, metadata_count_pos, SEEK_SET);
464  avio_wb32(pb, metadata_count);
465 
466  avio_seek(pb, flv->metadata_size_pos, SEEK_SET);
467  avio_wb24(pb, flv->metadata_totalsize);
468  avio_skip(pb, flv->metadata_totalsize + 10 - 3);
469  flv->metadata_totalsize_pos = avio_tell(pb);
470  avio_wb32(pb, flv->metadata_totalsize + 11);
471 }

484 static void flv_write_codec_header(AVFormatContext* s, AVCodecParameters* par) {
485  int64_t data_size;
486  AVIOContext *pb = s->pb;
487  FLVContext *flv = s->priv_data;
488 
489  if (par->codec_id == AV_CODEC_ID_AAC || par->codec_id == AV_CODEC_ID_H264
490  || par->codec_id == AV_CODEC_ID_MPEG4) {
491  int64_t pos;
492  avio_w8(pb,
493  par->codec_type == AVMEDIA_TYPE_VIDEO ?
494  FLV_TAG_TYPE_VIDEO : FLV_TAG_TYPE_AUDIO);
495  avio_wb24(pb, 0); // size patched later
496  avio_wb24(pb, 0); // ts
497  avio_w8(pb, 0); // ts ext
498  avio_wb24(pb, 0); // streamid
499  pos = avio_tell(pb);
500  if (par->codec_id == AV_CODEC_ID_AAC) {
501  avio_w8(pb, get_audio_flags(s, par));
502  avio_w8(pb, 0); // AAC sequence header
503 
504  if (!par->extradata_size && (flv->flags & FLV_AAC_SEQ_HEADER_DETECT)) {
505  PutBitContext pbc;
506  int samplerate_index;
507  int channels = flv->audio_par->channels
508  - (flv->audio_par->channels == 8 ? 1 : 0);
509  uint8_t data[2];
510 
511  for (samplerate_index = 0; samplerate_index < 16;
512  samplerate_index++)
513  if (flv->audio_par->sample_rate
514  == mpeg4audio_sample_rates[samplerate_index])
515  break;
516 
517  init_put_bits(&pbc, data, sizeof(data));
518  put_bits(&pbc, 5, flv->audio_par->profile + 1); //profile
519  put_bits(&pbc, 4, samplerate_index); //sample rate index
520  put_bits(&pbc, 4, channels);
521  put_bits(&pbc, 1, 0); //frame length - 1024 samples
522  put_bits(&pbc, 1, 0); //does not depend on core coder
523  put_bits(&pbc, 1, 0); //is not extension
524  flush_put_bits(&pbc);
525 
526  avio_w8(pb, data[0]);
527  avio_w8(pb, data[1]);
528 
529  av_log(s, AV_LOG_WARNING, "AAC sequence header: %02x %02x.\n",
530  data[0], data[1]);
531  }
532  avio_write(pb, par->extradata, par->extradata_size);
533  } else {
534  avio_w8(pb, par->codec_tag | FLV_FRAME_KEY); // flags
535  avio_w8(pb, 0); // AVC sequence header
536  avio_wb24(pb, 0); // composition time
537  ff_isom_write_avcc(pb, par->extradata, par->extradata_size);
538  }
539  data_size = avio_tell(pb) - pos;
540  avio_seek(pb, -data_size - 10, SEEK_CUR);
541  avio_wb24(pb, data_size);
542  avio_skip(pb, data_size + 10 - 3);
543  avio_wb32(pb, data_size + 11); // previous tag size
544  }
545 }

547 static int flv_append_keyframe_info(AVFormatContext *s, FLVContext *flv, double ts, int64_t pos)
548 {
549  FLVFileposition *position = av_malloc(sizeof(FLVFileposition));
550 
551  if (!position) {
552  av_log(s, AV_LOG_WARNING, "no mem for add keyframe index!\n");
553  return AVERROR(ENOMEM);
554  }

574 static int shift_data(AVFormatContext *s)
575 {
576  int ret = 0;
577  int n = 0;
578  int64_t metadata_size = 0;
579  FLVContext *flv = s->priv_data;
580  int64_t pos, pos_end = avio_tell(s->pb);
581  uint8_t *buf, *read_buf[2];
582  int read_buf_id = 0;
583  int read_size[2];
584  AVIOContext *read_pb;
585 
586  metadata_size = flv->filepositions_count * 9 * 2 + 10; /* filepositions and times value */
587  metadata_size += 2 + 13; /* filepositions String */
588  metadata_size += 2 + 5; /* times String */
589  metadata_size += 3; /* Object end */
590 
591  flv->keyframe_index_size = metadata_size;
592 
593  if (metadata_size < 0)
594  return metadata_size;
595 
596  buf = av_malloc_array(metadata_size, 2);
597  if (!buf) {
598  return AVERROR(ENOMEM);
599  }
600  read_buf[0] = buf;
601  read_buf[1] = buf + metadata_size;
602 
603  avio_seek(s->pb, flv->metadata_size_pos, SEEK_SET);
604  avio_wb24(s->pb, flv->metadata_totalsize + metadata_size);
605 
606  avio_seek(s->pb, flv->metadata_totalsize_pos, SEEK_SET);
607  avio_wb32(s->pb, flv->metadata_totalsize + 11 + metadata_size);
608  avio_seek(s->pb, pos_end, SEEK_SET);
609 
610  /* Shift the data: the AVIO context of the output can only be used for
611  * writing, so we re-open the same output, but for reading. It also avoids
612  * a read/seek/write/seek back and forth. */
613  avio_flush(s->pb);
614  ret = s->io_open(s, &read_pb, s->filename, AVIO_FLAG_READ, NULL);
615  if (ret < 0) {
616  av_log(s, AV_LOG_ERROR, "Unable to re-open %s output file for "
617  "the second pass (add_keyframe_index)\n", s->filename);
618  goto end;
619  }
620 
621  /* mark the end of the shift to up to the last data we wrote, and get ready
622  * for writing */
623  pos_end = avio_tell(s->pb);
624  avio_seek(s->pb, flv->keyframes_info_offset + metadata_size, SEEK_SET);
625 
626  /* start reading at where the keyframe index information will be placed */
627  avio_seek(read_pb, flv->keyframes_info_offset, SEEK_SET);
628  pos = avio_tell(read_pb);
629 
630  /* shift data by chunk of at most keyframe *filepositions* and *times* size */
631  read_size[read_buf_id] = avio_read(read_pb, read_buf[read_buf_id], metadata_size); \
632  read_buf_id ^= 1;
633  do {
634 
635  read_size[read_buf_id] = avio_read(read_pb, read_buf[read_buf_id], metadata_size); \
636  read_buf_id ^= 1;
637  n = read_size[read_buf_id];
638  if (n < 0)
639  break;
640  avio_write(s->pb, read_buf[read_buf_id], n);
641  pos += n;
642  } while (pos <= pos_end);
643 
644  ff_format_io_close(s, &read_pb);
645 
646 end:
647  av_free(buf);
648  return ret;
649 }

652  write header for flv, including:
653  9 bytes flv header

657 static int flv_write_header(AVFormatContext *s)
658 {
659  int i;
660  AVIOContext *pb = s->pb;
661  FLVContext *flv = s->priv_data;
662 
663  for (i = 0; i < s->nb_streams; i++) {
664  AVCodecParameters *par = s->streams[i]->codecpar;
665  FLVStreamContext *sc;
666  switch (par->codec_type) {
667  case AVMEDIA_TYPE_VIDEO:
668  if (s->streams[i]->avg_frame_rate.den &&
669  s->streams[i]->avg_frame_rate.num) {
670  flv->framerate = av_q2d(s->streams[i]->avg_frame_rate);
671  }
672  if (flv->video_par) {
673  av_log(s, AV_LOG_ERROR,
674  "at most one video stream is supported in flv\n");
675  return AVERROR(EINVAL);
676  }
677  flv->video_par = par;
678  if (!ff_codec_get_tag(flv_video_codec_ids, par->codec_id))
679  return unsupported_codec(s, "Video", par->codec_id);
680 
681  if (par->codec_id == AV_CODEC_ID_MPEG4 ||
682  par->codec_id == AV_CODEC_ID_H263) {
683  int error = s->strict_std_compliance > FF_COMPLIANCE_UNOFFICIAL;
684  av_log(s, error ? AV_LOG_ERROR : AV_LOG_WARNING,
685  "Codec %s is not supported in the official FLV specification,\n", avcodec_get_name(par->codec_id));
686 
687  if (error) {
688  av_log(s, AV_LOG_ERROR,
689  "use vstrict=-1 / -strict -1 to use it anyway.\n");
690  return AVERROR(EINVAL);
691  }
692  } else if (par->codec_id == AV_CODEC_ID_VP6) {
693  av_log(s, AV_LOG_WARNING,
694  "Muxing VP6 in flv will produce flipped video on playback.\n");
695  }
696  break;
697  case AVMEDIA_TYPE_AUDIO:
698  if (flv->audio_par) {
699  av_log(s, AV_LOG_ERROR,
700  "at most one audio stream is supported in flv\n");
701  return AVERROR(EINVAL);
702  }
703  flv->audio_par = par;
704  if (get_audio_flags(s, par) < 0)//+Felix: returned value is not used. For AAC, 48000HZ is not checked.
705  return unsupported_codec(s, "Audio", par->codec_id);
706  if (par->codec_id == AV_CODEC_ID_PCM_S16BE)
707  av_log(s, AV_LOG_WARNING,
708  "16-bit big-endian audio in flv is valid but most likely unplayable (hardware dependent); use s16le\n");
709  break;
710  case AVMEDIA_TYPE_DATA:
711  if (par->codec_id != AV_CODEC_ID_TEXT && par->codec_id != AV_CODEC_ID_NONE)
712  return unsupported_codec(s, "Data", par->codec_id);
713  flv->data_par = par;
714  break;
715  case AVMEDIA_TYPE_SUBTITLE:
716  if (par->codec_id != AV_CODEC_ID_TEXT) {
717  av_log(s, AV_LOG_ERROR, "Subtitle codec '%s' for stream %d is not compatible with FLV\n",
718  avcodec_get_name(par->codec_id), i);
719  return AVERROR_INVALIDDATA;
720  }
721  flv->data_par = par;
722  break;
723  default:
724  av_log(s, AV_LOG_ERROR, "Codec type '%s' for stream %d is not compatible with FLV\n",
725  av_get_media_type_string(par->codec_type), i);
726  return AVERROR(EINVAL);
727  }
728  avpriv_set_pts_info(s->streams[i], 32, 1, 1000); /* 32 bit pts in ms */
729 
730  sc = av_mallocz(sizeof(FLVStreamContext));
731  if (!sc)
732  return AVERROR(ENOMEM);
733  s->streams[i]->priv_data = sc;
734  sc->last_ts = -1;
735  }
736 
737  flv->delay = AV_NOPTS_VALUE;
738 
739  avio_write(pb, "FLV", 3);//+Felix: 3 bytes flv tag
740  avio_w8(pb, 1);
741  avio_w8(pb, FLV_HEADER_FLAG_HASAUDIO * !!flv->audio_par +
742  FLV_HEADER_FLAG_HASVIDEO * !!flv->video_par);//+Felix: good trick
743  avio_wb32(pb, 9);//+felix:DataOffset
744  avio_wb32(pb, 0);//+felix:PreviousTagSize0
745 
746  for (i = 0; i < s->nb_streams; i++)
747  if (s->streams[i]->codecpar->codec_tag == 5) {//+felix:Null type
748  avio_w8(pb, 8); // message type
749  avio_wb24(pb, 0); // include flags
750  avio_wb24(pb, 0); // time stamp
751  avio_wb32(pb, 0); // reserved
752  avio_wb32(pb, 11); // size
753  flv->reserved = 5;
754  }
755 
756  if (flv->flags & FLV_NO_METADATA) {
757  pb->seekable = 0;
758  } else {
759  write_metadata(s, 0);//+felix: onMetaData
760  }
761 
762  for (i = 0; i < s->nb_streams; i++) {
763  flv_write_codec_header(s, s->streams[i]->codecpar);
764  }
765 
766  flv->datastart_offset = avio_tell(pb);
767  return 0;
768 }

770 static int flv_write_trailer(AVFormatContext *s)

778

812  put_amf_double(pb, newflv_posinfo->keyframe_timestamp);

817  p = newflv_posinfo->next;
818  if (p) {
819  newflv_posinfo->next = p->next;
820  av_free(p);
821  p = NULL;
822  } else {
823  av_free(newflv_posinfo);
824  newflv_posinfo = NULL;
825  }
826  }
827 
828  put_amf_string(pb, "");

831  avio_seek(pb, cur_pos + flv->keyframe_index_size, SEEK_SET);
832  }
833

836  av_log(s, AV_LOG_DEBUG, "FLV no sequence end mode open\n");

840  AVCodecParameters *par = s->streams[i]->codecpar;
841  FLVStreamContext *sc = s->streams[i]->priv_data;
842  if (par->codec_type == AVMEDIA_TYPE_VIDEO &&

844  put_avc_eos_tag(pb, sc->last_ts);

860  put_amf_double(pb, flv->duration / (double)1000);

862  if (avio_seek(pb, flv->filesize_offset, SEEK_SET) < 0) {
863  av_log(s, AV_LOG_WARNING, "Failed to update header with correct filesize.\n");
864  } else {
865  put_amf_double(pb, file_size);
866  }
867  }
868  avio_seek(pb, file_size, SEEK_SET);
869  return 0;

872 static int flv_write_packet(AVFormatContext *s, AVPacket *pkt)
873 {

875  AVCodecParameters *par = s->streams[pkt->stream_index]->codecpar;

877  FLVStreamContext *sc = s->streams[pkt->stream_index]->priv_data;

885  par->codec_id == AV_CODEC_ID_VP6 || par->codec_id == AV_CODEC_ID_AAC)

888  flags_size = 5;
889  else
890  flags_size = 1;

897  av_free(par->extradata);

906  flv->delay = -pkt->dts;
907 
908  if (pkt->dts < -flv->delay) {
909  av_log(s, AV_LOG_WARNING,
910  "Packets are not in the proper order with respect to DTS\n");
911  return AVERROR(EINVAL);
912  }

914  ts = pkt->dts;
915 
916  if (s->event_flags & AVSTREAM_EVENT_FLAG_METADATA_UPDATED) {
917  write_metadata(s, ts);
918  s->event_flags &= ~AVSTREAM_EVENT_FLAG_METADATA_UPDATED;
919  }
920 
921  avio_write_marker(pb, av_rescale(ts, AV_TIME_BASE, 1000),
922  pkt->flags & AV_PKT_FLAG_KEY && (flv->video_par ? par->codec_type == AVMEDIA_TYPE_VIDEO : 1) ? AVIO_DATA_MARKER_SYNC_POINT : AVIO_DATA_MARKER_BOUNDARY_POINT);
923 
924  switch (par->codec_type) {
925  case AVMEDIA_TYPE_VIDEO:
926  avio_w8(pb, FLV_TAG_TYPE_VIDEO);
927 
928  flags = ff_codec_get_tag(flv_video_codec_ids, par->codec_id);
929 
930  flags |= pkt->flags & AV_PKT_FLAG_KEY ? FLV_FRAME_KEY : FLV_FRAME_INTER;
931  break;
932  case AVMEDIA_TYPE_AUDIO:
933  flags = get_audio_flags(s, par);
934 
935  av_assert0(size);
936 
937  avio_w8(pb, FLV_TAG_TYPE_AUDIO);
938  break;
939  case AVMEDIA_TYPE_SUBTITLE:
940  case AVMEDIA_TYPE_DATA:
941  avio_w8(pb, FLV_TAG_TYPE_META);
942  break;
943  default:
944  return AVERROR(EINVAL);
945  }

948  /* check if extradata looks like mp4 formatted */
949  if (par->extradata_size > 0 && *(uint8_t*)par->extradata != 1)
950  if ((ret = ff_avc_parse_nal_units_buf(pkt->data, &data, &size)) < 0)
951  return ret;
952  } else if (par->codec_id == AV_CODEC_ID_AAC && pkt->size > 2 &&

959  }
960  av_log(s, AV_LOG_WARNING, "aac bitstream error\n");

965  av_log(s, AV_LOG_WARNING, "Warning: Speex stream has more than "
966  "8 frames per packet. Adobe Flash "
967  "Player cannot handle this!\n");

969  if (sc->last_ts < ts)
970  sc->last_ts = ts;
971 
972  if (size + flags_size >= 1<<24) {
973  av_log(s, AV_LOG_ERROR, "Too large packet with size %u >= %u\n",
974  size + flags_size, 1<<24);
975  return AVERROR(EINVAL);
976  }
977 
978  avio_wb24(pb, size + flags_size);
979  avio_wb24(pb, ts & 0xFFFFFF);
980  avio_w8(pb, (ts >> 24) & 0x7F); // timestamps are 32 bits _signed_
981  avio_wb24(pb, flv->reserved);
982 
983  if (par->codec_type == AVMEDIA_TYPE_DATA ||
984  par->codec_type == AVMEDIA_TYPE_SUBTITLE ) {
985  int data_size;
986  int64_t metadata_size_pos = avio_tell(pb);
987  if (par->codec_id == AV_CODEC_ID_TEXT) {
988  // legacy FFmpeg magic?
989  avio_w8(pb, AMF_DATA_TYPE_STRING);
990  put_amf_string(pb, "onTextData");
991  avio_w8(pb, AMF_DATA_TYPE_MIXEDARRAY);
992  avio_wb32(pb, 2);
993  put_amf_string(pb, "type");
994  avio_w8(pb, AMF_DATA_TYPE_STRING);
995  put_amf_string(pb, "Text");
996  put_amf_string(pb, "text");
997  avio_w8(pb, AMF_DATA_TYPE_STRING);
998  put_amf_string(pb, pkt->data);
999  put_amf_string(pb, "");
1000  avio_w8(pb, AMF_END_OF_OBJECT);
1001  } else {
1002  // just pass the metadata through
1003  avio_write(pb, data ? data : pkt->data, size);
1004  }
1005  /* write total size of tag */
1006  data_size = avio_tell(pb) - metadata_size_pos;
1007  avio_seek(pb, metadata_size_pos - 10, SEEK_SET);
1008  avio_wb24(pb, data_size);
1009  avio_seek(pb, data_size + 10 - 3, SEEK_CUR);
1010  avio_wb32(pb, data_size + 11);
1011  } else {
1012  av_assert1(flags>=0);
1013  avio_w8(pb,flags);
1014  if (par->codec_id == AV_CODEC_ID_VP6)
1015  avio_w8(pb,0);
1016  if (par->codec_id == AV_CODEC_ID_VP6F || par->codec_id == AV_CODEC_ID_VP6A) {
1017  if (par->extradata_size)
1018  avio_w8(pb, par->extradata[0]);
1019  else
1020  avio_w8(pb, ((FFALIGN(par->width, 16) - par->width) << 4) |
1021  (FFALIGN(par->height, 16) - par->height));
1022  } else if (par->codec_id == AV_CODEC_ID_AAC)
1023  avio_w8(pb, 1); // AAC raw
1024  else if (par->codec_id == AV_CODEC_ID_H264 || par->codec_id == AV_CODEC_ID_MPEG4) {
1025  avio_w8(pb, 1); // AVC NALU
1026  avio_wb24(pb, pkt->pts - pkt->dts);
1027  }
1028 
1029  avio_write(pb, data ? data : pkt->data, size);
1030 
1031  avio_wb32(pb, size + flags_size + 11); // previous tag size
1032  flv->duration = FFMAX(flv->duration,
1033  pkt->pts + flv->delay + pkt->duration);
1034  }
1035 
1036  if (flv->flags & FLV_ADD_KEYFRAME_INDEX) {
1037  switch (par->codec_type) {
1038  case AVMEDIA_TYPE_VIDEO:
1039  flv->videosize += (avio_tell(pb) - cur_offset);
1040  flv->lasttimestamp = flv->acurframeindex / flv->framerate;
1041  if (pkt->flags & AV_PKT_FLAG_KEY) {
1042  double ts = flv->acurframeindex / flv->framerate;
1043  int64_t pos = cur_offset;
1044 
1045  flv->lastkeyframetimestamp = flv->acurframeindex / flv->framerate;
1046  flv->lastkeyframelocation = pos;
1047  flv_append_keyframe_info(s, flv, ts, pos);
1048  }
1049  flv->acurframeindex++;
1050  break;
1051 
1052  case AVMEDIA_TYPE_AUDIO:
1053  flv->audiosize += (avio_tell(pb) - cur_offset);
1054  break;
1055 
1056  default:
1057  av_log(s, AV_LOG_WARNING, "par->codec_type is type = [%d]\n", par->codec_type);
1058  break;
1059  }
1060  }
1061 
1062  av_free(data);
1063 
1064  return pb->error;
1065 }
