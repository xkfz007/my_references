101 /* Evil hack: detect incorrect assumptions that 32-bit ints are zero-extended to 64-bit.
102  * This is done by clobbering the stack with junk around the stack pointer and calling the
103  * assembly function through checked_call() with added dummy arguments which forces all
104  * real arguments to be passed on the stack and not in registers. For 32-bit arguments the
105  * upper half of the 64-bit register locations on the stack will now contain junk which will
106  * cause misbehaving functions to either produce incorrect output or segfault. Note that
107  * even though this works extremely well in practice, it's technically not guaranteed
108  * and false negatives is theoretically possible, but there can never be any false positives.
109  */
110 void checkasm_stack_clobber(uint64_t clobber, ...);

153 #ifdef AV_READ_TIME
154 #define bench_new(...)\
155  do {\
156  if (checkasm_bench_func()) {\
157  func_type *tfunc = func_new;\
158  uint64_t tsum = 0;\
159  int ti, tcount = 0;\
160  for (ti = 0; ti < BENCH_RUNS; ti++) {\
161  uint64_t t = AV_READ_TIME();\
162  tfunc(__VA_ARGS__);\
163  tfunc(__VA_ARGS__);\
164  tfunc(__VA_ARGS__);\
165  tfunc(__VA_ARGS__);\
166  t = AV_READ_TIME() - t;\
167  if (t*tcount <= tsum*4 && ti > 0) {\
168  tsum += t;\
169  tcount++;\
170  }\
171  }\
172  emms_c();\
173  checkasm_update_bench(tcount, tsum);\
174  }\
175  } while (0)
