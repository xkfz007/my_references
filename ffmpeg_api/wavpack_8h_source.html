163 static av_always_inline int wp_exp2(int16_t val)
164 {
165  int res, neg = 0;
166 
167  if (val < 0) {
168  val = -val;
169  neg = 1;
170  }
171 
172  res = wp_exp2_table[val & 0xFF] | 0x100;
173  val >>= 8;
174  res = (val > 9) ? (res << (val - 9)) : (res >> (9 - val));
175  return neg ? -res : res;
176 }

178 static av_always_inline int wp_log2(int32_t val)
179 {
180  int bits;
181 
182  if (!val)
183  return 0;
184  if (val == 1)
185  return 256;
186  val += val >> 9;
187  bits = av_log2(val) + 1;
188  if (bits < 9)
189  return (bits << 8) + wp_log2_table[(val << (9 - bits)) & 0xFF];
190  else
191  return (bits << 8) + wp_log2_table[(val >> (bits - 9)) & 0xFF];
192 }
