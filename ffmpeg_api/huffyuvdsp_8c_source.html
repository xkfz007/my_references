30 static void add_bytes_c(uint8_t *dst, uint8_t *src, intptr_t w)
31 {
32  long i;
33 
34  for (i = 0; i <= w - (int) sizeof(long); i += sizeof(long)) {
35  long a = *(long *) (src + i);
36  long b = *(long *) (dst + i);
37  *(long *) (dst + i) = ((a & pb_7f) + (b & pb_7f)) ^ ((a ^ b) & pb_80);
38  }
