55  register vec_s16 t0, t1, i0, i1, i4, i2, i3;
56  register vec_s32 res = zero_s32v;
57 #if HAVE_BIGENDIAN
58  register vec_u8 align = vec_lvsl(0, v2);
59  i2 = vec_ld(0, v2);
60  i3 = vec_ld(0, v3);
61 #endif
62  int32_t ires;
63 
64  order >>= 4;
65  do {
66  GET_T(t0,t1,v2,i1,i2);
67  i0 = pv1[0];
68  i1 = pv1[1];
69  res = vec_msum(t0, i0, res);
70  res = vec_msum(t1, i1, res);
71  GET_T(t0,t1,v3,i4,i3);
72  pv1[0] = vec_mladd(t0, muls, i0);
73  pv1[1] = vec_mladd(t1, muls, i1);
74  pv1 += 2;
75  v2 += 16;
76  v3 += 16;
77  } while (--order);
78  res = vec_splat(vec_sums(res, zero_s32v), 3);
79  vec_ste(res, 0, &ires);
