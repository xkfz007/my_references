46  int log_offset, void *log_ctx)
47 {
48  char c;
49  int ret;
50 
51  if (sscanf(str, "%d:%d%c", &q->num, &q->den, &c) != 2) {
52  double d;
53  ret = av_expr_parse_and_eval(&d, str, NULL, NULL,
54  NULL, NULL, NULL, NULL,
55  NULL, log_offset, log_ctx);

57  return ret;
58  *q = av_d2q(d, max);

144 {

147  const char *p;

152  width = video_size_abbrs[i].width;
153  height = video_size_abbrs[i].height;

176 int av_parse_video_rate(AVRational *rate, const char *arg)
177 {
178  int i, ret;
179  int n = FF_ARRAY_ELEMS(video_rate_abbrs);
180 
181  /* First, we check our abbreviation table */
182  for (i = 0; i < n; ++i)
183  if (!strcmp(video_rate_abbrs[i].abbr, arg)) {
184  *rate = video_rate_abbrs[i].rate;
185  return 0;
186  }
187 
188  /* Then, we try to parse it as fraction */
189  if ((ret = av_parse_ratio_quiet(rate, arg, 1001000)) < 0)
190  return ret;
191  if (rate->num <= 0 || rate->den <= 0)
192  return AVERROR(EINVAL);
193  return 0;

351 int av_parse_color(uint8_t *rgba_color, const char *color_string, int slen,
352  void *log_ctx)
353 {
354  char *tail, color_string2[128];
355  const ColorEntry *entry;
356  int len, hex_offset = 0;
357 
358  if (color_string[0] == '#') {
359  hex_offset = 1;
360  } else if (!strncmp(color_string, "0x", 2))
361  hex_offset = 2;
362 
363  if (slen < 0)
364  slen = strlen(color_string);
365  av_strlcpy(color_string2, color_string + hex_offset,
366  FFMIN(slen-hex_offset+1, sizeof(color_string2)));
367  if ((tail = strchr(color_string2, ALPHA_SEP)))
368  *tail++ = 0;
369  len = strlen(color_string2);
370  rgba_color[3] = 255;
371 
372  if (!av_strcasecmp(color_string2, "random") || !av_strcasecmp(color_string2, "bikeshed")) {
373  int rgba = av_get_random_seed();
374  rgba_color[0] = rgba >> 24;
375  rgba_color[1] = rgba >> 16;
376  rgba_color[2] = rgba >> 8;
377  rgba_color[3] = rgba;
378  } else if (hex_offset ||
379  strspn(color_string2, "0123456789ABCDEFabcdef") == len) {
380  char *tail;
381  unsigned int rgba = strtoul(color_string2, &tail, 16);
382 
383  if (*tail || (len != 6 && len != 8)) {
384  av_log(log_ctx, AV_LOG_ERROR, "Invalid 0xRRGGBB[AA] color string: '%s'\n", color_string2);
385  return AVERROR(EINVAL);
386  }
387  if (len == 8) {
388  rgba_color[3] = rgba;
389  rgba >>= 8;
390  }
391  rgba_color[0] = rgba >> 16;
392  rgba_color[1] = rgba >> 8;
393  rgba_color[2] = rgba;
394  } else {
395  entry = bsearch(color_string2,
396  color_table,
397  FF_ARRAY_ELEMS(color_table),
398  sizeof(ColorEntry),
399  color_table_compare);
400  if (!entry) {
401  av_log(log_ctx, AV_LOG_ERROR, "Cannot find color '%s'\n", color_string2);
402  return AVERROR(EINVAL);
403  }
404  memcpy(rgba_color, entry->rgb_color, 3);
405  }
406 
407  if (tail) {
408  double alpha;
409  const char *alpha_string = tail;
410  if (!strncmp(alpha_string, "0x", 2)) {
411  alpha = strtoul(alpha_string, &tail, 16);
412  } else {
413  double norm_alpha = strtod(alpha_string, &tail);
414  if (norm_alpha < 0.0 || norm_alpha > 1.0)
415  alpha = 256;
416  else
417  alpha = 255 * norm_alpha;
418  }
419 
420  if (tail == alpha_string || *tail || alpha > 255 || alpha < 0) {
421  av_log(log_ctx, AV_LOG_ERROR, "Invalid alpha value specifier '%s' in '%s'\n",
422  alpha_string, color_string);
423  return AVERROR(EINVAL);
424  }
425  rgba_color[3] = alpha;
426  }
427 
428  return 0;
429 }

447 static int date_get_num(const char **pp,
448  int n_min, int n_max, int len_max)
449 {
450  int i, val, c;
451  const char *p;

453  p = *pp;

456  c = *p;

471 char *av_small_strptime(const char *p, const char *fmt, struct tm *dt)
472 {
473  int c, val;
474 
475  while((c = *fmt++)) {
476  if (c != '%') {
477  if (av_isspace(c))
478  for (; *p && av_isspace(*p); p++);
479  else if (*p != c)
480  return NULL;
481  else p++;
482  continue;
483  }
484 
485  c = *fmt++;
486  switch(c) {
487  case 'H':
488  case 'J':
489  val = date_get_num(&p, 0, c == 'H' ? 23 : INT_MAX, 2);
490 
491  if (val == -1)
492  return NULL;
493  dt->tm_hour = val;
494  break;
495  case 'M':
496  val = date_get_num(&p, 0, 59, 2);
497  if (val == -1)
498  return NULL;
499  dt->tm_min = val;
500  break;
501  case 'S':
502  val = date_get_num(&p, 0, 59, 2);
503  if (val == -1)
504  return NULL;
505  dt->tm_sec = val;
506  break;
507  case 'Y':
508  val = date_get_num(&p, 0, 9999, 4);
509  if (val == -1)
510  return NULL;
511  dt->tm_year = val - 1900;
512  break;
513  case 'm':
514  val = date_get_num(&p, 1, 12, 2);
515  if (val == -1)
516  return NULL;
517  dt->tm_mon = val - 1;
518  break;
519  case 'd':
520  val = date_get_num(&p, 1, 31, 2);
521  if (val == -1)
522  return NULL;
523  dt->tm_mday = val;
524  break;
525  case 'T':
526  p = av_small_strptime(p, "%H:%M:%S", dt);
527  if (!p)
528  return NULL;
529  break;
530  case '%':
531  if (*p++ != '%')
532  return NULL;
533  break;
534  default:
535  return NULL;
536  }
537  }
538 
539  return (char*)p;
540 }

598  if (q)

617  for (i = 0; i < FF_ARRAY_ELEMS(time_fmt); i++) {
618  q = av_small_strptime(p, time_fmt[i], &dt);
619  if (q)

624  if (p[0] == '-') {
625  negative = 1;
626  ++p;
627  }

629  q = av_small_strptime(p, "%J:%M:%S", &dt);
630  if (!q) {

632  q = av_small_strptime(p, "%M:%S", &dt);
633  dt.tm_hour = 0;
634  }

638  dt.tm_sec = strtol(p, &o, 10);

640  return AVERROR(EINVAL);
641  dt.tm_min = 0;
642  dt.tm_hour = 0;
643  q = o;
644  }
645  }

648  if (!q)
649  return AVERROR(EINVAL);

652  if (*q == '.') {
653  int n;
654  q++;
655  for (n = 100000; n >= 1; n /= 10, q++) {
656  if (!av_isdigit(*q))
657  break;
658  microseconds += n * (*q - '0');
659  }
660  while (av_isdigit(*q))
661  q++;
662  }
663

668  int tzoffset = 0;

673  q++;

683  is_utc = 1;
684  }

686  struct tm dt2 = is_utc ? *gmtime_r(&now, &tmbuf) : *localtime_r(&now, &tmbuf);
687  dt2.tm_hour = dt.tm_hour;
688  dt2.tm_min = dt.tm_min;
689  dt2.tm_sec = dt.tm_sec;
690  dt = dt2;
691  }

694  t += tzoffset;
695  }

698  if (*q)

702  t += microseconds;
703  *timeval = negative ? -t : t;
