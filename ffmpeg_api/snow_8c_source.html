72  int edges_needed = av_codec_is_encoder(s->avctx->codec);

75  frame->height = s->avctx->height;

80  if ((ret = ff_get_buffer(s->avctx, frame, AV_GET_BUFFER_FLAG_REF)) < 0)
81  return ret;
82  if (edges_needed) {

88  }
89  frame->width = s->avctx->width;
