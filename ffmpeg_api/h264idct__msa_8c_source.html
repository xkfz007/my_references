56 static void avc_idct4x4_addblk_dc_msa(uint8_t *dst, int16_t *src,
57  int32_t dst_stride)
58 {
59  int16_t dc;
60  uint32_t src0, src1, src2, src3;
61  v16u8 pred = { 0 };
62  v16i8 out;
63  v8i16 input_dc, pred_r, pred_l;
64 
65  dc = (src[0] + 32) >> 6;

67  src[0] = 0;
