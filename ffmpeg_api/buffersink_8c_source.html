88 static int add_buffer_ref(AVFilterContext *ctx, AVFrame *ref)
89 {
90  BufferSinkContext *buf = ctx->priv;
91 
92  if (av_fifo_space(buf->fifo) < FIFO_INIT_ELEMENT_SIZE) {
93  /* realloc fifo size */
94  if (av_fifo_realloc2(buf->fifo, av_fifo_size(buf->fifo) * 2) < 0) {
95  av_log(ctx, AV_LOG_ERROR,
96  "Cannot buffer more frames. Consume some available frames "
97  "before adding new ones.\n");
98  return AVERROR(ENOMEM);
99  }
100  }
101 
102  /* cache frame */
103  av_fifo_generic_write(buf->fifo, &ref, FIFO_INIT_ELEMENT_SIZE, NULL);
104  return 0;
105 }

107 static int filter_frame(AVFilterLink *link, AVFrame *frame)
108 {
109  AVFilterContext *ctx = link->dst;
110  BufferSinkContext *buf = link->dst->priv;
111  int ret;
112 
113  if ((ret = add_buffer_ref(ctx, frame)) < 0)
114  return ret;
115  if (buf->warning_limit &&
116  av_fifo_size(buf->fifo) / FIFO_INIT_ELEMENT_SIZE >= buf->warning_limit) {
117  av_log(ctx, AV_LOG_WARNING,
118  "%d buffers queued in %s, something may be wrong.\n",
119  buf->warning_limit,
120  (char *)av_x_if_null(ctx->name, ctx->filter->name));
121  buf->warning_limit *= 10;
122  }
123  return 0;
124 }

131 int attribute_align_arg av_buffersink_get_frame_flags(AVFilterContext *ctx, AVFrame *frame, int flags)
132 {
133  BufferSinkContext *buf = ctx->priv;
134  AVFilterLink *inlink = ctx->inputs[0];
135  int peek_in_framequeue = 0, ret;
136  int64_t frame_count;
137  AVFrame *cur_frame;
138 
139  /* no picref available, fetch it from the filterchain */
140  while (!av_fifo_size(buf->fifo)) {
141  /* if peek_in_framequeue is true later, then ff_request_frame() and
142  the ff_filter_graph_run_once() loop will take a frame from it and
143  move it to the internal fifo, ending the global loop */
144  av_assert0(!peek_in_framequeue);
145  if (inlink->status_out)
146  return inlink->status_out;
147  peek_in_framequeue = ff_framequeue_queued_frames(&inlink->fifo) &&
148  ff_framequeue_queued_samples(&inlink->fifo) >= inlink->min_samples;
149  if ((flags & AV_BUFFERSINK_FLAG_NO_REQUEST) && !peek_in_framequeue)
150  return AVERROR(EAGAIN);
151  if ((ret = ff_request_frame(inlink)) < 0)
152  return ret;
153  frame_count = inlink->frame_count_out;
154  while (frame_count == inlink->frame_count_out) {
155  ret = ff_filter_graph_run_once(ctx->graph);
156  if (ret < 0)
157  return ret;
158  }
159  }
160 
161  if (flags & AV_BUFFERSINK_FLAG_PEEK) {
162  cur_frame = *((AVFrame **)av_fifo_peek2(buf->fifo, 0));
163  if ((ret = av_frame_ref(frame, cur_frame)) < 0)
164  return ret;
165  } else {
166  av_fifo_generic_read(buf->fifo, &cur_frame, sizeof(cur_frame), NULL);
167  av_frame_move_ref(frame, cur_frame);
168  av_frame_free(&cur_frame);
169  }
170 
171  return 0;
172 }

174 static int read_from_fifo(AVFilterContext *ctx, AVFrame *frame,
175  int nb_samples)
176 {
177  BufferSinkContext *s = ctx->priv;
178  AVFilterLink *link = ctx->inputs[0];
179  AVFrame *tmp;
180 
181  if (!(tmp = ff_get_audio_buffer(link, nb_samples)))
182  return AVERROR(ENOMEM);
183  av_audio_fifo_read(s->audio_fifo, (void**)tmp->extended_data, nb_samples);
184 
185  tmp->pts = s->next_pts;
186  if (s->next_pts != AV_NOPTS_VALUE)
187  s->next_pts += av_rescale_q(nb_samples, (AVRational){1, link->sample_rate},
188  link->time_base);
189 
190  av_frame_move_ref(frame, tmp);
191  av_frame_free(&tmp);
192 
193  return 0;
194 }

196 int attribute_align_arg av_buffersink_get_samples(AVFilterContext *ctx,
197  AVFrame *frame, int nb_samples)
198 {
199  BufferSinkContext *s = ctx->priv;
200  AVFilterLink *link = ctx->inputs[0];
201  AVFrame *cur_frame;
202  int ret = 0;
203 
204  if (!s->audio_fifo) {
205  int nb_channels = link->channels;
206  if (!(s->audio_fifo = av_audio_fifo_alloc(link->format, nb_channels, nb_samples)))
207  return AVERROR(ENOMEM);
208  }
209 
210  while (ret >= 0) {
211  if (av_audio_fifo_size(s->audio_fifo) >= nb_samples)
212  return read_from_fifo(ctx, frame, nb_samples);
213 
214  if (!(cur_frame = av_frame_alloc()))
215  return AVERROR(ENOMEM);
216  ret = av_buffersink_get_frame_flags(ctx, cur_frame, 0);
217  if (ret == AVERROR_EOF && av_audio_fifo_size(s->audio_fifo)) {
218  av_frame_free(&cur_frame);
219  return read_from_fifo(ctx, frame, av_audio_fifo_size(s->audio_fifo));
220  } else if (ret < 0) {
221  av_frame_free(&cur_frame);
222  return ret;
223  }
224 
225  if (cur_frame->pts != AV_NOPTS_VALUE) {
226  s->next_pts = cur_frame->pts -
227  av_rescale_q(av_audio_fifo_size(s->audio_fifo),
228  (AVRational){ 1, link->sample_rate },
229  link->time_base);
230  }
231 
232  ret = av_audio_fifo_write(s->audio_fifo, (void**)cur_frame->extended_data,
233  cur_frame->nb_samples);
234  av_frame_free(&cur_frame);
235  }
236 
237  return ret;
238 }

350 static int asink_query_formats(AVFilterContext *ctx)
351 {
352  BufferSinkContext *buf = ctx->priv;
353  AVFilterFormats *formats = NULL;
354  AVFilterChannelLayouts *layouts = NULL;
355  unsigned i;
356  int ret;
357 
358  CHECK_LIST_SIZE(sample_fmts)
359  CHECK_LIST_SIZE(sample_rates)
360  CHECK_LIST_SIZE(channel_layouts)
361  CHECK_LIST_SIZE(channel_counts)
362 
363  if (buf->sample_fmts_size) {
364  for (i = 0; i < NB_ITEMS(buf->sample_fmts); i++)
365  if ((ret = ff_add_format(&formats, buf->sample_fmts[i])) < 0)
366  return ret;
367  if ((ret = ff_set_common_formats(ctx, formats)) < 0)
368  return ret;
369  }
370 
371  if (buf->channel_layouts_size || buf->channel_counts_size ||
372  buf->all_channel_counts) {
373  for (i = 0; i < NB_ITEMS(buf->channel_layouts); i++)
374  if ((ret = ff_add_channel_layout(&layouts, buf->channel_layouts[i])) < 0)
375  return ret;
376  for (i = 0; i < NB_ITEMS(buf->channel_counts); i++)
377  if ((ret = ff_add_channel_layout(&layouts, FF_COUNT2LAYOUT(buf->channel_counts[i]))) < 0)
378  return ret;
379  if (buf->all_channel_counts) {
380  if (layouts)
381  av_log(ctx, AV_LOG_WARNING,
382  "Conflicting all_channel_counts and list in options\n");
383  else if (!(layouts = ff_all_channel_counts()))
384  return AVERROR(ENOMEM);
385  }
386  if ((ret = ff_set_common_channel_layouts(ctx, layouts)) < 0)
387  return ret;
388  }
389 
390  if (buf->sample_rates_size) {
391  formats = NULL;
392  for (i = 0; i < NB_ITEMS(buf->sample_rates); i++)
393  if ((ret = ff_add_format(&formats, buf->sample_rates[i])) < 0)
394  return ret;
395  if ((ret = ff_set_common_samplerates(ctx, formats)) < 0)
396  return ret;
397  }
398 
399  return 0;
400 }
