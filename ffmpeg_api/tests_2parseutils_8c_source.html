170  "2012 - 02-22 17:44:07",
171  "2000-12-20T010247.6",
172  "2000-12-12 1:35:46+05:30",
173  "2002-12-12 22:30:40-02",
174  };
175  static const char * const duration_string[] = {
176  "2:34:56.79",
177  "-1:23:45.67",
178  "42.1729",
179  "-1729.42",
180  "12:34",
181  };
182 
183  av_log_set_level(AV_LOG_DEBUG);
184  putenv(tzstr);
185  printf("(now is 2012-03-17 09:14:13.2 +0100, local time is UTC+1)\n");
186  for (i = 0; i < FF_ARRAY_ELEMS(time_string); i++) {
187  printf("%-24s -> ", time_string[i]);
188  if (av_parse_time(&tv, time_string[i], 0)) {
189  printf("error\n");
190  } else {
191  tvi = tv / 1000000;
192  tm = gmtime(&tvi);
193  printf("%14"PRIi64".%06d = %04d-%02d-%02dT%02d:%02d:%02dZ\n",
194  tv / 1000000, (int)(tv % 1000000),
195  tm->tm_year + 1900, tm->tm_mon + 1, tm->tm_mday,
196  tm->tm_hour, tm->tm_min, tm->tm_sec);
197  }
198  }
199  for (i = 0; i < FF_ARRAY_ELEMS(duration_string); i++) {
200  printf("%-24s -> ", duration_string[i]);
201  if (av_parse_time(&tv, duration_string[i], 1)) {
202  printf("error\n");
203  } else {
204  printf("%+21"PRIi64"\n", tv);
205  }
206  }
207 }
