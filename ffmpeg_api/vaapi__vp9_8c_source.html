27 static void fill_picture_parameters(AVCodecContext *avctx,
28  const VP9SharedContext *h,
29  VADecPictureParameterBufferVP9 *pp)
30 {
31  const AVPixFmtDescriptor *pixdesc = av_pix_fmt_desc_get(avctx->sw_pix_fmt);
32  int i;
33 
34  pp->frame_width = avctx->width;
35  pp->frame_height = avctx->height;
36 
37  pp->frame_header_length_in_bytes = h->h.uncompressed_header_size;
38  pp->first_partition_size = h->h.compressed_header_size;
39 
40  pp->profile = h->h.profile;
41  pp->bit_depth = h->h.bpp;
42 
43  pp->filter_level = h->h.filter.level;
44  pp->sharpness_level = h->h.filter.sharpness;
45  pp->log2_tile_rows = h->h.tiling.log2_tile_rows;
46  pp->log2_tile_columns = h->h.tiling.log2_tile_cols;
47 
48  pp->pic_fields.bits.subsampling_x = pixdesc->log2_chroma_w;
49  pp->pic_fields.bits.subsampling_y = pixdesc->log2_chroma_h;
50  pp->pic_fields.bits.frame_type = !h->h.keyframe;
51  pp->pic_fields.bits.show_frame = !h->h.invisible;
52  pp->pic_fields.bits.error_resilient_mode = h->h.errorres;
53  pp->pic_fields.bits.intra_only = h->h.intraonly;
54  pp->pic_fields.bits.allow_high_precision_mv = h->h.keyframe ? 0 : h->h.highprecisionmvs;
55  pp->pic_fields.bits.mcomp_filter_type = h->h.filtermode ^ (h->h.filtermode <= 1);
56  pp->pic_fields.bits.frame_parallel_decoding_mode = h->h.parallelmode;
57  pp->pic_fields.bits.reset_frame_context = h->h.resetctx;
58  pp->pic_fields.bits.refresh_frame_context = h->h.refreshctx;
59  pp->pic_fields.bits.frame_context_idx = h->h.framectxid;
60 
61  pp->pic_fields.bits.segmentation_enabled = h->h.segmentation.enabled;
62  pp->pic_fields.bits.segmentation_temporal_update = h->h.segmentation.temporal;
63  pp->pic_fields.bits.segmentation_update_map = h->h.segmentation.update_map;
64 
65  pp->pic_fields.bits.last_ref_frame = h->h.refidx[0];
66  pp->pic_fields.bits.last_ref_frame_sign_bias = h->h.signbias[0];
67  pp->pic_fields.bits.golden_ref_frame = h->h.refidx[1];
68  pp->pic_fields.bits.golden_ref_frame_sign_bias = h->h.signbias[1];
69  pp->pic_fields.bits.alt_ref_frame = h->h.refidx[2];
70  pp->pic_fields.bits.alt_ref_frame_sign_bias = h->h.signbias[2];
71  pp->pic_fields.bits.lossless_flag = h->h.lossless;
72 
73  for (i = 0; i < 7; i++)
74  pp->mb_segment_tree_probs[i] = h->h.segmentation.prob[i];
75 
76  if (h->h.segmentation.temporal) {
77  for (i = 0; i < 3; i++)
78  pp->segment_pred_probs[i] = h->h.segmentation.pred_prob[i];
79  } else {
80  memset(pp->segment_pred_probs, 255, sizeof(pp->segment_pred_probs));
81  }
82 
83  for (i = 0; i < 8; i++) {
84  if (h->refs[i].f->buf[0]) {
85  pp->reference_frames[i] = ff_vaapi_get_surface_id(h->refs[i].f);
86  } else {
87  pp->reference_frames[i] = VA_INVALID_ID;
88  }
89  }
90 }

92 static int vaapi_vp9_start_frame(AVCodecContext *avctx,
93  av_unused const uint8_t *buffer,
94  av_unused uint32_t size)
95 {
96  const VP9SharedContext *h = avctx->priv_data;
97  FFVAContext * const vactx = ff_vaapi_get_context(avctx);
98  VADecPictureParameterBufferVP9 *pic_param;
99 
100  vactx->slice_param_size = sizeof(VASliceParameterBufferVP9);
101 
102  pic_param = ff_vaapi_alloc_pic_param(vactx, sizeof(VADecPictureParameterBufferVP9));
103  if (!pic_param)
104  return -1;
105  fill_picture_parameters(avctx, h, pic_param);
106 
107  return 0;
108 }

110 static int vaapi_vp9_end_frame(AVCodecContext *avctx)
111 {
112  FFVAContext * const vactx = ff_vaapi_get_context(avctx);
113  const VP9SharedContext *h = avctx->priv_data;
114  int ret;
115 
116  ret = ff_vaapi_commit_slices(vactx);
117  if (ret < 0)
118  goto finish;
119 
120  ret = ff_vaapi_render_picture(vactx, ff_vaapi_get_surface_id(h->frames[CUR_FRAME].tf.f));
121  if (ret < 0)
122  goto finish;
123 
124 finish:
125  ff_vaapi_common_end_frame(avctx);
126  return ret;
127 }

129 static int vaapi_vp9_decode_slice(AVCodecContext *avctx,
130  const uint8_t *buffer,
131  uint32_t size)
132 {
133  FFVAContext * const vactx = ff_vaapi_get_context(avctx);
134  const VP9SharedContext *h = avctx->priv_data;
135  VASliceParameterBufferVP9 *slice_param;
136  int i;
137 
138  slice_param = (VASliceParameterBufferVP9*)ff_vaapi_alloc_slice(vactx, buffer, size);
139  if (!slice_param)
140  return -1;
141 
142  for (i = 0; i < 8; i++) {
143  slice_param->seg_param[i].segment_flags.fields.segment_reference_enabled = h->h.segmentation.feat[i].ref_enabled;
144  slice_param->seg_param[i].segment_flags.fields.segment_reference = h->h.segmentation.feat[i].ref_val;
145  slice_param->seg_param[i].segment_flags.fields.segment_reference_skipped = h->h.segmentation.feat[i].skip_enabled;
146 
147  memcpy(slice_param->seg_param[i].filter_level, h->h.segmentation.feat[i].lflvl, sizeof(slice_param->seg_param[i].filter_level));
148 
149  slice_param->seg_param[i].luma_dc_quant_scale = h->h.segmentation.feat[i].qmul[0][0];
150  slice_param->seg_param[i].luma_ac_quant_scale = h->h.segmentation.feat[i].qmul[0][1];
151  slice_param->seg_param[i].chroma_dc_quant_scale = h->h.segmentation.feat[i].qmul[1][0];
152  slice_param->seg_param[i].chroma_ac_quant_scale = h->h.segmentation.feat[i].qmul[1][1];
153  }
154 
155  return 0;
156 }
