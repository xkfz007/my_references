30 {

132 /** Pack FFmpeg bitplanes into a VABitPlaneBuffer element */
133 static inline void vc1_pack_bitplanes(uint8_t *bitplane, int n, const uint8_t *ff_bp[3], int x, int y, int stride)
134 {
135  const int bitplane_index = n / 2;
136  const int ff_bp_index = y * stride + x;
137  uint8_t v = 0;
138  if (ff_bp[0])
139  v = ff_bp[0][ff_bp_index];
140  if (ff_bp[1])
141  v |= ff_bp[1][ff_bp_index] << 1;
142  if (ff_bp[2])
143  v |= ff_bp[2][ff_bp_index] << 2;
144  bitplane[bitplane_index] = (bitplane[bitplane_index] << 4) | v;
145 }

312 static int vaapi_vc1_decode_slice(AVCodecContext *avctx, const uint8_t *buffer, uint32_t size)
313 {
314  VC1Context * const v = avctx->priv_data;
315  MpegEncContext * const s = &v->s;
316  FFVAContext * const vactx = ff_vaapi_get_context(avctx);
317  VASliceParameterBufferVC1 *slice_param;
318 
319  /* Current bit buffer is beyond any marker for VC-1, so skip it */
320  if (avctx->codec_id == AV_CODEC_ID_VC1 && IS_MARKER(AV_RB32(buffer))) {
321  buffer += 4;
322  size -= 4;
323  }
324 
325  /* Fill in VASliceParameterBufferVC1 */
326  slice_param = (VASliceParameterBufferVC1 *)ff_vaapi_alloc_slice(vactx, buffer, size);
327  if (!slice_param)
328  return -1;
329  slice_param->macroblock_offset = get_bits_count(&s->gb);
330  slice_param->slice_vertical_position = s->mb_y;
331  return 0;
332 }
