35 static void h261_loop_filter(uint8_t *src, int stride)
36 {
37  int x, y, xy, yz;
38  int temp[64];
39 
40  for (x = 0; x < 8; x++) {
41  temp[x] = 4 * src[x];
42  temp[x + 7 * 8] = 4 * src[x + 7 * stride];
43  }
44  for (y = 1; y < 7; y++) {
45  for (x = 0; x < 8; x++) {
46  xy = y * stride + x;
47  yz = y * 8 + x;
48  temp[yz] = src[xy - stride] + 2 * src[xy] + src[xy + stride];
49  }
50  }
51 
52  for (y = 0; y < 8; y++) {
53  src[y * stride] = (temp[y * 8] + 2) >> 2;
54  src[y * stride + 7] = (temp[y * 8 + 7] + 2) >> 2;
55  for (x = 1; x < 7; x++) {
56  xy = y * stride + x;
57  yz = y * 8 + x;
58  src[xy] = (temp[yz - 1] + 2 * temp[yz] + temp[yz + 1] + 8) >> 4;
59  }
60  }
61 }
