26 static int probe(AVProbeData *p)
27 {
28  if (p->buf_size < 2096)
29  return 0;
30  if ( AV_RL32(p->buf ) != 1
31  || AV_RL32(p->buf + 8) > 100000
32  || AV_RL32(p->buf + 12) > 8
33  || AV_RL32(p->buf + 16) != 2096
34  ||!AV_RL32(p->buf + 21)
35  || AV_RL16(p->buf + 25) != 2096
36  || AV_RL32(p->buf + 48) % AV_RL32(p->buf + 21)
37  )
38  return 0;
39  return AVPROBE_SCORE_EXTENSION;
40 }
