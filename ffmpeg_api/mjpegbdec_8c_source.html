34 static uint32_t read_offs(AVCodecContext *avctx, GetBitContext *gb, uint32_t size, const char *err_msg){
35  uint32_t offs= get_bits_long(gb, 32);
36  if(offs >= size){
37  av_log(avctx, AV_LOG_WARNING, err_msg, offs, size);
38  return 0;
39  }
40  return offs;
41 }

43 static int mjpegb_decode_frame(AVCodecContext *avctx,
44  void *data, int *got_frame,
45  AVPacket *avpkt)
46 {
47  const uint8_t *buf = avpkt->data;
48  int buf_size = avpkt->size;
49  MJpegDecodeContext *s = avctx->priv_data;
50  const uint8_t *buf_end, *buf_ptr;
51  GetBitContext hgb; /* for the header */
52  uint32_t dqt_offs, dht_offs, sof_offs, sos_offs, second_field_offs;
53  uint32_t field_size, sod_offs;
54  int ret;
55 
56  buf_ptr = buf;
57  buf_end = buf + buf_size;
58  s->got_picture = 0;
59 
60 read_header:
61  /* reset on every SOI */
62  s->restart_interval = 0;
63  s->restart_count = 0;
64  s->mjpb_skiptosod = 0;
65 
66  if (buf_end - buf_ptr >= 1 << 28)
67  return AVERROR_INVALIDDATA;
68 
69  init_get_bits(&hgb, buf_ptr, /*buf_size*/(buf_end - buf_ptr)*8);
70 
71  skip_bits(&hgb, 32); /* reserved zeros */
72 
73  if (get_bits_long(&hgb, 32) != MKBETAG('m','j','p','g'))
74  {
75  av_log(avctx, AV_LOG_WARNING, "not mjpeg-b (bad fourcc)\n");
76  return AVERROR_INVALIDDATA;
77  }
78 
79  field_size = get_bits_long(&hgb, 32); /* field size */
80  av_log(avctx, AV_LOG_DEBUG, "field size: 0x%"PRIx32"\n", field_size);
81  skip_bits(&hgb, 32); /* padded field size */
82  second_field_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "second_field_offs is %d and size is %d\n");
83  av_log(avctx, AV_LOG_DEBUG, "second field offs: 0x%"PRIx32"\n",
84  second_field_offs);
85 
86  dqt_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "dqt is %d and size is %d\n");
87  av_log(avctx, AV_LOG_DEBUG, "dqt offs: 0x%"PRIx32"\n", dqt_offs);
88  if (dqt_offs)
89  {
90  init_get_bits(&s->gb, buf_ptr+dqt_offs, (buf_end - (buf_ptr+dqt_offs))*8);
91  s->start_code = DQT;
92  if (ff_mjpeg_decode_dqt(s) < 0 &&
93  (avctx->err_recognition & AV_EF_EXPLODE))
94  return AVERROR_INVALIDDATA;
95  }
96 
97  dht_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "dht is %d and size is %d\n");
98  av_log(avctx, AV_LOG_DEBUG, "dht offs: 0x%"PRIx32"\n", dht_offs);
99  if (dht_offs)
100  {
101  init_get_bits(&s->gb, buf_ptr+dht_offs, (buf_end - (buf_ptr+dht_offs))*8);
102  s->start_code = DHT;
103  ff_mjpeg_decode_dht(s);
104  }
105 
106  sof_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "sof is %d and size is %d\n");
107  av_log(avctx, AV_LOG_DEBUG, "sof offs: 0x%"PRIx32"\n", sof_offs);
108  if (sof_offs)
109  {
110  init_get_bits(&s->gb, buf_ptr+sof_offs, (buf_end - (buf_ptr+sof_offs))*8);
111  s->start_code = SOF0;
112  if (ff_mjpeg_decode_sof(s) < 0)
113  return -1;
114  }
115 
116  sos_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "sos is %d and size is %d\n");
117  av_log(avctx, AV_LOG_DEBUG, "sos offs: 0x%"PRIx32"\n", sos_offs);
118  sod_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "sof is %d and size is %d\n");
119  av_log(avctx, AV_LOG_DEBUG, "sod offs: 0x%"PRIx32"\n", sod_offs);
120  if (sos_offs)
121  {
122  init_get_bits(&s->gb, buf_ptr + sos_offs,
123  8 * FFMIN(field_size, buf_end - buf_ptr - sos_offs));
124  s->mjpb_skiptosod = (sod_offs - sos_offs - show_bits(&s->gb, 16));
125  s->start_code = SOS;
126  if (ff_mjpeg_decode_sos(s, NULL, 0, NULL) < 0 &&
127  (avctx->err_recognition & AV_EF_EXPLODE))
128  return AVERROR_INVALIDDATA;
129  }
130 
131  if (s->interlaced) {
132  s->bottom_field ^= 1;
133  /* if not bottom field, do not output image yet */
134  if (s->bottom_field != s->interlace_polarity && second_field_offs)
135  {
136  buf_ptr = buf + second_field_offs;
137  goto read_header;
138  }
139  }
140 
141  //XXX FIXME factorize, this looks very similar to the EOI code
142 
143  if(!s->got_picture) {
144  av_log(avctx, AV_LOG_WARNING, "no picture\n");
145  return buf_size;
146  }
147 
148  if ((ret = av_frame_ref(data, s->picture_ptr)) < 0)
149  return ret;
150  *got_frame = 1;
151 
152  if (!s->lossless && avctx->debug & FF_DEBUG_QP) {
153  av_log(avctx, AV_LOG_DEBUG, "QP: %d\n",
154  FFMAX3(s->qscale[0], s->qscale[1], s->qscale[2]));
155  }
156 
157  return buf_size;
158 }
