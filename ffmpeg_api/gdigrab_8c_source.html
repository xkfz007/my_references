74 /**
75  * Callback to handle Windows messages for the region outline window.
76  *
77  * In particular, this handles painting the frame rectangle.
78  *
79  * @param hwnd The region outline window handle.
80  * @param msg The Windows message.
81  * @param wparam First Windows message parameter.
82  * @param lparam Second Windows message parameter.
83  * @return 0 success, !0 failure
84  */
85 static LRESULT CALLBACK
86 gdigrab_region_wnd_proc(HWND hwnd, UINT msg, WPARAM wparam, LPARAM lparam)
87 {
88  PAINTSTRUCT ps;
89  HDC hdc;
90  RECT rect;
91 
92  switch (msg) {
93  case WM_PAINT:
94  hdc = BeginPaint(hwnd, &ps);
95 
96  GetClientRect(hwnd, &rect);
97  FrameRect(hdc, &rect, GetStockObject(BLACK_BRUSH));
98 
99  rect.left++; rect.top++; rect.right--; rect.bottom--;
100  FrameRect(hdc, &rect, GetStockObject(WHITE_BRUSH));
101 
102  rect.left++; rect.top++; rect.right--; rect.bottom--;
103  FrameRect(hdc, &rect, GetStockObject(BLACK_BRUSH));
104 
105  EndPaint(hwnd, &ps);
106  break;
107  default:
108  return DefWindowProc(hwnd, msg, wparam, lparam);
109  }
110  return 0;
111 }

113 /**
114  * Initialize the region outline window.
115  *
116  * @param s1 The format context.
117  * @param gdigrab gdigrab context.
118  * @return 0 success, !0 failure
119  */
120 static int
121 gdigrab_region_wnd_init(AVFormatContext *s1, struct gdigrab *gdigrab)
122 {
123  HWND hwnd;
124  RECT rect = gdigrab->clip_rect;
125  HRGN region = NULL;
126  HRGN region_interior = NULL;
127 
128  DWORD style = WS_POPUP | WS_VISIBLE;
129  DWORD ex = WS_EX_TOOLWINDOW | WS_EX_TOPMOST | WS_EX_TRANSPARENT;
130 
131  rect.left -= REGION_WND_BORDER; rect.top -= REGION_WND_BORDER;
132  rect.right += REGION_WND_BORDER; rect.bottom += REGION_WND_BORDER;
133 
134  AdjustWindowRectEx(&rect, style, FALSE, ex);
135 
136  // Create a window with no owner; use WC_DIALOG instead of writing a custom
137  // window class
138  hwnd = CreateWindowEx(ex, WC_DIALOG, NULL, style, rect.left, rect.top,
139  rect.right - rect.left, rect.bottom - rect.top,
140  NULL, NULL, NULL, NULL);
141  if (!hwnd) {
142  WIN32_API_ERROR("Could not create region display window");
143  goto error;
144  }
145 
146  // Set the window shape to only include the border area
147  GetClientRect(hwnd, &rect);
148  region = CreateRectRgn(0, 0,
149  rect.right - rect.left, rect.bottom - rect.top);
150  region_interior = CreateRectRgn(REGION_WND_BORDER, REGION_WND_BORDER,
151  rect.right - rect.left - REGION_WND_BORDER,
152  rect.bottom - rect.top - REGION_WND_BORDER);
153  CombineRgn(region, region, region_interior, RGN_DIFF);
154  if (!SetWindowRgn(hwnd, region, FALSE)) {
155  WIN32_API_ERROR("Could not set window region");
156  goto error;
157  }
158  // The "region" memory is now owned by the window
159  region = NULL;
160  DeleteObject(region_interior);
161 
162  SetWindowLongPtr(hwnd, GWLP_WNDPROC, (LONG_PTR) gdigrab_region_wnd_proc);
163 
164  ShowWindow(hwnd, SW_SHOW);
165 
166  gdigrab->region_hwnd = hwnd;
167 
168  return 0;
169 
170 error:
171  if (region)
172  DeleteObject(region);
173  if (region_interior)
174  DeleteObject(region_interior);
175  if (hwnd)
176  DestroyWindow(hwnd);
177  return 1;
178 }

196  *
197  * This is important to prevent Windows from thinking the window has become
198  * unresponsive. As well, things like WM_PAINT (to actually draw the window
199  * contents) are handled from the message queue context.

202  * @param gdigrab gdigrab context.

206 {
207  HWND hwnd = gdigrab->region_hwnd;
208  MSG msg;
209

211  DispatchMessage(&msg);
212  }

215 /**
216  * Initializes the gdi grab device demuxer (public device demuxer API).
217  *
218  * @param s1 Context from avformat core
219  * @return AVERROR_IO error, 0 success
220  */
221 static int
222 gdigrab_read_header(AVFormatContext *s1)
223 {
224  struct gdigrab *gdigrab = s1->priv_data;
225 
226  HWND hwnd;
227  HDC source_hdc = NULL;
228  HDC dest_hdc = NULL;
229  BITMAPINFO bmi;
230  HBITMAP hbmp = NULL;
231  void *buffer = NULL;
232 
233  const char *filename = s1->filename;
234  const char *name = NULL;
235  AVStream *st = NULL;
236 
237  int bpp;
238  int vertres;
239  int desktopvertres;
240  RECT virtual_rect;
241  RECT clip_rect;
242  BITMAP bmp;
243  int ret;
244 
245  if (!strncmp(filename, "title=", 6)) {
246  name = filename + 6;
247  hwnd = FindWindow(NULL, name);
248  if (!hwnd) {
249  av_log(s1, AV_LOG_ERROR,
250  "Can't find window '%s', aborting.\n", name);
251  ret = AVERROR(EIO);
252  goto error;
253  }
254  if (gdigrab->show_region) {
255  av_log(s1, AV_LOG_WARNING,
256  "Can't show region when grabbing a window.\n");
257  gdigrab->show_region = 0;
258  }
259  } else if (!strcmp(filename, "desktop")) {
260  hwnd = NULL;
261  } else {
262  av_log(s1, AV_LOG_ERROR,
263  "Please use \"desktop\" or \"title=<windowname>\" to specify your target.\n");
264  ret = AVERROR(EIO);
265  goto error;
266  }
267 
268  /* This will get the device context for the selected window, or if
269  * none, the primary screen */
270  source_hdc = GetDC(hwnd);
271  if (!source_hdc) {
272  WIN32_API_ERROR("Couldn't get window device context");
273  ret = AVERROR(EIO);
274  goto error;
275  }
276  bpp = GetDeviceCaps(source_hdc, BITSPIXEL);
277 
278  if (hwnd) {
279  GetClientRect(hwnd, &virtual_rect);
280  } else {
281  /* desktop -- get the right height and width for scaling DPI */
282  vertres = GetDeviceCaps(source_hdc, VERTRES);
283  desktopvertres = GetDeviceCaps(source_hdc, DESKTOPVERTRES);
284  virtual_rect.left = GetSystemMetrics(SM_XVIRTUALSCREEN);
285  virtual_rect.top = GetSystemMetrics(SM_YVIRTUALSCREEN);
286  virtual_rect.right = (virtual_rect.left + GetSystemMetrics(SM_CXVIRTUALSCREEN)) * desktopvertres / vertres;
287  virtual_rect.bottom = (virtual_rect.top + GetSystemMetrics(SM_CYVIRTUALSCREEN)) * desktopvertres / vertres;
288  }
289 
290  /* If no width or height set, use full screen/window area */
291  if (!gdigrab->width || !gdigrab->height) {
292  clip_rect.left = virtual_rect.left;
293  clip_rect.top = virtual_rect.top;
294  clip_rect.right = virtual_rect.right;
295  clip_rect.bottom = virtual_rect.bottom;
296  } else {
297  clip_rect.left = gdigrab->offset_x;
298  clip_rect.top = gdigrab->offset_y;
299  clip_rect.right = gdigrab->width + gdigrab->offset_x;
300  clip_rect.bottom = gdigrab->height + gdigrab->offset_y;
301  }
302 
303  if (clip_rect.left < virtual_rect.left ||
304  clip_rect.top < virtual_rect.top ||
305  clip_rect.right > virtual_rect.right ||
306  clip_rect.bottom > virtual_rect.bottom) {
307  av_log(s1, AV_LOG_ERROR,
308  "Capture area (%li,%li),(%li,%li) extends outside window area (%li,%li),(%li,%li)",
309  clip_rect.left, clip_rect.top,
310  clip_rect.right, clip_rect.bottom,
311  virtual_rect.left, virtual_rect.top,
312  virtual_rect.right, virtual_rect.bottom);
313  ret = AVERROR(EIO);
314  goto error;
315  }
316 
317 
318  if (name) {
319  av_log(s1, AV_LOG_INFO,
320  "Found window %s, capturing %lix%lix%i at (%li,%li)\n",
321  name,
322  clip_rect.right - clip_rect.left,
323  clip_rect.bottom - clip_rect.top,
324  bpp, clip_rect.left, clip_rect.top);
325  } else {
326  av_log(s1, AV_LOG_INFO,
327  "Capturing whole desktop as %lix%lix%i at (%li,%li)\n",
328  clip_rect.right - clip_rect.left,
329  clip_rect.bottom - clip_rect.top,
330  bpp, clip_rect.left, clip_rect.top);
331  }
332 
333  if (clip_rect.right - clip_rect.left <= 0 ||
334  clip_rect.bottom - clip_rect.top <= 0 || bpp%8) {
335  av_log(s1, AV_LOG_ERROR, "Invalid properties, aborting\n");
336  ret = AVERROR(EIO);
337  goto error;
338  }
339 
340  dest_hdc = CreateCompatibleDC(source_hdc);
341  if (!dest_hdc) {
342  WIN32_API_ERROR("Screen DC CreateCompatibleDC");
343  ret = AVERROR(EIO);
344  goto error;
345  }
346 
347  /* Create a DIB and select it into the dest_hdc */
348  bmi.bmiHeader.biSize = sizeof(BITMAPINFOHEADER);
349  bmi.bmiHeader.biWidth = clip_rect.right - clip_rect.left;
350  bmi.bmiHeader.biHeight = -(clip_rect.bottom - clip_rect.top);
351  bmi.bmiHeader.biPlanes = 1;
352  bmi.bmiHeader.biBitCount = bpp;
353  bmi.bmiHeader.biCompression = BI_RGB;
354  bmi.bmiHeader.biSizeImage = 0;
355  bmi.bmiHeader.biXPelsPerMeter = 0;
356  bmi.bmiHeader.biYPelsPerMeter = 0;
357  bmi.bmiHeader.biClrUsed = 0;
358  bmi.bmiHeader.biClrImportant = 0;
359  hbmp = CreateDIBSection(dest_hdc, &bmi, DIB_RGB_COLORS,
360  &buffer, NULL, 0);
361  if (!hbmp) {
362  WIN32_API_ERROR("Creating DIB Section");
363  ret = AVERROR(EIO);
364  goto error;
365  }
366 
367  if (!SelectObject(dest_hdc, hbmp)) {
368  WIN32_API_ERROR("SelectObject");
369  ret = AVERROR(EIO);
370  goto error;
371  }
372 
373  /* Get info from the bitmap */
374  GetObject(hbmp, sizeof(BITMAP), &bmp);
375 
376  st = avformat_new_stream(s1, NULL);
377  if (!st) {
378  ret = AVERROR(ENOMEM);
379  goto error;
380  }
381  avpriv_set_pts_info(st, 64, 1, 1000000); /* 64 bits pts in us */
382 
383  gdigrab->frame_size = bmp.bmWidthBytes * bmp.bmHeight * bmp.bmPlanes;
384  gdigrab->header_size = sizeof(BITMAPFILEHEADER) + sizeof(BITMAPINFOHEADER) +
385  (bpp <= 8 ? (1 << bpp) : 0) * sizeof(RGBQUAD) /* palette size */;
386  gdigrab->time_base = av_inv_q(gdigrab->framerate);
387  gdigrab->time_frame = av_gettime() / av_q2d(gdigrab->time_base);
388 
389  gdigrab->hwnd = hwnd;
390  gdigrab->source_hdc = source_hdc;
391  gdigrab->dest_hdc = dest_hdc;
392  gdigrab->hbmp = hbmp;
393  gdigrab->bmi = bmi;
394  gdigrab->buffer = buffer;
395  gdigrab->clip_rect = clip_rect;
396 
397  gdigrab->cursor_error_printed = 0;
398 
399  if (gdigrab->show_region) {
400  if (gdigrab_region_wnd_init(s1, gdigrab)) {
401  ret = AVERROR(EIO);
402  goto error;
403  }
404  }
405 
406  st->avg_frame_rate = av_inv_q(gdigrab->time_base);
407 
408  st->codecpar->codec_type = AVMEDIA_TYPE_VIDEO;
409  st->codecpar->codec_id = AV_CODEC_ID_BMP;
410  st->codecpar->bit_rate = (gdigrab->header_size + gdigrab->frame_size) * 1/av_q2d(gdigrab->time_base) * 8;
411 
412  return 0;
413 
414 error:
415  if (source_hdc)
416  ReleaseDC(hwnd, source_hdc);
417  if (dest_hdc)
418  DeleteDC(dest_hdc);
419  if (hbmp)
420  DeleteObject(hbmp);
421  if (source_hdc)
422  DeleteDC(source_hdc);
423  return ret;
424 }

426 /**
427  * Paints a mouse pointer in a Win32 image.
428  *
429  * @param s1 Context of the log information
430  * @param s Current grad structure
431  */
432 static void paint_mouse_pointer(AVFormatContext *s1, struct gdigrab *gdigrab)
433 {
434  CURSORINFO ci = {0};
435 
436 #define CURSOR_ERROR(str) \
437  if (!gdigrab->cursor_error_printed) { \
438  WIN32_API_ERROR(str); \
439  gdigrab->cursor_error_printed = 1; \
440  }
441 
442  ci.cbSize = sizeof(ci);
443 
444  if (GetCursorInfo(&ci)) {
445  HCURSOR icon = CopyCursor(ci.hCursor);
446  ICONINFO info;
447  POINT pos;
448  RECT clip_rect = gdigrab->clip_rect;
449  HWND hwnd = gdigrab->hwnd;
450  int vertres = GetDeviceCaps(gdigrab->source_hdc, VERTRES);
451  int desktopvertres = GetDeviceCaps(gdigrab->source_hdc, DESKTOPVERTRES);
452  info.hbmMask = NULL;
453  info.hbmColor = NULL;
454 
455  if (ci.flags != CURSOR_SHOWING)
456  return;
457 
458  if (!icon) {
459  /* Use the standard arrow cursor as a fallback.
460  * You'll probably only hit this in Wine, which can't fetch
461  * the current system cursor. */
462  icon = CopyCursor(LoadCursor(NULL, IDC_ARROW));
463  }
464 
465  if (!GetIconInfo(icon, &info)) {
466  CURSOR_ERROR("Could not get icon info");
467  goto icon_error;
468  }
469 
470  pos.x = ci.ptScreenPos.x - clip_rect.left - info.xHotspot;
471  pos.y = ci.ptScreenPos.y - clip_rect.top - info.yHotspot;
472 
473  if (hwnd) {
474  RECT rect;
475 
476  if (GetWindowRect(hwnd, &rect)) {
477  pos.x -= rect.left;
478  pos.y -= rect.top;
479  } else {
480  CURSOR_ERROR("Couldn't get window rectangle");
481  goto icon_error;
482  }
483  }
484 
485  //that would keep the correct location of mouse with hidpi screens
486  pos.x = pos.x * desktopvertres / vertres;
487  pos.y = pos.y * desktopvertres / vertres;
488 
489  av_log(s1, AV_LOG_DEBUG, "Cursor pos (%li,%li) -> (%li,%li)\n",
490  ci.ptScreenPos.x, ci.ptScreenPos.y, pos.x, pos.y);
491 
492  if (pos.x >= 0 && pos.x <= clip_rect.right - clip_rect.left &&
493  pos.y >= 0 && pos.y <= clip_rect.bottom - clip_rect.top) {
494  if (!DrawIcon(gdigrab->dest_hdc, pos.x, pos.y, icon))
495  CURSOR_ERROR("Couldn't draw icon");
496  }
497 
498 icon_error:
499  if (info.hbmMask)
500  DeleteObject(info.hbmMask);
501  if (info.hbmColor)
502  DeleteObject(info.hbmColor);
503  if (icon)
504  DestroyCursor(icon);
505  } else {
506  CURSOR_ERROR("Couldn't get cursor info");
507  }
508 }

510 /**
511  * Grabs a frame from gdi (public device demuxer API).
512  *
513  * @param s1 Context from avformat core
514  * @param pkt Packet holding the grabbed frame
515  * @return frame size in bytes
516  */
517 static int gdigrab_read_packet(AVFormatContext *s1, AVPacket *pkt)
518 {
519  struct gdigrab *gdigrab = s1->priv_data;
520 
521  HDC dest_hdc = gdigrab->dest_hdc;
522  HDC source_hdc = gdigrab->source_hdc;
523  RECT clip_rect = gdigrab->clip_rect;
524  AVRational time_base = gdigrab->time_base;
525  int64_t time_frame = gdigrab->time_frame;
526 
527  BITMAPFILEHEADER bfh;
528  int file_size = gdigrab->header_size + gdigrab->frame_size;
529 
530  int64_t curtime, delay;
531 
532  /* Calculate the time of the next frame */
533  time_frame += INT64_C(1000000);
534 
535  /* Run Window message processing queue */
536  if (gdigrab->show_region)
537  gdigrab_region_wnd_update(s1, gdigrab);
538 
539  /* wait based on the frame rate */
540  for (;;) {
541  curtime = av_gettime();
542  delay = time_frame * av_q2d(time_base) - curtime;
543  if (delay <= 0) {
544  if (delay < INT64_C(-1000000) * av_q2d(time_base)) {
545  time_frame += INT64_C(1000000);
546  }
547  break;
548  }
549  if (s1->flags & AVFMT_FLAG_NONBLOCK) {
550  return AVERROR(EAGAIN);
551  } else {
552  av_usleep(delay);
553  }
554  }
555 
556  if (av_new_packet(pkt, file_size) < 0)
557  return AVERROR(ENOMEM);
558  pkt->pts = curtime;
559 
560  /* Blit screen grab */
561  if (!BitBlt(dest_hdc, 0, 0,
562  clip_rect.right - clip_rect.left,
563  clip_rect.bottom - clip_rect.top,
564  source_hdc,
565  clip_rect.left, clip_rect.top, SRCCOPY | CAPTUREBLT)) {
566  WIN32_API_ERROR("Failed to capture image");
567  return AVERROR(EIO);
568  }
569  if (gdigrab->draw_mouse)
570  paint_mouse_pointer(s1, gdigrab);
571 
572  /* Copy bits to packet data */
573 
574  bfh.bfType = 0x4d42; /* "BM" in little-endian */
575  bfh.bfSize = file_size;
576  bfh.bfReserved1 = 0;
577  bfh.bfReserved2 = 0;
578  bfh.bfOffBits = gdigrab->header_size;
579 
580  memcpy(pkt->data, &bfh, sizeof(bfh));
581 
582  memcpy(pkt->data + sizeof(bfh), &gdigrab->bmi.bmiHeader, sizeof(gdigrab->bmi.bmiHeader));
583 
584  if (gdigrab->bmi.bmiHeader.biBitCount <= 8)
585  GetDIBColorTable(dest_hdc, 0, 1 << gdigrab->bmi.bmiHeader.biBitCount,
586  (RGBQUAD *) (pkt->data + sizeof(bfh) + sizeof(gdigrab->bmi.bmiHeader)));
587 
588  memcpy(pkt->data + gdigrab->header_size, gdigrab->buffer, gdigrab->frame_size);
589 
590  gdigrab->time_frame = time_frame;
591 
592  return gdigrab->header_size + gdigrab->frame_size;
593 }

595 /**
596  * Closes gdi frame grabber (public device demuxer API).
597  *
598  * @param s1 Context from avformat core
599  * @return 0 success, !0 failure
600  */
601 static int gdigrab_read_close(AVFormatContext *s1)
602 {
603  struct gdigrab *s = s1->priv_data;
604 
605  if (s->show_region)
606  gdigrab_region_wnd_destroy(s1, s);
607 
608  if (s->source_hdc)
609  ReleaseDC(s->hwnd, s->source_hdc);
610  if (s->dest_hdc)
611  DeleteDC(s->dest_hdc);
612  if (s->hbmp)
613  DeleteObject(s->hbmp);
614  if (s->source_hdc)
615  DeleteDC(s->source_hdc);
616 
617  return 0;
618 }
