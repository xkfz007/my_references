199 static void hpeldsp_init_mmx(HpelDSPContext *c, int flags, int cpu_flags)
200 {
201  SET_HPEL_FUNCS(put, [0], 16, mmx);
202  SET_HPEL_FUNCS(put_no_rnd, [0], 16, mmx);
203  SET_HPEL_FUNCS(avg, [0], 16, mmx);
204  SET_HPEL_FUNCS(avg_no_rnd, , 16, mmx);
205  SET_HPEL_FUNCS(put, [1], 8, mmx);
206  SET_HPEL_FUNCS(put_no_rnd, [1], 8, mmx);
207  if (HAVE_MMX_EXTERNAL) {
208  c->avg_pixels_tab[1][0] = ff_avg_pixels8_mmx;
209  c->avg_pixels_tab[1][1] = ff_avg_pixels8_x2_mmx;
210  }
211 #if HAVE_MMX_INLINE
212  c->avg_pixels_tab[1][2] = avg_pixels8_y2_mmx;
213  c->avg_pixels_tab[1][3] = ff_avg_pixels8_xy2_mmx;
214 #endif
215 }
