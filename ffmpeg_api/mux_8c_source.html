55  * f = val + (num / den) + 0.5.
56  *
57  * 'num' is normalized so that it is such as 0 <= num < den.
58  *
59  * @param f fractional number
60  * @param val integer value
61  * @param num must be >= 0
62  * @param den must be >= 1

64 static void frac_init(FFFrac *f, int64_t val, int64_t num, int64_t den)
65 {
66  num += (den >> 1);
67  if (num >= den) {
68  val += num / den;
69  num = num % den;
70  }
71  f->val = val;
72  f->num = num;
73  f->den = den;
74 }

76 /**
77  * Fractional addition to f: f = f + (incr / f->den).
78  *
79  * @param f fractional number
80  * @param incr increment, can be positive or negative
81  */
82 static void frac_add(FFFrac *f, int64_t incr)
83 {
84  int64_t num, den;
85 
86  num = f->num + incr;
87  den = f->den;
88  if (num < 0) {
89  f->val += num / den;
90  num = num % den;
91  if (num < 0) {
92  num += den;
93  f->val--;
94  }
95  } else if (num >= den) {
96  f->val += num / den;
97  num = num % den;
98  }
99  f->num = num;
100 }

103 {
104  AVRational q;
105  int j;
106 
107  q = st->time_base;
108 
109  for (j=2; j<14; j+= 1+(j>2))
110  while (q.den / q.num < min_precision && q.num % j == 0)
111  q.num /= j;
112  while (q.den / q.num < min_precision && q.den < (1<<24))
113  q.den <<= 1;

202 static int validate_codec_tag(AVFormatContext *s, AVStream *st)
203 {
204  const AVCodecTag *avctag;
205  int n;
206  enum AVCodecID id = AV_CODEC_ID_NONE;
207  int64_t tag = -1;
208 
209  /**
210  * Check that tag + id is in the table
211  * If neither is in the table -> OK
212  * If tag is in the table with another id -> FAIL
213  * If id is in the table with another tag -> FAIL unless strict < normal
214  */
215  for (n = 0; s->oformat->codec_tag[n]; n++) {
216  avctag = s->oformat->codec_tag[n];
217  while (avctag->id != AV_CODEC_ID_NONE) {
218  if (avpriv_toupper4(avctag->tag) == avpriv_toupper4(st->codecpar->codec_tag)) {
219  id = avctag->id;
220  if (id == st->codecpar->codec_id)
221  return 1;
222  }
223  if (avctag->id == st->codecpar->codec_id)
224  tag = avctag->tag;
225  avctag++;
226  }
227  }
228  if (id != AV_CODEC_ID_NONE)
229  return 0;
230  if (tag >= 0 && (s->strict_std_compliance >= FF_COMPLIANCE_NORMAL))
231  return 0;
232  return 1;
233 }

690 {

692  int64_t pts_backup, dts_backup;
693 
694  pts_backup = pkt->pts;
695  dts_backup = pkt->dts;
696 
697  // If the timestamp offsetting below is adjusted, adjust
698  // ff_interleaved_peek similarly.
699  if (s->output_ts_offset) {
700  AVStream *st = s->streams[pkt->stream_index];
701  int64_t offset = av_rescale_q(s->output_ts_offset, AV_TIME_BASE_Q, st->time_base);
702 
703  if (pkt->dts != AV_NOPTS_VALUE)
704  pkt->dts += offset;
705  if (pkt->pts != AV_NOPTS_VALUE)
706  pkt->pts += offset;
707  }
708 
709  if (s->avoid_negative_ts > 0) {
710  AVStream *st = s->streams[pkt->stream_index];
711  int64_t offset = st->mux_ts_offset;
712  int64_t ts = s->internal->avoid_negative_ts_use_pts ? pkt->pts : pkt->dts;
713 
714  if (s->internal->offset == AV_NOPTS_VALUE && ts != AV_NOPTS_VALUE &&
715  (ts < 0 || s->avoid_negative_ts == AVFMT_AVOID_NEG_TS_MAKE_ZERO)) {
716  s->internal->offset = -ts;
717  s->internal->offset_timebase = st->time_base;
718  }

720  if (s->internal->offset != AV_NOPTS_VALUE && !offset) {
721  offset = st->mux_ts_offset =
722  av_rescale_q_rnd(s->internal->offset,
723  s->internal->offset_timebase,
724  st->time_base,
725  AV_ROUND_UP);
726  }
727 
728  if (pkt->dts != AV_NOPTS_VALUE)
729  pkt->dts += offset;
730  if (pkt->pts != AV_NOPTS_VALUE)
731  pkt->pts += offset;
732 
733  if (s->internal->avoid_negative_ts_use_pts) {
734  if (pkt->pts != AV_NOPTS_VALUE && pkt->pts < 0) {
735  av_log(s, AV_LOG_WARNING, "failed to avoid negative "
736  "pts %s in stream %d.\n"
737  "Try -avoid_negative_ts 1 as a possible workaround.\n",
738  av_ts2str(pkt->dts),
739  pkt->stream_index
740  );
741  }
742  } else {
743  av_assert2(pkt->dts == AV_NOPTS_VALUE || pkt->dts >= 0 || s->max_interleave_delta > 0);
744  if (pkt->dts != AV_NOPTS_VALUE && pkt->dts < 0) {
745  av_log(s, AV_LOG_WARNING,
746  "Packets poorly interleaved, failed to avoid negative "
747  "timestamp %s in stream %d.\n"
748  "Try -max_interleave_delta 0 as a possible workaround.\n",
749  av_ts2str(pkt->dts),
750  pkt->stream_index
751  );
752  }
753  }
754  }
755 
756  did_split = av_packet_split_side_data(pkt);
757 
758  if (!s->internal->header_written) {
759  ret = s->internal->write_header_ret ? s->internal->write_header_ret : write_header_internal(s);
760  if (ret < 0)
761  goto fail;
762  }
763 
764  if ((pkt->flags & AV_PKT_FLAG_UNCODED_FRAME)) {
765  AVFrame *frame = (AVFrame *)pkt->data;
766  av_assert0(pkt->size == UNCODED_FRAME_PACKET_SIZE);
767  ret = s->oformat->write_uncoded_frame(s, pkt->stream_index, &frame, 0);
768  av_frame_free(&frame);
769  } else {
770  ret = s->oformat->write_packet(s, pkt);
771  }
772 
773  if (s->pb && ret >= 0) {
774  if (s->flush_packets && s->flags & AVFMT_FLAG_FLUSH_PACKETS)
775  avio_flush(s->pb);
776  if (s->pb->error < 0)
777  ret = s->pb->error;
778  }
779 
780 fail:

782  av_packet_merge_side_data(pkt);
783

785  pkt->pts = pts_backup;
786  pkt->dts = dts_backup;
787  }
788 
789  return ret;
790 }

1038  *next_point = this_pktl;

1088  ++noninterleaved_count;
1089  }
1090  }

1093  flush = 1;

1096  s->internal->packet_buffer &&

1099  ) {

1109

1115  AV_TIME_BASE_Q);

1121  "Delay between the first packet and last packet in the "
1122  "muxing queue is %"PRId64" > %"PRId64": forcing output\n",

1124  flush = 1;
1125  }

1129  eof &&
1130  (s->flags & AVFMT_FLAG_SHORTEST) &&
1131  s->internal->shortest_end == AV_NOPTS_VALUE) {
1132  AVPacket *top_pkt = &s->internal->packet_buffer->pkt;

1135  s->streams[top_pkt->stream_index]->time_base,
1136  AV_TIME_BASE_Q);
