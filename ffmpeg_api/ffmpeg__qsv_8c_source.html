58 int qsv_init(AVCodecContext *s)
59 {
60  InputStream *ist = s->opaque;
61  AVHWFramesContext *frames_ctx;
62  AVQSVFramesContext *frames_hwctx;
63  int ret;
64 
65  if (!hw_device_ctx) {
66  ret = qsv_device_init(ist);
67  if (ret < 0)
68  return ret;
69  }
70 
71  av_buffer_unref(&ist->hw_frames_ctx);
72  ist->hw_frames_ctx = av_hwframe_ctx_alloc(hw_device_ctx);
73  if (!ist->hw_frames_ctx)
74  return AVERROR(ENOMEM);
75 
76  frames_ctx = (AVHWFramesContext*)ist->hw_frames_ctx->data;
77  frames_hwctx = frames_ctx->hwctx;
78 
79  frames_ctx->width = FFALIGN(s->coded_width, 32);
80  frames_ctx->height = FFALIGN(s->coded_height, 32);
81  frames_ctx->format = AV_PIX_FMT_QSV;
82  frames_ctx->sw_format = s->sw_pix_fmt;
83  frames_ctx->initial_pool_size = 64;
84  frames_hwctx->frame_type = MFX_MEMTYPE_VIDEO_MEMORY_DECODER_TARGET;
85 
86  ret = av_hwframe_ctx_init(ist->hw_frames_ctx);
87  if (ret < 0) {
88  av_log(NULL, AV_LOG_ERROR, "Error initializing a QSV frame pool\n");
89  return ret;
90  }
91 
92  ist->hwaccel_get_buffer = qsv_get_buffer;
93  ist->hwaccel_uninit = qsv_uninit;
94 
95  return 0;
96 }

98 int qsv_transcode_init(OutputStream *ost)
99 {
100  InputStream *ist;
101  const enum AVPixelFormat *pix_fmt;
102 
103  int err, i;
104  AVBufferRef *encode_frames_ref = NULL;
105  AVHWFramesContext *encode_frames;
106  AVQSVFramesContext *qsv_frames;
107 
108  /* check if the encoder supports QSV */
109  if (!ost->enc->pix_fmts)
110  return 0;
111  for (pix_fmt = ost->enc->pix_fmts; *pix_fmt != AV_PIX_FMT_NONE; pix_fmt++)
112  if (*pix_fmt == AV_PIX_FMT_QSV)
113  break;
114  if (*pix_fmt == AV_PIX_FMT_NONE)
115  return 0;
116 
117  if (strcmp(ost->avfilter, "null") || ost->source_index < 0)
118  return 0;
119 
120  /* check if the decoder supports QSV and the output only goes to this stream */
121  ist = input_streams[ost->source_index];
122  if (ist->hwaccel_id != HWACCEL_QSV || !ist->dec || !ist->dec->pix_fmts)
123  return 0;
124  for (pix_fmt = ist->dec->pix_fmts; *pix_fmt != AV_PIX_FMT_NONE; pix_fmt++)
125  if (*pix_fmt == AV_PIX_FMT_QSV)
126  break;
127  if (*pix_fmt == AV_PIX_FMT_NONE)
128  return 0;
129 
130  for (i = 0; i < nb_output_streams; i++)
131  if (output_streams[i] != ost &&
132  output_streams[i]->source_index == ost->source_index)
133  return 0;
134 
135  av_log(NULL, AV_LOG_VERBOSE, "Setting up QSV transcoding\n");
136 
137  if (!hw_device_ctx) {
138  err = qsv_device_init(ist);
139  if (err < 0)
140  goto fail;
141  }
142 
143  // This creates a dummy hw_frames_ctx for the encoder to be
144  // suitably initialised. It only contains one real frame, so
145  // hopefully doesn't waste too much memory.
146 
147  encode_frames_ref = av_hwframe_ctx_alloc(hw_device_ctx);
148  if (!encode_frames_ref) {
149  err = AVERROR(ENOMEM);
150  goto fail;
151  }
152  encode_frames = (AVHWFramesContext*)encode_frames_ref->data;
153  qsv_frames = encode_frames->hwctx;
154 
155  encode_frames->width = FFALIGN(ist->resample_width, 32);
156  encode_frames->height = FFALIGN(ist->resample_height, 32);
157  encode_frames->format = AV_PIX_FMT_QSV;
158  encode_frames->sw_format = AV_PIX_FMT_NV12;
159  encode_frames->initial_pool_size = 1;
160 
161  qsv_frames->frame_type = MFX_MEMTYPE_VIDEO_MEMORY_DECODER_TARGET;
162 
163  err = av_hwframe_ctx_init(encode_frames_ref);
164  if (err < 0)
165  goto fail;
166 
167  ist->dec_ctx->pix_fmt = AV_PIX_FMT_QSV;
168  ist->resample_pix_fmt = AV_PIX_FMT_QSV;
169 
170  ost->enc_ctx->pix_fmt = AV_PIX_FMT_QSV;
171  ost->enc_ctx->hw_frames_ctx = encode_frames_ref;
172 
173  return 0;
174 
175 fail:
176  av_buffer_unref(&encode_frames_ref);
177  return err;
178 }
