43 static int timed_line(const char *ptr)
44 {
45  char c;
46  int fs, fe;
47  return (sscanf(ptr, "%*u:%*u:%*u.%*u %*u:%*u:%*u.%*u %c", &c) == 1 ||
48  (sscanf(ptr, "@%u @%u %c", &fs, &fe, &c) == 3 && fs < fe));
49 }

51 static int jacosub_probe(AVProbeData *p)
52 {
53  const char *ptr = p->buf;
54  const char *ptr_end = p->buf + p->buf_size;
55 
56  if (AV_RB24(ptr) == 0xEFBBBF)
57  ptr += 3; /* skip UTF-8 BOM */
58 
59  while (ptr < ptr_end) {
60  while (jss_whitespace(*ptr))
61  ptr++;
62  if (*ptr != '#' && *ptr != '\n') {
63  if (timed_line(ptr))
64  return AVPROBE_SCORE_EXTENSION + 1;
65  return 0;
66  }
67  ptr += ff_subtitles_next_line(ptr);
68  }
69  return 0;
70 }

103 static const char *read_ts(JACOsubContext *jacosub, const char *buf,
104  int64_t *start, int64_t *duration)
105 {
106  int len;
107  unsigned hs, ms, ss, fs; // hours, minutes, seconds, frame start
108  unsigned he, me, se, fe; // hours, minutes, seconds, frame end
109  int ts_start, ts_end;
110 
111  /* timed format */
112  if (sscanf(buf, "%u:%u:%u.%u %u:%u:%u.%u %n",
113  &hs, &ms, &ss, &fs,
114  &he, &me, &se, &fe, &len) == 8) {
115  ts_start = (hs*3600 + ms*60 + ss) * jacosub->timeres + fs;
116  ts_end = (he*3600 + me*60 + se) * jacosub->timeres + fe;
117  goto shift_and_ret;
118  }
119 
120  /* timestamps format */
121  if (sscanf(buf, "@%u @%u %n", &ts_start, &ts_end, &len) == 2)
122  goto shift_and_ret;
123 
124  return NULL;
125 
126 shift_and_ret:
127  ts_start = (ts_start + jacosub->shift) * 100 / jacosub->timeres;
128  ts_end = (ts_end + jacosub->shift) * 100 / jacosub->timeres;
129  *start = ts_start;
130  *duration = ts_start + ts_end;
131  return buf + len;
132 }

134 static int get_shift(int timeres, const char *buf)
135 {
136  int sign = 1;
137  int a = 0, b = 0, c = 0, d = 0;
138 #define SSEP "%*1[.:]"
139  int n = sscanf(buf, "%d"SSEP"%d"SSEP"%d"SSEP"%d", &a, &b, &c, &d);
140 #undef SSEP
141 
142  if (*buf == '-' || a < 0) {
143  sign = -1;
144  a = FFABS(a);
145  }
146 
147  switch (n) {
148  case 4: return sign * ((a*3600 + b*60 + c) * timeres + d);
149  case 3: return sign * (( a*60 + b) * timeres + c);
150  case 2: return sign * (( a) * timeres + b);
151  }
152 
153  return 0;
154 }

156 static int jacosub_read_header(AVFormatContext *s)
157 {
158  AVBPrint header;
159  AVIOContext *pb = s->pb;
160  char line[JSS_MAX_LINESIZE];
161  JACOsubContext *jacosub = s->priv_data;
162  int shift_set = 0; // only the first shift matters
163  int merge_line = 0;
164  int i, ret;
165 
166  AVStream *st = avformat_new_stream(s, NULL);
167  if (!st)
168  return AVERROR(ENOMEM);
169  avpriv_set_pts_info(st, 64, 1, 100);
170  st->codecpar->codec_type = AVMEDIA_TYPE_SUBTITLE;
171  st->codecpar->codec_id = AV_CODEC_ID_JACOSUB;
172 
173  jacosub->timeres = 30;
174 
175  av_bprint_init(&header, 1024+AV_INPUT_BUFFER_PADDING_SIZE, 4096);
176 
177  while (!avio_feof(pb)) {
178  int cmd_len;
179  const char *p = line;
180  int64_t pos = avio_tell(pb);
181  int len = ff_get_line(pb, line, sizeof(line));
182 
183  p = jss_skip_whitespace(p);
184 
185  /* queue timed line */
186  if (merge_line || timed_line(p)) {
187  AVPacket *sub;
188 
189  sub = ff_subtitles_queue_insert(&jacosub->q, line, len, merge_line);
190  if (!sub)
191  return AVERROR(ENOMEM);
192  sub->pos = pos;
193  merge_line = len > 1 && !strcmp(&line[len - 2], "\\\n");
194  continue;
195  }
196 
197  /* skip all non-compiler commands and focus on the command */
198  if (*p != '#')
199  continue;
200  p++;
201  i = get_jss_cmd(p[0]);
202  if (i == -1)
203  continue;
204 
205  /* trim command + spaces */
206  cmd_len = strlen(cmds[i]);
207  if (av_strncasecmp(p, cmds[i], cmd_len) == 0)
208  p += cmd_len;
209  else
210  p++;
211  p = jss_skip_whitespace(p);
212 
213  /* handle commands which affect the whole script */
214  switch (cmds[i][0]) {
215  case 'S': // SHIFT command affect the whole script...
216  if (!shift_set) {
217  jacosub->shift = get_shift(jacosub->timeres, p);
218  shift_set = 1;
219  }
220  av_bprintf(&header, "#S %s", p);
221  break;
222  case 'T': // ...but must be placed after TIMERES
223  jacosub->timeres = strtol(p, NULL, 10);
224  if (!jacosub->timeres)
225  jacosub->timeres = 30;
226  else
227  av_bprintf(&header, "#T %s", p);
228  break;
229  }
230  }
231 
232  /* general/essential directives in the extradata */
233  ret = ff_bprint_to_codecpar_extradata(st->codecpar, &header);
234  if (ret < 0)
235  goto fail;
236 
237  /* SHIFT and TIMERES affect the whole script so packet timing can only be
238  * done in a second pass */
239  for (i = 0; i < jacosub->q.nb_subs; i++) {
240  AVPacket *sub = &jacosub->q.subs[i];
241  read_ts(jacosub, sub->data, &sub->pts, &sub->duration);
242  }
243  ff_subtitles_queue_finalize(s, &jacosub->q);
244 
245  return 0;
246 fail:
247  jacosub_read_close(s);
248  return ret;
249 }
