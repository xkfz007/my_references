36 static int pjs_probe(AVProbeData *p)
37 {
38  char c;
39  int64_t start, end;
40  const unsigned char *ptr = p->buf;
41 
42  if (sscanf(ptr, "%"SCNd64",%"SCNd64",%c", &start, &end, &c) == 3) {
43  size_t q1pos = strcspn(ptr, "\"");
44  size_t q2pos = q1pos + strcspn(ptr + q1pos + 1, "\"") + 1;
45  if (strcspn(ptr, "\r\n") > q2pos)
46  return AVPROBE_SCORE_MAX;
47  }
48  return 0;
49 }

51 static int64_t read_ts(char **line, int *duration)
52 {
53  int64_t start, end;
54 
55  if (sscanf(*line, "%"SCNd64",%"SCNd64, &start, &end) == 2) {
56  *line += strcspn(*line, "\"");
57  *line += !!**line;
58  *duration = end - start;
59  return start;
60  }
61  return AV_NOPTS_VALUE;
62 }

64 static int pjs_read_header(AVFormatContext *s)
65 {
66  PJSContext *pjs = s->priv_data;
67  AVStream *st = avformat_new_stream(s, NULL);
68  int res = 0;
69 
70  if (!st)
71  return AVERROR(ENOMEM);
72  avpriv_set_pts_info(st, 64, 1, 10);
73  st->codecpar->codec_type = AVMEDIA_TYPE_SUBTITLE;
74  st->codecpar->codec_id = AV_CODEC_ID_PJS;
75 
76  while (!avio_feof(s->pb)) {
77  char line[4096];
78  char *p = line;
79  const int64_t pos = avio_tell(s->pb);
80  int len = ff_get_line(s->pb, line, sizeof(line));
81  int64_t pts_start;
82  int duration;
83 
84  if (!len)
85  break;
86 
87  line[strcspn(line, "\r\n")] = 0;
88 
89  pts_start = read_ts(&p, &duration);
90  if (pts_start != AV_NOPTS_VALUE) {
91  AVPacket *sub;
92 
93  p[strcspn(p, "\"")] = 0;
94  sub = ff_subtitles_queue_insert(&pjs->q, p, strlen(p), 0);
95  if (!sub)
96  return AVERROR(ENOMEM);
97  sub->pos = pos;
98  sub->pts = pts_start;
99  sub->duration = duration;
100  }
101  }
102 
103  ff_subtitles_queue_finalize(s, &pjs->q);
104  return res;
105 }
