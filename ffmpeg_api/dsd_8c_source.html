30 static av_cold void dsd_ctables_tableinit(void)
31 {
32  int t, e, m, sign;
33  double acc[CTABLES];
34  for (e = 0; e < 256; ++e) {
35  memset(acc, 0, sizeof(acc));
36  for (m = 0; m < 8; ++m) {
37  sign = (((e >> (7 - m)) & 1) * 2 - 1);
38  for (t = 0; t < CTABLES; ++t)
39  acc[t] += sign * htaps[t * 8 + m];
40  }
41  for (t = 0; t < CTABLES; ++t)
42  ctables[CTABLES - 1 - t][e] = acc[t];
43  }
44 }
