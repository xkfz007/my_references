79 static void sd_1d53(int *p, int i0, int i1)
80 {
81  int i;
82 
83  if (i1 <= i0 + 1) {
84  if (i0 == 1)
85  p[1] <<= 1;
86  return;
87  }
88 
89  extend53(p, i0, i1);
90 
91  for (i = ((i0+1)>>1) - 1; i < (i1+1)>>1; i++)

94  p[2*i] += (p[2*i-1] + p[2*i+1] + 2) >> 2;

538  int decomp_levels, int type)
539 {

542

551  b[1][1] - b[1][0]);
