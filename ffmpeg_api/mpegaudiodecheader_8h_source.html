65 /* fast header check for resync */
66 static inline int ff_mpa_check_header(uint32_t header){
67  /* header */
68  if ((header & 0xffe00000) != 0xffe00000)
69  return -1;
70  /* layer check */
71  if ((header & (3<<17)) == 0)
72  return -1;
73  /* bit rate */
74  if ((header & (0xf<<12)) == 0xf<<12)
75  return -1;
76  /* frequency */
77  if ((header & (3<<10)) == 3<<10)
78  return -1;
79  return 0;
80 }
