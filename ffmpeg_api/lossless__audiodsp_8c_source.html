26 static int32_t scalarproduct_and_madd_int16_c(int16_t *v1, const int16_t *v2,
27  const int16_t *v3,
28  int order, int mul)
29 {
30  int res = 0;
31 
32  do {
33  res += *v1 * *v2++;
34  *v1++ += mul * *v3++;

37  } while (order-=2);
38  return res;
39 }
