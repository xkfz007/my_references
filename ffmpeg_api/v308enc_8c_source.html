37 static int v308_encode_frame(AVCodecContext *avctx, AVPacket *pkt,
38  const AVFrame *pic, int *got_packet)
39 {
40  uint8_t *dst;
41  uint8_t *y, *u, *v;
42  int i, j, ret;
43 
44  if ((ret = ff_alloc_packet2(avctx, pkt, avctx->width * avctx->height * 3, 0)) < 0)
45  return ret;
46  dst = pkt->data;
47 
48  y = pic->data[0];
49  u = pic->data[1];
50  v = pic->data[2];
51 
52  for (i = 0; i < avctx->height; i++) {
53  for (j = 0; j < avctx->width; j++) {
54  *dst++ = v[j];
55  *dst++ = y[j];
56  *dst++ = u[j];
57  }
58  y += pic->linesize[0];
59  u += pic->linesize[1];
60  v += pic->linesize[2];
61  }
62 
63  pkt->flags |= AV_PKT_FLAG_KEY;
64  *got_packet = 1;
65  return 0;
66 }
