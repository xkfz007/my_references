47 static void hex_dump_internal(void *avcl, FILE *f, int level,
48  const uint8_t *buf, int size)
49 {
50  int len, i, j, c;
51 
52  for (i = 0; i < size; i += 16) {
53  len = size - i;
54  if (len > 16)
55  len = 16;
56  HEXDUMP_PRINT("%08x ", i);
57  for (j = 0; j < 16; j++) {
58  if (j < len)
59  HEXDUMP_PRINT(" %02x", buf[i + j]);
60  else
61  HEXDUMP_PRINT(" ");
62  }
63  HEXDUMP_PRINT(" ");
64  for (j = 0; j < len; j++) {
65  c = buf[i + j];
66  if (c < ' ' || c > '~')
67  c = '.';
68  HEXDUMP_PRINT("%c", c);
69  }
70  HEXDUMP_PRINT("\n");
71  }
72 }

84 static void pkt_dump_internal(void *avcl, FILE *f, int level, const AVPacket *pkt,
85  int dump_payload, AVRational time_base)
86 {
87  HEXDUMP_PRINT("stream #%d:\n", pkt->stream_index);
88  HEXDUMP_PRINT(" keyframe=%d\n", (pkt->flags & AV_PKT_FLAG_KEY) != 0);
89  HEXDUMP_PRINT(" duration=%0.3f\n", pkt->duration * av_q2d(time_base));
90  /* DTS is _always_ valid after av_read_frame() */
91  HEXDUMP_PRINT(" dts=");
92  if (pkt->dts == AV_NOPTS_VALUE)
93  HEXDUMP_PRINT("N/A");
94  else
95  HEXDUMP_PRINT("%0.3f", pkt->dts * av_q2d(time_base));
96  /* PTS may not be known if B-frames are present. */
97  HEXDUMP_PRINT(" pts=");
98  if (pkt->pts == AV_NOPTS_VALUE)
99  HEXDUMP_PRINT("N/A");
100  else
101  HEXDUMP_PRINT("%0.3f", pkt->pts * av_q2d(time_base));
102  HEXDUMP_PRINT("\n");
103  HEXDUMP_PRINT(" size=%d\n", pkt->size);
104  if (dump_payload)
105  hex_dump_internal(avcl, f, level, pkt->data, pkt->size);
106 }

120 static void print_fps(double d, const char *postfix)
121 {
122  uint64_t v = lrintf(d * 100);
123  if (!v)
124  av_log(NULL, AV_LOG_INFO, "%1.4f %s", d, postfix);

133 static void dump_metadata(void *ctx, AVDictionary *m, const char *indent)
134 {
135  if (m && !(av_dict_count(m) == 1 && av_dict_get(m, "language", NULL, 0))) {
136  AVDictionaryEntry *tag = NULL;
137 
138  av_log(ctx, AV_LOG_INFO, "%sMetadata:\n", indent);
139  while ((tag = av_dict_get(m, "", tag, AV_DICT_IGNORE_SUFFIX)))
140  if (strcmp("language", tag->key)) {
141  const char *p = tag->value;
142  av_log(ctx, AV_LOG_INFO,
143  "%s %-16s: ", indent, tag->key);
144  while (*p) {
145  char tmp[256];
146  size_t len = strcspn(p, "\x8\xa\xb\xc\xd");
147  av_strlcpy(tmp, p, FFMIN(sizeof(tmp), len+1));
148  av_log(ctx, AV_LOG_INFO, "%s", tmp);
149  p += len;
150  if (*p == 0xd) av_log(ctx, AV_LOG_INFO, " ");
151  if (*p == 0xa) av_log(ctx, AV_LOG_INFO, "\n%s %-16s: ", indent, "");
152  if (*p) p++;
153  }
154  av_log(ctx, AV_LOG_INFO, "\n");
155  }
156  }

159 /* param change side data*/
160 static void dump_paramchange(void *ctx, AVPacketSideData *sd)
161 {
162  int size = sd->size;
163  const uint8_t *data = sd->data;
164  uint32_t flags, channels, sample_rate, width, height;
165  uint64_t layout;
166 
167  if (!data || sd->size < 4)
168  goto fail;
169 
170  flags = AV_RL32(data);
171  data += 4;
172  size -= 4;
173 
174  if (flags & AV_SIDE_DATA_PARAM_CHANGE_CHANNEL_COUNT) {
175  if (size < 4)
176  goto fail;
177  channels = AV_RL32(data);
178  data += 4;
179  size -= 4;
180  av_log(ctx, AV_LOG_INFO, "channel count %"PRIu32", ", channels);
181  }
182  if (flags & AV_SIDE_DATA_PARAM_CHANGE_CHANNEL_LAYOUT) {
183  if (size < 8)
184  goto fail;
185  layout = AV_RL64(data);
186  data += 8;
187  size -= 8;
188  av_log(ctx, AV_LOG_INFO,
189  "channel layout: %s, ", av_get_channel_name(layout));
190  }
191  if (flags & AV_SIDE_DATA_PARAM_CHANGE_SAMPLE_RATE) {
192  if (size < 4)
193  goto fail;
194  sample_rate = AV_RL32(data);
195  data += 4;
196  size -= 4;
197  av_log(ctx, AV_LOG_INFO, "sample_rate %"PRIu32", ", sample_rate);
198  }
199  if (flags & AV_SIDE_DATA_PARAM_CHANGE_DIMENSIONS) {
200  if (size < 8)
201  goto fail;
202  width = AV_RL32(data);
203  data += 4;
204  size -= 4;
205  height = AV_RL32(data);
206  data += 4;
207  size -= 4;
208  av_log(ctx, AV_LOG_INFO, "width %"PRIu32" height %"PRIu32, width, height);
209  }
210 
211  return;
212 fail:
213  av_log(ctx, AV_LOG_INFO, "unknown param");
214 }

346 static void dump_spherical(void *ctx, AVPacketSideData *sd)
347 {
348  AVSphericalMapping *spherical = (AVSphericalMapping *)sd->data;
349  double yaw, pitch, roll;
350 
351  if (sd->size < sizeof(*spherical)) {
352  av_log(ctx, AV_LOG_INFO, "invalid data");
353  return;
354  }
355 
356  if (spherical->projection == AV_SPHERICAL_EQUIRECTANGULAR)
357  av_log(ctx, AV_LOG_INFO, "equirectangular ");
358  else if (spherical->projection == AV_SPHERICAL_CUBEMAP)
359  av_log(ctx, AV_LOG_INFO, "cubemap ");
360  else {
361  av_log(ctx, AV_LOG_WARNING, "unknown");
362  return;
363  }
364 
365  yaw = ((double)spherical->yaw) / (1 << 16);
366  pitch = ((double)spherical->pitch) / (1 << 16);
367  roll = ((double)spherical->roll) / (1 << 16);
368  av_log(ctx, AV_LOG_INFO, "(%f/%f/%f) ", yaw, pitch, roll);
369 }

371 static void dump_sidedata(void *ctx, AVStream *st, const char *indent)
372 {
373  int i;
374 
375  if (st->nb_side_data)
376  av_log(ctx, AV_LOG_INFO, "%sSide data:\n", indent);
377 
378  for (i = 0; i < st->nb_side_data; i++) {
379  AVPacketSideData sd = st->side_data[i];
380  av_log(ctx, AV_LOG_INFO, "%s ", indent);
381 
382  switch (sd.type) {
383  case AV_PKT_DATA_PALETTE:
384  av_log(ctx, AV_LOG_INFO, "palette");
385  break;
386  case AV_PKT_DATA_NEW_EXTRADATA:
387  av_log(ctx, AV_LOG_INFO, "new extradata");
388  break;
389  case AV_PKT_DATA_PARAM_CHANGE:
390  av_log(ctx, AV_LOG_INFO, "paramchange: ");
391  dump_paramchange(ctx, &sd);
392  break;
393  case AV_PKT_DATA_H263_MB_INFO:
394  av_log(ctx, AV_LOG_INFO, "H.263 macroblock info");
395  break;
396  case AV_PKT_DATA_REPLAYGAIN:
397  av_log(ctx, AV_LOG_INFO, "replaygain: ");
398  dump_replaygain(ctx, &sd);
399  break;
400  case AV_PKT_DATA_DISPLAYMATRIX:
401  av_log(ctx, AV_LOG_INFO, "displaymatrix: rotation of %.2f degrees",
402  av_display_rotation_get((int32_t *)sd.data));
403  break;
404  case AV_PKT_DATA_STEREO3D:
405  av_log(ctx, AV_LOG_INFO, "stereo3d: ");
406  dump_stereo3d(ctx, &sd);
407  break;
408  case AV_PKT_DATA_AUDIO_SERVICE_TYPE:
409  av_log(ctx, AV_LOG_INFO, "audio service type: ");
410  dump_audioservicetype(ctx, &sd);
411  break;
412  case AV_PKT_DATA_QUALITY_STATS:
413  av_log(ctx, AV_LOG_INFO, "quality factor: %d, pict_type: %c", AV_RL32(sd.data), av_get_picture_type_char(sd.data[4]));
414  break;
415  case AV_PKT_DATA_CPB_PROPERTIES:
416  av_log(ctx, AV_LOG_INFO, "cpb: ");
417  dump_cpb(ctx, &sd);
418  break;
419  case AV_PKT_DATA_MASTERING_DISPLAY_METADATA:
420  dump_mastering_display_metadata(ctx, &sd);
421  break;
422  case AV_PKT_DATA_SPHERICAL:
423  av_log(ctx, AV_LOG_INFO, "spherical: ");
424  dump_spherical(ctx, &sd);
425  break;
426  default:
427  av_log(ctx, AV_LOG_INFO,
428  "unknown side data type %d (%d bytes)", sd.type, sd.size);
429  break;
430  }
431 
432  av_log(ctx, AV_LOG_INFO, "\n");
433  }

437 static void dump_stream_format(AVFormatContext *ic, int i,
438  int index, int is_output)
439 {
440  char buf[256];
441  int flags = (is_output ? ic->oformat->flags : ic->iformat->flags);
442  AVStream *st = ic->streams[i];
443  AVDictionaryEntry *lang = av_dict_get(st->metadata, "language", NULL, 0);
444  char *separator = ic->dump_separator;
445  AVCodecContext *avctx;
446  int ret;
447 
448  avctx = avcodec_alloc_context3(NULL);
449  if (!avctx)
450  return;
451 
452  ret = avcodec_parameters_to_context(avctx, st->codecpar);
453  if (ret < 0) {
454  avcodec_free_context(&avctx);
455  return;
456  }
457 
458  // Fields which are missing from AVCodecParameters need to be taken from the AVCodecContext
459  avctx->properties = st->codec->properties;
460  avctx->codec = st->codec->codec;
461  avctx->qmin = st->codec->qmin;
462  avctx->qmax = st->codec->qmax;
463  avctx->coded_width = st->codec->coded_width;
464  avctx->coded_height = st->codec->coded_height;
465 
466  if (separator)
467  av_opt_set(avctx, "dump_separator", separator, 0);
468  avcodec_string(buf, sizeof(buf), avctx, is_output);
469  avcodec_free_context(&avctx);
470 
471  av_log(NULL, AV_LOG_INFO, " Stream #%d:%d", index, i);
472 
473  /* the pid is an important information, so we display it */
474  /* XXX: add a generic system */
475  if (flags & AVFMT_SHOW_IDS)
476  av_log(NULL, AV_LOG_INFO, "[0x%x]", st->id);
477  if (lang)
478  av_log(NULL, AV_LOG_INFO, "(%s)", lang->value);
479  av_log(NULL, AV_LOG_DEBUG, ", %d, %d/%d", st->codec_info_nb_frames,
480  st->time_base.num, st->time_base.den);
481  av_log(NULL, AV_LOG_INFO, ": %s", buf);
482 
483  if (st->sample_aspect_ratio.num &&
484  av_cmp_q(st->sample_aspect_ratio, st->codecpar->sample_aspect_ratio)) {
485  AVRational display_aspect_ratio;
486  av_reduce(&display_aspect_ratio.num, &display_aspect_ratio.den,
487  st->codecpar->width * (int64_t)st->sample_aspect_ratio.num,
488  st->codecpar->height * (int64_t)st->sample_aspect_ratio.den,
489  1024 * 1024);
490  av_log(NULL, AV_LOG_INFO, ", SAR %d:%d DAR %d:%d",
491  st->sample_aspect_ratio.num, st->sample_aspect_ratio.den,
492  display_aspect_ratio.num, display_aspect_ratio.den);
493  }
494 
495  if (st->codecpar->codec_type == AVMEDIA_TYPE_VIDEO) {
496  int fps = st->avg_frame_rate.den && st->avg_frame_rate.num;
497  int tbr = st->r_frame_rate.den && st->r_frame_rate.num;
498  int tbn = st->time_base.den && st->time_base.num;
499  int tbc = st->codec->time_base.den && st->codec->time_base.num;
500 
501  if (fps || tbr || tbn || tbc)
502  av_log(NULL, AV_LOG_INFO, "%s", separator);
503 
504  if (fps)
505  print_fps(av_q2d(st->avg_frame_rate), tbr || tbn || tbc ? "fps, " : "fps");
506  if (tbr)
507  print_fps(av_q2d(st->r_frame_rate), tbn || tbc ? "tbr, " : "tbr");
508  if (tbn)
509  print_fps(1 / av_q2d(st->time_base), tbc ? "tbn, " : "tbn");
510  if (tbc)
511  print_fps(1 / av_q2d(st->codec->time_base), "tbc");
512  }
513 
514  if (st->disposition & AV_DISPOSITION_DEFAULT)
515  av_log(NULL, AV_LOG_INFO, " (default)");
516  if (st->disposition & AV_DISPOSITION_DUB)
517  av_log(NULL, AV_LOG_INFO, " (dub)");
518  if (st->disposition & AV_DISPOSITION_ORIGINAL)
519  av_log(NULL, AV_LOG_INFO, " (original)");
520  if (st->disposition & AV_DISPOSITION_COMMENT)
521  av_log(NULL, AV_LOG_INFO, " (comment)");
522  if (st->disposition & AV_DISPOSITION_LYRICS)
523  av_log(NULL, AV_LOG_INFO, " (lyrics)");
524  if (st->disposition & AV_DISPOSITION_KARAOKE)
525  av_log(NULL, AV_LOG_INFO, " (karaoke)");
526  if (st->disposition & AV_DISPOSITION_FORCED)
527  av_log(NULL, AV_LOG_INFO, " (forced)");
528  if (st->disposition & AV_DISPOSITION_HEARING_IMPAIRED)
529  av_log(NULL, AV_LOG_INFO, " (hearing impaired)");
530  if (st->disposition & AV_DISPOSITION_VISUAL_IMPAIRED)
531  av_log(NULL, AV_LOG_INFO, " (visual impaired)");
532  if (st->disposition & AV_DISPOSITION_CLEAN_EFFECTS)
533  av_log(NULL, AV_LOG_INFO, " (clean effects)");
534  av_log(NULL, AV_LOG_INFO, "\n");
535 
536  dump_metadata(NULL, st->metadata, " ");
537 
538  dump_sidedata(NULL, st, " ");
539 }

541 void av_dump_format(AVFormatContext *ic, int index,
542  const char *url, int is_output)
543 {
544  int i;
545  uint8_t *printed = ic->nb_streams ? av_mallocz(ic->nb_streams) : NULL;
546  if (ic->nb_streams && !printed)
547  return;
548

555 
556  if (!is_output) {
557  av_log(NULL, AV_LOG_INFO, " Duration: ");
