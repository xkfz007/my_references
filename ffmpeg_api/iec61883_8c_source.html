98 static int iec61883_callback(unsigned char *data, int length,
99  int complete, void *callback_data)
100 {
101  struct iec61883_data *dv = callback_data;
102  DVPacket *packet;
103  int ret;
104 
105 #if THREADS
106  pthread_mutex_lock(&dv->mutex);
107 #endif
108 
109  if (dv->packets >= dv->max_packets) {
110  av_log(NULL, AV_LOG_ERROR, "DV packet queue overrun, dropping.\n");
111  ret = 0;
112  goto exit;
113  }
114 
115  packet = av_mallocz(sizeof(*packet));
116  if (!packet) {
117  ret = -1;
118  goto exit;
119  }
120 
121  packet->buf = av_malloc(length);
122  if (!packet->buf) {
123  ret = -1;
124  goto exit;
125  }
126  packet->len = length;
127 
128  memcpy(packet->buf, data, length);
129 
130  if (dv->queue_first) {
131  dv->queue_last->next = packet;
132  dv->queue_last = packet;
133  } else {
134  dv->queue_first = packet;
135  dv->queue_last = packet;
136  }
137  dv->packets++;
138 
139  ret = 0;
140 
141 exit:
142 #if THREADS
143  pthread_cond_broadcast(&dv->cond);
144  pthread_mutex_unlock(&dv->mutex);
145 #endif
146  return ret;
147 }

149 static void *iec61883_receive_task(void *opaque)
150 {
151  struct iec61883_data *dv = (struct iec61883_data *)opaque;
152  int result;
153 
154 #if THREADS
155  while (dv->thread_loop)
156 #endif
157  {
158  while ((result = poll(&dv->raw1394_poll, 1, 200)) < 0) {
159  if (!(errno == EAGAIN || errno == EINTR)) {
160  av_log(NULL, AV_LOG_ERROR, "Raw1394 poll error occurred.\n");
161  dv->receive_error = AVERROR(EIO);
162  return NULL;
163  }
164  }
165  if (result > 0 && ((dv->raw1394_poll.revents & POLLIN)
166  || (dv->raw1394_poll.revents & POLLPRI))) {
167  dv->receiving = 1;
168  raw1394_loop_iterate(dv->raw1394);
169  } else if (dv->receiving) {
170  av_log(NULL, AV_LOG_ERROR, "No more input data available\n");
171 #if THREADS
172  pthread_mutex_lock(&dv->mutex);
173  dv->eof = 1;
174  pthread_cond_broadcast(&dv->cond);
175  pthread_mutex_unlock(&dv->mutex);
176 #else
177  dv->eof = 1;
178 #endif
179  return NULL;
180  }
181  }
182 
183  return NULL;
184 }

186 static int iec61883_parse_queue_dv(struct iec61883_data *dv, AVPacket *pkt)
187 {
188  DVPacket *packet;
189  int size;
190 
191  size = avpriv_dv_get_packet(dv->dv_demux, pkt);
192  if (size > 0)
193  return size;
194 
195  packet = dv->queue_first;
196  if (!packet)
197  return -1;
198 
199  size = avpriv_dv_produce_packet(dv->dv_demux, pkt,
200  packet->buf, packet->len, -1);
201  dv->queue_first = packet->next;
202  av_free(packet);
203  dv->packets--;
204 
205  if (size > 0)
206  return size;
207 
208  return -1;
209 }

232 static int iec61883_read_header(AVFormatContext *context)
233 {
234  struct iec61883_data *dv = context->priv_data;
235  struct raw1394_portinfo pinf[16];
236  rom1394_directory rom_dir;
237  char *endptr;
238  int inport;
239  int nb_ports;
240  int port = -1;
241  int response;
242  int i, j = 0;
243  uint64_t guid = 0;
244 
245  dv->input_port = -1;
246  dv->output_port = -1;
247  dv->channel = -1;
248 
249  dv->raw1394 = raw1394_new_handle();
250 
251  if (!dv->raw1394) {
252  av_log(context, AV_LOG_ERROR, "Failed to open IEEE1394 interface.\n");
253  return AVERROR(EIO);
254  }
255 
256  if ((nb_ports = raw1394_get_port_info(dv->raw1394, pinf, 16)) < 0) {
257  av_log(context, AV_LOG_ERROR, "Failed to get number of IEEE1394 ports.\n");
258  goto fail;
259  }
260 
261  inport = strtol(context->filename, &endptr, 10);
262  if (endptr != context->filename && *endptr == '\0') {
263  av_log(context, AV_LOG_INFO, "Selecting IEEE1394 port: %d\n", inport);
264  j = inport;
265  nb_ports = inport + 1;
266  } else if (strcmp(context->filename, "auto")) {
267  av_log(context, AV_LOG_ERROR, "Invalid input \"%s\", you should specify "
268  "\"auto\" for auto-detection, or the port number.\n", context->filename);
269  goto fail;
270  }
271 
272  if (dv->device_guid) {
273  if (sscanf(dv->device_guid, "%"SCNu64, &guid) != 1) {
274  av_log(context, AV_LOG_INFO, "Invalid dvguid parameter: %s\n",
275  dv->device_guid);
276  goto fail;
277  }
278  }
279 
280  for (; j < nb_ports && port==-1; ++j) {
281  raw1394_destroy_handle(dv->raw1394);
282 
283  if (!(dv->raw1394 = raw1394_new_handle_on_port(j))) {
284  av_log(context, AV_LOG_ERROR, "Failed setting IEEE1394 port.\n");
285  goto fail;
286  }
287 
288  for (i=0; i<raw1394_get_nodecount(dv->raw1394); ++i) {
289 
290  /* Select device explicitly by GUID */
291 
292  if (guid > 1) {
293  if (guid == rom1394_get_guid(dv->raw1394, i)) {
294  dv->node = i;
295  port = j;
296  break;
297  }
298  } else {
299 
300  /* Select first AV/C tape recorder player node */
301 
302  if (rom1394_get_directory(dv->raw1394, i, &rom_dir) < 0)
303  continue;
304  if (((rom1394_get_node_type(&rom_dir) == ROM1394_NODE_TYPE_AVC) &&
305  avc1394_check_subunit_type(dv->raw1394, i, AVC1394_SUBUNIT_TYPE_VCR)) ||
306  (rom_dir.unit_spec_id == MOTDCT_SPEC_ID)) {
307  rom1394_free_directory(&rom_dir);
308  dv->node = i;
309  port = j;
310  break;
311  }
312  rom1394_free_directory(&rom_dir);
313  }
314  }
315  }
316 
317  if (port == -1) {
318  av_log(context, AV_LOG_ERROR, "No AV/C devices found.\n");
319  goto fail;
320  }
321 
322  /* Provide bus sanity for multiple connections */
323 
324  iec61883_cmp_normalize_output(dv->raw1394, 0xffc0 | dv->node);
325 
326  /* Find out if device is DV or HDV */
327 
328  if (dv->type == IEC61883_AUTO) {
329  response = avc1394_transaction(dv->raw1394, dv->node,
330  AVC1394_CTYPE_STATUS |
331  AVC1394_SUBUNIT_TYPE_TAPE_RECORDER |
332  AVC1394_SUBUNIT_ID_0 |
333  AVC1394_VCR_COMMAND_OUTPUT_SIGNAL_MODE |
334  0xFF, 2);
335  response = AVC1394_GET_OPERAND0(response);
336  dv->type = (response == 0x10 || response == 0x90 || response == 0x1A || response == 0x9A) ?
337  IEC61883_HDV : IEC61883_DV;
338  }
339 
340  /* Connect to device, and do initialization */
341 
342  dv->channel = iec61883_cmp_connect(dv->raw1394, dv->node, &dv->output_port,
343  raw1394_get_local_id(dv->raw1394),
344  &dv->input_port, &dv->bandwidth);
345 
346  if (dv->channel < 0)
347  dv->channel = 63;
348 
349  if (!dv->max_packets)
350  dv->max_packets = 100;
351 
352  if (CONFIG_MPEGTS_DEMUXER && dv->type == IEC61883_HDV) {
353 
354  /* Init HDV receive */
355 
356  avformat_new_stream(context, NULL);
357 
358  dv->mpeg_demux = avpriv_mpegts_parse_open(context);
359  if (!dv->mpeg_demux)
360  goto fail;
361 
362  dv->parse_queue = iec61883_parse_queue_hdv;
363 
364  dv->iec61883_mpeg2 = iec61883_mpeg2_recv_init(dv->raw1394,
365  (iec61883_mpeg2_recv_t)iec61883_callback,
366  dv);
367 
368  dv->max_packets *= 766;
369  } else {
370 
371  /* Init DV receive */
372 
373  dv->dv_demux = avpriv_dv_init_demux(context);
374  if (!dv->dv_demux)
375  goto fail;
376 
377  dv->parse_queue = iec61883_parse_queue_dv;
378 
379  dv->iec61883_dv = iec61883_dv_fb_init(dv->raw1394, iec61883_callback, dv);
380  }
381 
382  dv->raw1394_poll.fd = raw1394_get_fd(dv->raw1394);
383  dv->raw1394_poll.events = POLLIN | POLLERR | POLLHUP | POLLPRI;
384 
385  /* Actually start receiving */
386 
387  if (dv->type == IEC61883_HDV)
388  iec61883_mpeg2_recv_start(dv->iec61883_mpeg2, dv->channel);
389  else
390  iec61883_dv_fb_start(dv->iec61883_dv, dv->channel);
391 
392 #if THREADS
393  dv->thread_loop = 1;
394  if (pthread_mutex_init(&dv->mutex, NULL))
395  goto fail;
396  if (pthread_cond_init(&dv->cond, NULL))
397  goto fail;
398  if (pthread_create(&dv->receive_task_thread, NULL, iec61883_receive_task, dv))
399  goto fail;
400 #endif
401 
402  return 0;
403 
404 fail:
405  raw1394_destroy_handle(dv->raw1394);
406  return AVERROR(EIO);
407 }

409 static int iec61883_read_packet(AVFormatContext *context, AVPacket *pkt)
410 {
411  struct iec61883_data *dv = context->priv_data;
412  int size;
413 
414  /**
415  * Try to parse frames from queue
416  */
417 
418 #if THREADS
419  pthread_mutex_lock(&dv->mutex);
420  while ((size = dv->parse_queue(dv, pkt)) == -1)
421  if (!dv->eof)
422  pthread_cond_wait(&dv->cond, &dv->mutex);
423  else
424  break;
425  pthread_mutex_unlock(&dv->mutex);
426 #else
427  int result;
428  while ((size = dv->parse_queue(dv, pkt)) == -1) {
429  iec61883_receive_task((void *)dv);
430  if (dv->receive_error)
431  return dv->receive_error;
432  }
433 #endif
434 
435  return size;
436 }

438 static int iec61883_close(AVFormatContext *context)
439 {
440  struct iec61883_data *dv = context->priv_data;
441 
442 #if THREADS
443  dv->thread_loop = 0;
444  pthread_join(dv->receive_task_thread, NULL);
445  pthread_cond_destroy(&dv->cond);
446  pthread_mutex_destroy(&dv->mutex);
447 #endif
448 
449  if (CONFIG_MPEGTS_DEMUXER && dv->type == IEC61883_HDV) {
450  iec61883_mpeg2_recv_stop(dv->iec61883_mpeg2);
451  iec61883_mpeg2_close(dv->iec61883_mpeg2);
452  avpriv_mpegts_parse_close(dv->mpeg_demux);
453  } else {
454  iec61883_dv_fb_stop(dv->iec61883_dv);
455  iec61883_dv_fb_close(dv->iec61883_dv);
456  }
457  while (dv->queue_first) {
458  DVPacket *packet = dv->queue_first;
459  dv->queue_first = packet->next;
460  av_freep(&packet->buf);
461  av_freep(&packet);
462  }
463 
464  iec61883_cmp_disconnect(dv->raw1394, dv->node, dv->output_port,
465  raw1394_get_local_id(dv->raw1394),
466  dv->input_port, dv->channel, dv->bandwidth);
467 
468  raw1394_destroy_handle(dv->raw1394);
469 
470  return 0;
471 }
