34 static void log_packet(const AVFormatContext *fmt_ctx, const AVPacket *pkt, const char *tag)
35 {
36  AVRational *time_base = &fmt_ctx->streams[pkt->stream_index]->time_base;
37 
38  printf("%s: pts:%s pts_time:%s dts:%s dts_time:%s duration:%s duration_time:%s stream_index:%d\n",
39  tag,
40  av_ts2str(pkt->pts), av_ts2timestr(pkt->pts, time_base),
41  av_ts2str(pkt->dts), av_ts2timestr(pkt->dts, time_base),
42  av_ts2str(pkt->duration), av_ts2timestr(pkt->duration, time_base),
43  pkt->stream_index);
44 }

46 int main(int argc, char **argv)
47 {
48  AVOutputFormat *ofmt = NULL;
49  AVFormatContext *ifmt_ctx = NULL, *ofmt_ctx = NULL;
50  AVPacket pkt;
51  const char *in_filename, *out_filename;
52  int ret, i;
53 
54  if (argc < 3) {
55  printf("usage: %s input output\n"
56  "API example program to remux a media file with libavformat and libavcodec.\n"
57  "The output format is guessed according to the file extension.\n"
58  "\n", argv[0]);
59  return 1;
60  }
61 
62  in_filename = argv[1];
63  out_filename = argv[2];
64 
65  av_register_all();
66 
67  if ((ret = avformat_open_input(&ifmt_ctx, in_filename, 0, 0)) < 0) {
68  fprintf(stderr, "Could not open input file '%s'", in_filename);
69  goto end;
70  }
71 
72  if ((ret = avformat_find_stream_info(ifmt_ctx, 0)) < 0) {
73  fprintf(stderr, "Failed to retrieve input stream information");
74  goto end;
75  }
76 
77  av_dump_format(ifmt_ctx, 0, in_filename, 0);
78 
79  avformat_alloc_output_context2(&ofmt_ctx, NULL, NULL, out_filename);
80  if (!ofmt_ctx) {
81  fprintf(stderr, "Could not create output context\n");
82  ret = AVERROR_UNKNOWN;
83  goto end;
84  }
85 
86  ofmt = ofmt_ctx->oformat;
87 
88  for (i = 0; i < ifmt_ctx->nb_streams; i++) {
89  AVStream *in_stream = ifmt_ctx->streams[i];
90  AVStream *out_stream = avformat_new_stream(ofmt_ctx, in_stream->codec->codec);
91  if (!out_stream) {
92  fprintf(stderr, "Failed allocating output stream\n");
93  ret = AVERROR_UNKNOWN;
94  goto end;
95  }
96 
97  ret = avcodec_copy_context(out_stream->codec, in_stream->codec);
98  if (ret < 0) {
99  fprintf(stderr, "Failed to copy context from input to output stream codec context\n");
100  goto end;
101  }
102  out_stream->codec->codec_tag = 0;
103  if (ofmt_ctx->oformat->flags & AVFMT_GLOBALHEADER)
104  out_stream->codec->flags |= AV_CODEC_FLAG_GLOBAL_HEADER;
105  }
106  av_dump_format(ofmt_ctx, 0, out_filename, 1);
107 
108  if (!(ofmt->flags & AVFMT_NOFILE)) {
109  ret = avio_open(&ofmt_ctx->pb, out_filename, AVIO_FLAG_WRITE);
110  if (ret < 0) {
111  fprintf(stderr, "Could not open output file '%s'", out_filename);
112  goto end;
113  }
114  }
115 
116  ret = avformat_write_header(ofmt_ctx, NULL);
117  if (ret < 0) {
118  fprintf(stderr, "Error occurred when opening output file\n");
119  goto end;
120  }
121 
122  while (1) {
123  AVStream *in_stream, *out_stream;
124 
125  ret = av_read_frame(ifmt_ctx, &pkt);
126  if (ret < 0)
127  break;
128 
129  in_stream = ifmt_ctx->streams[pkt.stream_index];
130  out_stream = ofmt_ctx->streams[pkt.stream_index];
131 
132  log_packet(ifmt_ctx, &pkt, "in");
133 
134  /* copy packet */
135  pkt.pts = av_rescale_q_rnd(pkt.pts, in_stream->time_base, out_stream->time_base, AV_ROUND_NEAR_INF|AV_ROUND_PASS_MINMAX);
136  pkt.dts = av_rescale_q_rnd(pkt.dts, in_stream->time_base, out_stream->time_base, AV_ROUND_NEAR_INF|AV_ROUND_PASS_MINMAX);
137  pkt.duration = av_rescale_q(pkt.duration, in_stream->time_base, out_stream->time_base);
138  pkt.pos = -1;
139  log_packet(ofmt_ctx, &pkt, "out");
140 
141  ret = av_interleaved_write_frame(ofmt_ctx, &pkt);
142  if (ret < 0) {
143  fprintf(stderr, "Error muxing packet\n");
144  break;
145  }
146  av_packet_unref(&pkt);
147  }
148 
149  av_write_trailer(ofmt_ctx);
150 end:
151 
152  avformat_close_input(&ifmt_ctx);
153 
154  /* close output */
155  if (ofmt_ctx && !(ofmt->flags & AVFMT_NOFILE))
156  avio_closep(&ofmt_ctx->pb);
157  avformat_free_context(ofmt_ctx);
158 
159  if (ret < 0 && ret != AVERROR_EOF) {
160  fprintf(stderr, "Error occurred: %s\n", av_err2str(ret));
161  return 1;
162  }
163 
164  return 0;
165 }
