250 static inline void set_mvs(cavs_vector *mv, enum cavs_block size) {
251  switch(size) {
252  case BLK_16X16:
253  mv[MV_STRIDE ] = mv[0];
254  mv[MV_STRIDE+1] = mv[0];
255  case BLK_16X8:
256  mv[1] = mv[0];
257  break;
258  case BLK_8X16:
259  mv[MV_STRIDE] = mv[0];
260  break;
261  }
262 }
