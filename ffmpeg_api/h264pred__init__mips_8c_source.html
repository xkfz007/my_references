36

38  if (chroma_format_idc == 1) {
39  h->pred8x8[PLANE_PRED8x8] = ff_h264_intra_predict_plane_8x8_msa;
40  }
41  }
42  if (codec_id != AV_CODEC_ID_RV40 && codec_id != AV_CODEC_ID_VP7
43  && codec_id != AV_CODEC_ID_VP8) {
44  if (chroma_format_idc == 1) {
45  h->pred8x8[DC_PRED8x8] = ff_h264_intra_predict_dc_4blk_8x8_msa;

49  ff_h264_intra_predict_vert_dc_8x8_msa;
50  h->pred8x8[ALZHEIMER_DC_L0T_PRED8x8] =
51  ff_h264_intra_predict_mad_cow_dc_l0t_8x8_msa;
52  h->pred8x8[ALZHEIMER_DC_0LT_PRED8x8] =
53  ff_h264_intra_predict_mad_cow_dc_0lt_8x8_msa;

55  ff_h264_intra_predict_mad_cow_dc_l00_8x8_msa;

57  ff_h264_intra_predict_mad_cow_dc_0l0_8x8_msa;
58  }
59  } else {
60  if (codec_id == AV_CODEC_ID_VP7 || codec_id == AV_CODEC_ID_VP8) {
61  h->pred8x8[7] = ff_vp8_pred8x8_127_dc_8_msa;

64  }
65

68  }
69 
70  h->pred16x16[DC_PRED8x8] = ff_h264_intra_pred_dc_16x16_msa;
71  h->pred16x16[VERT_PRED8x8] = ff_h264_intra_pred_vert_16x16_msa;
72  h->pred16x16[HOR_PRED8x8] = ff_h264_intra_pred_horiz_16x16_msa;
73

76  ;
77  break;
78  case AV_CODEC_ID_RV40:
79  ;
80  break;
81  case AV_CODEC_ID_VP7:
82  case AV_CODEC_ID_VP8:

84  h->pred16x16[8] = ff_vp8_pred16x16_129_dc_8_msa;

94  h->pred16x16[DC_128_PRED8x8] = ff_h264_intra_pred_dc_128_16x16_msa;
95  }

100 static av_cold void h264_pred_init_mmi(H264PredContext *h, int codec_id,
101  const int bit_depth, const int chroma_format_idc)

107  } else {

110  }

114  h->pred16x16[HOR_PRED8x8 ] = ff_pred16x16_horizontal_8_mmi;
115  h->pred8x8l [TOP_DC_PRED ] = ff_pred8x8l_top_dc_8_mmi;
116  h->pred8x8l [DC_PRED ] = ff_pred8x8l_dc_8_mmi;
117 
118 #if ARCH_MIPS64
119  switch (codec_id) {
120  case AV_CODEC_ID_SVQ3:
121  h->pred16x16[PLANE_PRED8x8 ] = ff_pred16x16_plane_svq3_8_mmi;
122  break;
123  case AV_CODEC_ID_RV40:
124  h->pred16x16[PLANE_PRED8x8 ] = ff_pred16x16_plane_rv40_8_mmi;
125  break;
126  case AV_CODEC_ID_VP7:
127  case AV_CODEC_ID_VP8:
128  break;
129  default:
130  h->pred16x16[PLANE_PRED8x8 ] = ff_pred16x16_plane_h264_8_mmi;
131  break;
132  }
133 #endif
134 
135  if (codec_id == AV_CODEC_ID_SVQ3 || codec_id == AV_CODEC_ID_H264) {
136  if (chroma_format_idc == 1) {
137  h->pred8x8[TOP_DC_PRED8x8 ] = ff_pred8x8_top_dc_8_mmi;
138  h->pred8x8[DC_PRED8x8 ] = ff_pred8x8_dc_8_mmi;
139  }
140  }
141  }
142 }
143 #endif /* HAVE_MMI */
