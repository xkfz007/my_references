56 static av_always_inline av_const long int lrintf_mips(float x)
57 {
58  register int ret_int;
59 
60  __asm__ volatile (
61  "cvt.w.s %[x], %[x] \n\t"
62  "mfc1 %[ret_int], %[x] \n\t"
63 
64  :[x]"+f"(x), [ret_int]"=r"(ret_int)
65  );
66  return ret_int;
67 }
