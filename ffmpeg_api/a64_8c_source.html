27 static int a64_write_header(AVFormatContext *s)
28 {
29  AVCodecParameters *par = s->streams[0]->codecpar;
30  uint8_t header[5] = {
31  0x00, //load
32  0x40, //address
33  0x00, //mode
34  0x00, //charset_lifetime (multi only)
35  0x00 //fps in 50/fps;
36  };
37 
38  if (par->extradata_size < 4) {
39  av_log(s, AV_LOG_ERROR, "Missing extradata\n");
40  return AVERROR_INVALIDDATA;
41  }
42 
43  switch (par->codec_id) {
44  case AV_CODEC_ID_A64_MULTI:
45  header[2] = 0x00;
46  header[3] = AV_RB32(par->extradata+0);
47  header[4] = 2;
48  break;
49  case AV_CODEC_ID_A64_MULTI5:
50  header[2] = 0x01;
51  header[3] = AV_RB32(par->extradata+0);
52  header[4] = 3;
53  break;
54  default:
55  return AVERROR_INVALIDDATA;
56  }
57  avio_write(s->pb, header, 2);
58  return 0;
59 }
