40  const uint8_t **poutbuf, int *poutbuf_size,

42 {

44  ParseContext *pc = &s->pc;
45  int next;
46 
47  if (!s->block_size) {
48  av_assert1(avctx->codec_id == AV_CODEC_ID_G729);
49  /* FIXME: replace this heuristic block_size with more precise estimate */
50  s->block_size = (avctx->bit_rate < 8000) ? G729D_6K4_BLOCK_SIZE : G729_8K_BLOCK_SIZE;
51  s->duration = avctx->frame_size;
52  }
53 
54  if (!s->remaining)
55  s->remaining = s->block_size;
56  if (s->remaining <= buf_size) {
57  next = s->remaining;
58  s->remaining = 0;
59  } else {
60  next = END_NOT_FOUND;
61  s->remaining -= buf_size;
62  }
63 
64  if (ff_combine_frame(pc, next, &buf, &buf_size) < 0 || !buf_size) {
65  *poutbuf = NULL;
66  *poutbuf_size = 0;
67  return buf_size;
68  }
69 
70  s1->duration = s->duration;
71 
72  *poutbuf = buf;
73  *poutbuf_size = buf_size;
74  return next;
75 }
