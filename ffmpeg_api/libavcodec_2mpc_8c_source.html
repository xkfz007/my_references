62  int channels)
63 {

69  /* dequantize */
70  memset(c->sb_samples, 0, sizeof(c->sb_samples));
71  off = 0;
72  for(i = 0; i <= maxband; i++, off += SAMPLES_PER_BAND){
73  for(ch = 0; ch < 2; ch++){

79  mul = (mpc_CC+1)[bands[i].res[ch]] * mpc_SCF[bands[i].scf_idx[ch][1] & 0xFF];

83  for(; j < 36; j++)
